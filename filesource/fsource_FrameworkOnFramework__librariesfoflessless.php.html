  266.     /**
  267.      * Recursively compiles a block.
  268.      *
  269.      * A block is analogous to a CSS block in most cases. A single LESS document
  270.      * is encapsulated in a block when parsed, but it does not have parent tags
  271.      * so all of it's children appear on the root level when compiled.

  273.      * Blocks are made up of props and children.

  275.      * Props are property instructions, array tuples which describe an action
  276.      * to be taken, eg. write a property, set a variable, mixin a block.
  277.      *
  278.      * The children of a block are just all the blocks that are defined within.
  279.      * This is used to look up mixins when performing a mixin.
  280.      *
  281.      * Compiling the block involves pushing a fresh environment on the stack,
  282.      * and iterating through the props, compiling each one.
  283.      *

  291.     {
  292.         switch ($block->type)
  293.         {

  297.             case null:
  298.                 $this->compileCSSBlock($block);
  299.                 break;
  300.             case "media":
  301.                 $this->compileMedia($block);
  302.                 break;
  303.             case "directive":

  305.  
  306.                 if (!empty($block->value))
  307.                 {
  308.                     $name .= " " . $this->compileValue($this->reduce($block->value));
  309.                 }
  310.  
  311.                 $this->compileNestedBlock($block, array($name));
  312.                 break;
  313.             default:
  314.                 $this->throwError("unknown block type: $block->type\n");
  315.         }
  316.     }

  734.     /**

  738.      * @param   type  $callingArgs  X
  739.      *
  740.      * @return  boolean 
  741.      */

  743.     {
  744.         /**

  759.                     $negate = false;

  761.                     if ($guard[0] == "negate")

  823.                     {
  824.                         return false;

  834.         if ($block->isVararg)
  835.         {
  836.             // Not having enough is handled above
  837.             return true;

  841.             $numMatched = $i + 1;

  848.     /**
  849.      * Pattern match all
  850.      *

  871.     /**

  939.     }
  940.  
  941.     /**

  992.     /**

  1037.                 {

  1039.

  1050.                     {
  1051.                         $haveArgs = true;

  1066.                             && $subProp[0] == "assign"

  1069.                         {
  1070.                             $subProp[2] = array(
  1071.                                 'list', ' ',
  1072.                                 array($subProp[2], array('keyword', $suffix))
  1073.                             );
  1074.                         }
  1075.  
  1076.                         $this->compileProp($subProp, $mixin, $out);
