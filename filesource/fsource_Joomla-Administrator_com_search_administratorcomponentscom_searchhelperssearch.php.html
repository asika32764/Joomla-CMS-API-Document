  53.     public static function santiseSearchWord(&$searchword, $searchphrase)

  55.         $ignored = false;
  56.  
  57.         $lang          = JFactory::getLanguage();

  71.  
  72.         // First case is single ignored word
  73.         if (count($aterms) == 1 && in_array(JString::strtolower($searchword), $search_ignore))
  74.         {

  77.  
  78.         // Filter out search terms that are too small

  80.  
  81.         foreach ($aterms as $aterm)
  82.         {
  83.             if (JString::strlen($aterm) < $lower_limit)
  84.             {
  85.                 $search_ignore[] = $aterm;
  86.             }

  88.  
  89.         // Next is to remove ignored words from type 'all' or 'any' (not exact) searches with multiple words
  90.         if (count($aterms) > 1 && $searchphrase != 'exact')
  91.         {
  92.             $pruned     = array_diff($aterms, $search_ignore);
  93.             $searchword = implode(' ', $pruned);

  168.     /**
  169.      * Checks an object for search terms (after stripping fields of HTML)
  170.      *
  171.      * @param   object  $object      The object to check
  172.      * @param   string  $searchTerm  Search words to check for
  173.      * @param   array   $fields      List of object variables to check against
  174.      *
  175.      * @return  boolean True if searchTerm is in object, false otherwise
  176.      */
  177.     public static function checkNoHtml($object, $searchTerm, $fields)
  178.     {
  179.         $searchRegex = array(
  180.             '#<script[^>]*>.*?</script>#si',
  181.             '#<style[^>]*>.*?</style>#si',
  182.             '#<!.*?(--|]])>#si',

  184.         );
  185.         $terms = explode(' ', $searchTerm);
  186.  
  187.         if (empty($fields))
  188.         {
  189.             return false;
  190.         }
  191.  
  192.         foreach ($fields as $field)
  193.         {
  194.             if (!isset($object->$field))
  195.             {
  196.                 continue;
  197.             }
  198.  
  199.             $text = self::remove_accents($object->$field);
  200.  
  201.             foreach ($searchRegex as $regex)
  202.             {
  203.                 $text = preg_replace($regex, '', $text);
  204.             }
  205.  
  206.             foreach ($terms as $term)
  207.             {
  208.                 $term = self::remove_accents($term);
  209.  
  210.                 if (JString::stristr($text, $term) !== false)
  211.                 {
  212.                     return true;
  213.                 }
  214.             }
  215.         }
  216.  
  217.         return false;
  218.     }

  220.     /**

  239.      * returns substring of characters around a searchword

  242.      * @param   integer  $searchword  Number of chars to return

  244.      * @return  string 
  245.      *

  248.     public static function _smartSubstr($text, $searchword)
  249.     {
  250.         $lang        = JFactory::getLanguage();
  251.         $length      = $lang->getSearchDisplayedCharactersNumber();
  252.         $ltext       = self::remove_accents($text);
  253.         $textlen     = JString::strlen($ltext);
  254.         $lsearchword = JString::strtolower(self::remove_accents($searchword));
  255.         $wordfound   = false;
  256.         $pos         = 0;
  257.  
  258.         while ($wordfound === false && $pos < $textlen)
  259.         {
  260.             if (($wordpos = @JString::strpos($ltext, ' ', $pos + $length)) !== false)
  261.             {
  262.                 $chunk_size = $wordpos - $pos;
  263.             }
  264.             else
  265.             {
  266.                 $chunk_size = $length;
  267.             }
  268.  
  269.             $chunk     = JString::substr($ltext, $pos, $chunk_size);
  270.             $wordfound = JString::strpos(JString::strtolower($chunk), $lsearchword);
  271.  
  272.             if ($wordfound === false)
  273.             {
  274.                 $pos += $chunk_size + 1;
  275.             }
  276.         }
  277.  
  278.         if ($wordfound !== false)
  279.         {
  280.             return (($pos > 0) ? '...&#160;' : '') . JString::substr($text, $pos, $chunk_size) . '&#160;...';
  281.         }
  282.         else
  283.         {
  284.             if (($wordpos = @JString::strpos($text, ' ', $length)) !== false)
  285.             {
  286.                 return JString::substr($text, 0, $wordpos) . '&#160;...';
  287.             }
  288.             else
  289.             {
  290.                 return JString::substr($text, 0, $length);
  291.             }
  292.         }
