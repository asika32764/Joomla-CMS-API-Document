  77.         // Set the extensions name

  86.         // Get the component description

  88.

  90.         {
  91.             $this->parent->set('message', JText::_($description));

  93.         else
  94.         {
  95.             $this->parent->set('message', '');
  96.         }
  97.  
  98.         // Set the installation path
  99.         $files = $this->manifest->files;
  100.         $group = (string) $this->manifest->packagename;
  101.  
  102.         if (!empty($group))
  103.         {
  104.             $this->parent->setPath('extension_root', JPATH_MANIFESTS . '/packages/' . implode(DIRECTORY_SEPARATOR, explode('/', $group)));
  105.         }
  106.         else
  107.         {
  108.             $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_PACK_INSTALL_NO_PACK', JText::_('JLIB_INSTALLER_' . strtoupper($this->route))));

  112.  
  113.         /*

  118.         if (file_exists(JPATH_MANIFESTS . '/packages/' . basename($this->parent->getPath('manifest'))))
  119.         {
  120.             // Look for an update function or update tag
  121.             $updateElement = $this->manifest->update;
  122.  
  123.             // If $this->upgrade has already been set, or an update property exists in the manifest, update the extensions
  124.             if ($this->parent->isUpgrade() || $updateElement)
  125.             {
  126.                 // Use the update route for all packaged extensions
  127.                 $this->route = 'update';
  128.             }
  129.         }
  130.  
  131.         /**
  132.          * ---------------------------------------------------------------------------------------------
  133.          * Installer Trigger Loading
  134.          * ---------------------------------------------------------------------------------------------
  135.          */
  136.  
  137.         // If there is an manifest class file, lets load it; we'll copy it later (don't have dest yet)
  138.         $this->scriptElement = $this->manifest->scriptfile;
  139.         $manifestScript = (string) $this->manifest->scriptfile;
  140.

  142.         {
  143.             $manifestScriptFile = $this->parent->getPath('source') . '/' . $manifestScript;
  144.  
  145.             if (is_file($manifestScriptFile))
  146.             {
  147.                 // Load the file
  148.                 include_once $manifestScriptFile;
  149.             }
  150.  
  151.             // Set the class name
  152.             $classname = $element . 'InstallerScript';
  153.  
  154.             if (class_exists($classname))
  155.             {
  156.                 // Create a new instance
  157.                 $this->parent->manifestClass = new $classname($this);
  158.  
  159.                 // And set this so we can copy it later
  160.                 $this->set('manifest_script', $manifestScript);
  161.             }
  162.         }
  163.  
  164.         // Run preflight if possible (since we know we're not an update)
  165.         ob_start();
  166.         ob_implicit_flush(false);
  167.  
  168.         if ($this->parent->manifestClass && method_exists($this->parent->manifestClass, 'preflight'))
  169.         {
  170.             if ($this->parent->manifestClass->preflight($this->route, $this) === false)
  171.             {
  172.                 // Preflight failed, rollback changes
  173.                 $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_PACKAGE_INSTALL_CUSTOM_INSTALL_FAILURE'));
  174.  
  175.                 return false;
  176.             }
  177.         }
  178.  
  179.         // Create msg object; first use here
  180.         $msg = ob_get_contents();
  181.         ob_end_clean();
  182.  
  183.         /*
  184.          * ---------------------------------------------------------------------------------------------
  185.          * Filesystem Processing Section
  186.          * ---------------------------------------------------------------------------------------------
  187.          */
  188.  
  189.         if ($folder = $files->attributes()->folder)
  190.         {
  191.             $source = $this->parent->getPath('source') . '/' . $folder;
  192.         }
  193.         else
  194.         {
  195.             $source = $this->parent->getPath('source');
  196.         }
  197.  
  198.         // Install all necessary files
  199.         if (count($this->manifest->files->children()))
  200.         {
  201.             $i = 0;
  202.  
  203.             foreach ($this->manifest->files->children() as $child)
  204.             {
  205.                 $file = $source . '/' . $child;
  206.  
  207.                 if (is_dir($file))
  208.                 {
  209.                     // If it's actually a directory then fill it up
  210.                     $package = array();
  211.                     $package['dir'] = $file;
  212.                     $package['type'] = JInstallerHelper::detectType($file);
  213.                 }
  214.                 else
  215.                 {
  216.                     // If it's an archive
  217.                     $package = JInstallerHelper::unpack($file);
  218.                 }
  219.                 $tmpInstaller = new JInstaller;
  220.                 $installResult = $tmpInstaller->{$this->route}($package['dir']);
  221.  
  222.                 if (!$installResult)
  223.                 {
  224.                     $this->parent->abort(
  225.                         JText::sprintf(
  226.                             'JLIB_INSTALLER_ABORT_PACK_INSTALL_ERROR_EXTENSION', JText::_('JLIB_INSTALLER_' . strtoupper($this->route)),
  227.                             basename($file)
  228.                         )
  229.                     );
  230.  
  231.                     return false;
  232.                 }
  233.                 else
  234.                 {
  235.                     $results[$i] = array(
  236.                         'name' => $tmpInstaller->manifest->name,
  237.                         'result' => $installResult
  238.                     );
  239.                 }
  240.                 $i++;
  241.             }
  242.         }
  243.         else
  244.         {
  245.             $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_PACK_INSTALL_NO_FILES', JText::_('JLIB_INSTALLER_' . strtoupper($this->route))));
  246.  
  247.             return false;
  248.         }
  249.  
  250.         // Parse optional tags
  251.         $this->parent->parseLanguages($this->manifest->languages);
  252.  
  253.         /*
  254.          * ---------------------------------------------------------------------------------------------
  255.          * Extension Registration
  256.          * ---------------------------------------------------------------------------------------------
  257.          */
  258.  
  259.         $row = JTable::getInstance('extension');
  260.         $eid = $row->find(array('element' => strtolower($this->get('element')), 'type' => 'package'));
  261.  
  262.         if ($eid)
  263.         {
  264.             $row->load($eid);
  265.         }
  266.         else
  267.         {
  268.             $row->name = $this->get('name');
  269.             $row->type = 'package';
  270.             $row->element = $this->get('element');
  271.  
  272.             // There is no folder for modules
  273.             $row->folder = '';
  274.             $row->enabled = 1;
  275.             $row->protected = 0;
  276.             $row->access = 1;
  277.             $row->client_id = 0;
  278.  
  279.             // Custom data
  280.             $row->custom_data = '';
  281.             $row->params = $this->parent->getParams();
  282.         }
  283.         // Update the manifest cache for the entry
  284.         $row->manifest_cache = $this->parent->generateManifestCache();
  285.  
  286.         if (!$row->store())
  287.         {
  288.             // Install failed, roll back changes
  289.             $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_PACK_INSTALL_ROLLBACK', $row->getError()));
  290.  
  291.             return false;
  292.         }
  293.  
  294.         /*
  295.          * ---------------------------------------------------------------------------------------------
  296.          * Finalization and Cleanup Section
  297.          * ---------------------------------------------------------------------------------------------
  298.          */
  299.  
  300.         // Run the custom method based on the route
  301.         ob_start();
  302.         ob_implicit_flush(false);
  303.  
  304.         if ($this->parent->manifestClass && method_exists($this->parent->manifestClass, $this->route))
  305.         {
  306.             if ($this->parent->manifestClass->{$this->route}($this) === false)
  307.             {
  308.                 // Install failed, rollback changes
  309.                 $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_FILE_INSTALL_CUSTOM_INSTALL_FAILURE'));
  310.  
  311.                 return false;
  312.             }
  313.         }
  314.  
  315.         // Append messages
  316.         $msg .= ob_get_contents();
  317.         ob_end_clean();
  318.  
  319.         // Lastly, we will copy the manifest file to its appropriate place.
  320.         $manifest = array();
  321.         $manifest['src'] = $this->parent->getPath('manifest');
  322.         $manifest['dest'] = JPATH_MANIFESTS . '/packages/' . basename($this->parent->getPath('manifest'));
  323.  
  324.         if (!$this->parent->copyFiles(array($manifest), true))
  325.         {
  326.             // Install failed, rollback changes
  327.             $this->parent->abort(
  328.                 JText::sprintf('JLIB_INSTALLER_ABORT_PACK_INSTALL_COPY_SETUP', JText::_('JLIB_INSTALLER_ABORT_PACK_INSTALL_NO_FILES'))
  329.             );
  330.  
  331.             return false;
  332.         }
  333.  
  334.         // If there is a manifest script, let's copy it.
  335.         if ($this->get('manifest_script'))
  336.         {
  337.             // First, we have to create a folder for the script if one isn't present
  338.             if (!file_exists($this->parent->getPath('extension_root')))
  339.             {
  340.                 JFolder::create($this->parent->getPath('extension_root'));
  341.             }
  342.  
  343.             $path['src'] = $this->parent->getPath('source') . '/' . $this->get('manifest_script');
  344.             $path['dest'] = $this->parent->getPath('extension_root') . '/' . $this->get('manifest_script');
  345.  
  346.             if (!file_exists($path['dest']) || $this->parent->isOverwrite())
  347.             {
  348.                 if (!$this->parent->copyFiles(array($path)))
  349.                 {
  350.                     // Install failed, rollback changes
  351.                     $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_PACKAGE_INSTALL_MANIFEST'));
  352.  
  353.                     return false;
  354.                 }
  355.             }
  356.         }
  357.  
  358.         // And now we run the postflight
  359.         ob_start();
  360.         ob_implicit_flush(false);
  361.  
  362.         if ($this->parent->manifestClass && method_exists($this->parent->manifestClass, 'postflight'))
  363.         {
  364.             $this->parent->manifestClass->postflight($this->route, $this, $results);
  365.         }
  366.  
  367.         // Append messages
  368.         $msg .= ob_get_contents();
  369.         ob_end_clean();
  370.  
  371.         if ($msg != '')
  372.         {
  373.             $this->parent->set('extension_message', $msg);
  374.         }
  375.         return $row->extension_id;
  376.     }

  378.     /**
  379.      * Updates a package

  394.     /**
  395.      * Custom uninstall method
  396.      *
  397.      * @param   integer  $id  The id of the package to uninstall.
  398.      *
  399.      * @return  boolean  True on success
  400.      *
  401.      * @since   3.1
  402.      */
  403.     public function uninstall($id)
  404.     {
  405.         $row = null;
  406.         $retval = true;
  407.  
  408.         $row = JTable::getInstance('extension');
  409.         $row->load($id);
  410.  
  411.         if ($row->protected)
  412.         {
  413.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_PACK_UNINSTALL_WARNCOREPACK'), JLog::WARNING, 'jerror');
  414.  
  415.             return false;
  416.         }
  417.  
  418.         $manifestFile = JPATH_MANIFESTS . '/packages/' . $row->get('element') . '.xml';
  419.         $manifest = new JInstallerManifestPackage($manifestFile);
  420.  
  421.         // Set the package root path
  422.         $this->parent->setPath('extension_root', JPATH_MANIFESTS . '/packages/' . $manifest->packagename);
  423.  
  424.         // Because packages may not have their own folders we cannot use the standard method of finding an installation manifest
  425.         if (!file_exists($manifestFile))
  426.         {
  427.             // TODO: Fail?
  428.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_PACK_UNINSTALL_MISSINGMANIFEST'), JLog::WARNING, 'jerror');
  429.  
  430.             return false;
  431.  
  432.         }
  433.  
  434.         $xml = simplexml_load_file($manifestFile);
  435.  
  436.         // If we cannot load the XML file return false
  437.         if (!$xml)
  438.         {
  439.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_PACK_UNINSTALL_LOAD_MANIFEST'), JLog::WARNING, 'jerror');
  440.  
  441.             return false;
  442.         }
  443.  
  444.         // Check for a valid XML root tag.
  445.         if ($xml->getName() != 'extension')
  446.         {
  447.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_PACK_UNINSTALL_INVALID_MANIFEST'), JLog::WARNING, 'jerror');
  448.  
  449.             return false;
  450.         }
  451.  
  452.         // If there is an manifest class file, let's load it
  453.         $this->scriptElement = $manifest->scriptfile;
  454.         $manifestScript = (string) $manifest->scriptfile;
  455.  
  456.         if ($manifestScript)
  457.         {
  458.             $manifestScriptFile = $this->parent->getPath('extension_root') . '/' . $manifestScript;
  459.  
  460.             if (is_file($manifestScriptFile))
  461.             {
  462.                 // Load the file
  463.                 include_once $manifestScriptFile;
  464.             }
  465.  
  466.             // Set the class name
  467.             $classname = $row->element . 'InstallerScript';
  468.  
  469.             if (class_exists($classname))
  470.             {
  471.                 // Create a new instance
  472.                 $this->parent->manifestClass = new $classname($this);
  473.  
  474.                 // And set this so we can copy it later
  475.                 $this->set('manifest_script', $manifestScript);
  476.             }
  477.         }
  478.  
  479.         ob_start();
  480.         ob_implicit_flush(false);
  481.  
  482.         // Run uninstall if possible
  483.         if ($this->parent->manifestClass && method_exists($this->parent->manifestClass, 'uninstall'))
  484.         {
  485.             $this->parent->manifestClass->uninstall($this);
  486.         }
  487.  
  488.         $msg = ob_get_contents();
  489.         ob_end_clean();
  490.  
  491.         if ($msg != '')
  492.         {
  493.             $this->parent->set('extension_message', $msg);
  494.         }
  495.  
  496.         $error = false;
  497.  
  498.         foreach ($manifest->filelist as $extension)
  499.         {

  542.     /**
  543.      * Gets the extension id.
  544.      *
  545.      * @param   string   $type    The extension type.
  546.      * @param   string   $id      The name of the extension (the element field).
  547.      * @param   integer  $client  The application id (0: Joomla CMS site; 1: Joomla CMS administrator).
  548.      * @param   string   $group   The extension group (mainly for plugins).
  549.      *
  550.      * @return  integer 
  551.      *
  552.      * @since   3.1
  553.      */
  554.     protected function _getExtensionID($type, $id, $client, $group)
  555.     {
  556.         $db = $this->parent->getDbo();
  557.  
  558.         $query = $db->getQuery(true)
  559.             ->select('extension_id')
  560.             ->from('#__extensions')
  561.             ->where('type = ' . $db->quote($type))
  562.             ->where('element = ' . $db->quote($id));
  563.  
  564.         switch ($type)
  565.         {
  566.             case 'plugin':
  567.                 // Plugins have a folder but not a client
  568.                 $query->where('folder = ' . $db->quote($group));
  569.                 break;
  570.  
  571.             case 'library':
  572.             case 'package':
  573.             case 'component':
  574.                 // Components, packages and libraries don't have a folder or client.
  575.                 // Included for completeness.
  576.                 break;
  577.  
  578.             case 'language':
  579.             case 'module':
  580.             case 'template':
  581.                 // Languages, modules and templates have a client but not a folder
  582.                 $client = JApplicationHelper::getClientInfo($client, true);
  583.                 $query->where('client_id = ' . (int) $client->id);
  584.                 break;
  585.         }
  586.  
  587.         $db->setQuery($query);
  588.         $result = $db->loadResult();
  589.  
  590.         // Note: For templates, libraries and packages their unique name is their key.
  591.         // This means they come out the same way they came in.
  592.         return $result;
  593.     }
