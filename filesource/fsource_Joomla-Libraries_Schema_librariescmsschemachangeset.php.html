  148.                 case 1:

  211.     /**

  216.      * @return  array  Array of stdClass objects where:
  217.      *                     file=filename,

  222.     private function getUpdateQueries(array $sqlfiles)
  223.     {
  224.         // Hold results as array of objects
  225.         $result = array();
  226.         foreach ($sqlfiles as $file)
  227.         {
  228.             $buffer = file_get_contents($file);
  229.  
  230.             // Create an array of queries from the sql file
  231.             $queries = JDatabaseDriver::splitSql($buffer);
  232.             foreach ($queries as $query)
  233.             {
  234.                 if ($trimmedQuery = $this->trimQuery($query))
  235.                 {
  236.                     $fileQueries = new stdClass;
  237.                     $fileQueries->file = $file;
  238.                     $fileQueries->updateQuery = $trimmedQuery;
  239.                     $result[] = $fileQueries;
  240.                 }
  241.             }
  242.         }
  243.         return $result;

  246.     /**
  247.      * Trim comment and blank lines out of a query string
  248.      *
  249.      * @param   string  $query  query string to be trimmed
  250.      *
  251.      * @return  string  String with leading comment lines removed

  255.     private function trimQuery($query)
  256.     {
  257.         $query = trim($query);
  258.  
  259.         while (substr($query, 0, 1) == '#' || substr($query, 0, 2) == '--' || substr($query, 0, 2) == '/*')
  260.         {
  261.             $endChars = (substr($query, 0, 1) == '#' || substr($query, 0, 2) == '--') ? "\n" : "*/";
  262.  
  263.             if ($position = strpos($query, $endChars))
  264.             {
  265.                 $query = trim(substr($query, $position + strlen($endChars)));
  266.             }
  267.             else
  268.             {
  269.                 // If no newline, the rest of the file is a comment, so return an empty string.
  270.                 return '';
  271.             }
  272.         }
  273.  
  274.         return trim($query);
