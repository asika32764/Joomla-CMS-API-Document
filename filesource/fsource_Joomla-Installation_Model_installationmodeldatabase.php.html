  179.     public function createDatabase($options)

  214.  
  215.         if (!$db->isMinimumVersion())

  217.             $app->enqueueMessage(JText::sprintf('INSTL_DATABASE_INVALID_' . strtoupper($type) . '_VERSION', $db_version), 'notice');
  218.             return false;
  219.         }
  220.  
  221.         if (($type == 'mysql') || ($type == 'mysqli'))

  230.  
  231.         // @internal Check for spaces in beginning or end of name

  233.         {
  234.             $app->enqueueMessage(JText::_('INSTL_DATABASE_NAME_INVALID_SPACES'), 'notice');
  235.             return false;
  236.         }
  237.  
  238.         // @internal Check for asc(00) Null in name
  239.         if (strpos($options->db_name, chr(00)) !== false)
  240.         {
  241.             $app->enqueueMessage(JText::_('INSTL_DATABASE_NAME_INVALID_CHAR'), 'notice');
  242.             return false;
  243.         }
  244.  
  245.         // PostgreSQL database older than version 9.0.0 needs to run 'CREATE LANGUAGE' to create function.
  246.         if (($options->db_type == 'postgresql') && (version_compare($db_version, '9.0.0', '<')))

  249.             $db->execute();
  250.         }
  251.  
  252.         // Get database's UTF support
  253.         $utfSupport = $db->hasUTFSupport();

  255.         // Try to select the database
  256.         try
  257.         {
  258.             $db->select($options->db_name);
  259.         }
  260.         catch (RuntimeException $e)
  261.         {
  262.             // If the database could not be selected, attempt to create it and then select it.
  263.             if ($this->createDB($db, $options, $utfSupport))
  264.             {
  265.                 $db->select($options->db_name);
  266.             }
  267.             else
  268.             {
  269.                 $app->enqueueMessage(JText::sprintf('INSTL_DATABASE_ERROR_CREATE', $options->db_name), 'notice');
  270.                 return false;
  271.             }
  272.         }
  273.  
  274.         $options = (array) $options;
  275.  
  276.         // Remove *_errors value
  277.         foreach ($options as $i => $option)
  278.         {
  279.             if (isset($i['1']) && $i['1'] == '*')
  280.             {
  281.                 unset($options[$i]);
  282.                 break;
  283.             }
  284.         }
  285.         $options = array_merge(array('db_created' => 1), $options);
  286.  
  287.         // Restore autoselect value after database creation
  288.         $options['db_select'] = $tmpSelect;
  289.  
  290.         $session = JFactory::getSession();
  291.         $session->set('setup.options', $options);
  292.  
  293.         return true;
  294.     }

  296.     /**

  345.     /**

  392.  
  393.         // Check if the schema is a valid file

  399.  
  400.         // Attempt to import the database schema.

  405.  
  406.         // Attempt to update the table #__schema.

  415.             $pathPart .= 'sqlazure/';

  419.             $pathPart .= $type . '/';
  420.         }
  421.         $files = JFolder::files($pathPart, '\.sql$');
  422.

  428.         $version = '';
  429.  
  430.         foreach ($files as $file)
  431.         {
  432.             if (version_compare($version, JFile::stripExt($file)) < 0)
  433.             {
  434.                 $version = JFile::stripExt($file);

  436.         }
  437.         $query = $db->getQuery(true)

  442.                     $db->quoteName('version_id')
  443.                 )
  444.             )

  446.         $db->setQuery($query);
  447.  
  448.         try
  449.         {
  450.             $db->execute();
  451.         }
  452.         catch (RuntimeException $e)
  453.         {
  454.             $app->enqueueMessage($e->getMessage(), 'notice');
  455.             return false;
  456.         }
  457.  
  458.         // Attempt to refresh manifest caches
  459.         $query->clear()

  462.         $db->setQuery($query);
  463.  
  464.         $return = true;
  465.  
  466.         try

  469.         }
  470.         catch (RuntimeException $e)

  472.             $app->enqueueMessage($e->getMessage(), 'notice');
  473.             $return = false;
  474.         }
  475.  
  476.         JFactory::$database = $db;
  477.         $installer = JInstaller::getInstance();
  478.

  481.             if (!$installer->refreshManifestCache($extension->extension_id))
  482.             {
  483.                 $app->enqueueMessage(JText::sprintf('INSTL_DATABASE_COULD_NOT_REFRESH_MANIFEST_CACHE', $extension->name), 'notice');
  484.                 return false;
  485.             }
  486.         }
  487.  
  488.         // Load the localise.sql for translating the data in joomla.sql
  489.         if ($type == 'mysqli' || $type == 'mysql')
  490.         {
  491.             $dblocalise = 'sql/mysql/localise.sql';
  492.         }
