  89.     /**
  90.      * Don't allow categories to be deleted if they contain items or subcategories with items
  91.      *
  92.      * @param   string  $context  The context for the content passed to the plugin.
  93.      * @param   object  $data     The data relating to the content that was deleted.

  96.      *

  99.     public function onContentBeforeDelete($context, $data)
  100.     {
  101.         // Skip plugin if we are deleting something other than categories
  102.         if ($context != 'com_categories.category')
  103.         {

  105.         }
  106.  
  107.         // Check if this function is enabled.
  108.         if (!$this->params->def('check_categories', 1))
  109.         {
  110.             return true;
  111.         }
  112.  
  113.         $extension = JFactory::getApplication()->input->getString('extension');
  114.  
  115.         // Default to true if not a core extension
  116.         $result = true;
  117.  
  118.         $tableInfo = array(
  119.             'com_banners' => array('table_name' => '#__banners'),
  120.             'com_contact' => array('table_name' => '#__contact_details'),
  121.             'com_content' => array('table_name' => '#__content'),
  122.             'com_newsfeeds' => array('table_name' => '#__newsfeeds'),
  123.             'com_weblinks' => array('table_name' => '#__weblinks')
  124.         );
  125.  
  126.         // Now check to see if this is a known core extension
  127.         if (isset($tableInfo[$extension]))

  129.             // Get table name for known core extensions
  130.             $table = $tableInfo[$extension]['table_name'];
  131.  
  132.             // See if this category has any content items
  133.             $count = $this->_countItemsInCategory($table, $data->get('id'));
  134.  
  135.             // Return false if db error
  136.             if ($count === false)
  137.             {
  138.                 $result = false;
  139.             }
  140.             else
  141.             {
  142.                 // Show error if items are found in the category
  143.                 if ($count > 0)
  144.                 {
  145.                     $msg = JText::sprintf('COM_CATEGORIES_DELETE_NOT_ALLOWED', $data->get('title')) .
  146.                         JText::plural('COM_CATEGORIES_N_ITEMS_ASSIGNED', $count);
  147.                     JError::raiseWarning(403, $msg);

  149.                 }
  150.  
  151.                 // Check for items in any child categories (if it is a leaf, there are no child categories)
  152.                 if (!$data->isLeaf())
  153.                 {
  154.                     $count = $this->_countItemsInChildren($table, $data->get('id'), $data);
  155.  
  156.                     if ($count === false)
  157.                     {
  158.                         $result = false;

  161.                     {
  162.                         $msg = JText::sprintf('COM_CATEGORIES_DELETE_NOT_ALLOWED', $data->get('title')) .
  163.                             JText::plural('COM_CATEGORIES_HAS_SUBCATEGORY_ITEMS', $count);
  164.                         JError::raiseWarning(403, $msg);
  165.                         $result = false;
  166.                     }
  167.                 }

  174.     /**
  175.      * Get count of items in a category
  176.      *
  177.      * @param   string   $table  table name of component table (column is catid)
  178.      * @param   integer  $catid  id of the category to check
  179.      *

  184.     private function _countItemsInCategory($table, $catid)

  209.     /**
  210.      * Get count of items in a category's child categories
  211.      *
  212.      * @param   string   $table  table name of component table (column is catid)
  213.      * @param   integer  $catid  id of the category to check
  214.      * @param   object   $data   The data relating to the content that was deleted.
  215.      *
  216.      * @return  mixed  count of items found or false if db error
  217.      *
  218.      * @since   1.6

  220.     private function _countItemsInChildren($table, $catid, $data)
  221.     {
  222.         $db = JFactory::getDbo();
  223.  
  224.         // Create subquery for list of child categories
  225.         $childCategoryTree = $data->getTree();
  226.  
  227.         // First element in tree is the current category, so we can skip that one
  228.         unset($childCategoryTree[0]);
  229.         $childCategoryIds = array();
  230.  
  231.         foreach ($childCategoryTree as $node)
  232.         {
  233.             $childCategoryIds[] = $node->id;
  234.         }
  235.  
  236.         // Make sure we only do the query if we have some categories to look in
  237.         if (count($childCategoryIds))
  238.         {
  239.             // Count the items in this category
  240.             $query = $db->getQuery(true)
  241.                 ->select('COUNT(id)')
  242.                 ->from($table)
  243.                 ->where('catid IN (' . implode(',', $childCategoryIds) . ')');
  244.             $db->setQuery($query);
  245.  
  246.             try
  247.             {
  248.                 $count = $db->loadResult();
  249.             }
  250.             catch (RuntimeException $e)
  251.             {
  252.                 JError::raiseWarning(500, $e->getMessage());
  253.  
  254.                 return false;
  255.             }
  256.  
  257.             return $count;
  258.         }
  259.         else
  260.             // If we didn't have any categories to check, return 0
  261.         {
  262.             return 0;
  263.         }
  264.     }
