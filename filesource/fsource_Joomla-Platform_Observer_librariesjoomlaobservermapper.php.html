  56. /**
  57.  * Observer pattern interface for Joomla
  58.  *
  59.  * A class that wants to observe another class must:
  60.  *
  61.  * 1) Add: implements JObserverInterface
  62.  *    to its class
  63.  *
  64.  * 2) Implement a constructor, that can look like this:
  65.  *     public function __construct(JObservableInterface $observableObject)
  66.  *     {
  67.  *         $observableObject->attachObserver($this);
  68.  *      $this->observableObject = $observableObject;
  69.  *     }
  70.  *
  71.  * 3) and must implement the instanciator function createObserver() below, e.g. as follows:
  72.  *     public static function createObserver(JObservableInterface $observableObject, $params = array())
  73.  *     {
  74.  *         $observer = new self($observableObject);
  75.  *      $observer->... = $params['...']; ...
  76.  *         return $observer;
  77.  *     }
  78.  *
  79.  * 4) Then add functions corresponding to the events to be observed,
  80.  *    E.g. to respond to event: $this->_observers->update('onBeforeLoad', array($keys, $reset));
  81.  *    following function is needed in the obser:
  82.  *  public function onBeforeLoad($keys, $reset) { ... }
  83.  *
  84.  * 5) Finally, the binding is made outside the observable and observer classes, using:
  85.  * JObserverMapper::addObserverClassToClass('ObserverClassname', 'ObservableClassname', array('paramName' => 'paramValue'));
  86.  * where the last array will be provided to the observer instanciator function createObserver.
  87.  *
  88.  * @package     Joomla
  89.  * @subpackage  Observer

  302.     public static function attachAllObservers(JObservableInterface $observableObject)
  303.     {
  304.         $observableClass = get_class($observableObject);
  305.  
  306.         while ($observableClass != false)

  310.             {
  311.                 foreach (static::$observations[$observableClass] as $observerClass => $params)
  312.                 {
  313.                     // Attach an Observer to the Observable subject:
  314.                     /**
  315. /**

  318.                     $observerClass::createObserver($observableObject, $params);

  322.             // Get parent class name (or false if none), and redo the above on it:
  323.             $observableClass = get_parent_class($observableClass);
