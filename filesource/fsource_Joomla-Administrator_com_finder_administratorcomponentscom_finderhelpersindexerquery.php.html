  479.     protected function processStaticTaxonomy($filterId)
  480.     {

  482.         $db = JFactory::getDbo();

  484.         // Initialize user variables
  485.         $user = JFactory::getUser();
  486.         $groups = implode(',', $user->getAuthorisedViewLevels());
  487.  
  488.         // Load the predefined filter.

  492.             ->where('f.filter_id = ' . (int) $filterId);
  493.  
  494.         $db->setQuery($query);
  495.         $return = $db->loadObject();
  496.  
  497.         // Check the returned filter.
  498.         if (empty($return))
  499.         {
  500.             return false;
  501.         }
  502.  
  503.         // Set the filter.
  504.         $this->filter = (int) $filterId;
  505.  
  506.         // Get a parameter object for the filter date options.
  507.         $registry = new JRegistry;
  508.         $registry->loadString($return->params);
  509.         $params = $registry;
  510.  
  511.         // Set the dates if not already set.
  512.         $this->dates->def('d1', $params->get('d1'));
  513.         $this->dates->def('d2', $params->get('d2'));
  514.         $this->dates->def('w1', $params->get('w1'));
  515.         $this->dates->def('w2', $params->get('w2'));
  516.  
  517.         // Remove duplicates and sanitize.
  518.         $filters = explode(',', $return->data);
  519.         $filters = array_unique($filters);
  520.         JArrayHelper::toInteger($filters);
  521.  
  522.         // Remove any values of zero.
  523.         if (array_search(0, $filters, true) !== false)
  524.         {
  525.             unset($filters[array_search(0, $filters, true)]);
  526.         }
  527.  
  528.         // Check if we have any real input.
  529.         if (empty($filters))
  530.         {
  531.             return true;
  532.         }
  533.  
  534.         /*
  535.          * Create the query to get filters from the database. We do this for
  536.          * two reasons: one, it allows us to ensure that the filters being used
  537.          * are real; two, we need to sort the filters by taxonomy branch.
  538.          */
  539.         $query->clear()
  540.             ->select('t1.id, t1.title, t2.title AS branch')
  541.             ->from($db->quoteName('#__finder_taxonomy') . ' AS t1')
  542.             ->join('INNER', $db->quoteName('#__finder_taxonomy') . ' AS t2 ON t2.id = t1.parent_id')
  543.             ->where('t1.state = 1')
  544.             ->where('t1.access IN (' . $groups . ')')
  545.             ->where('t1.id IN (' . implode(',', $filters) . ')')
  546.             ->where('t2.state = 1')
  547.             ->where('t2.access IN (' . $groups . ')');
  548.  
  549.         // Load the filters.
  550.         $db->setQuery($query);
  551.         $results = $db->loadObjectList();
  552.  
  553.         // Sort the filter ids by branch.
  554.         foreach ($results as $result)
  555.         {
  556.             $this->filters[$result->branch][$result->title] = (int) $result->id;
  557.         }
  558.  
  559.         return true;
  560.     }
  561.  
  562.     /**
  563.      * Method to process the dynamic taxonomy input. The dynamic taxonomy input
  564.      * comes in the form of select fields that the user chooses from. The
  565.      * dynamic taxonomy input is processed AFTER the static taxonomy input
  566.      * because the dynamic options can be used to further narrow a static
  567.      * taxonomy filter.
  568.      *
  569.      * @param   array  $filters  An array of taxonomy node ids.
  570.      *
  571.      * @return  boolean  True on success.
  572.      *
  573.      * @since   2.5
  574.      * @throws  Exception on database error.
  575.      */
  576.     protected function processDynamicTaxonomy($filters)
  577.     {
  578.         // Initialize user variables
  579.         $user = JFactory::getUser();
  580.         $groups = implode(',', $user->getAuthorisedViewLevels());
  581.  
  582.         // Remove duplicates and sanitize.
  583.         $filters = array_unique($filters);
  584.         JArrayHelper::toInteger($filters);
  585.  
  586.         // Remove any values of zero.
  587.         if (array_search(0, $filters, true) !== false)
  588.         {
  589.             unset($filters[array_search(0, $filters, true)]);
  590.         }
  591.  
  592.         // Check if we have any real input.
  593.         if (empty($filters))
  594.         {
  595.             return true;
  596.         }
  597.  
  598.         // Get the database object.
  599.         $db = JFactory::getDbo();
  600.         $query = $db->getQuery(true);
  601.  
  602.         /*
  603.          * Create the query to get filters from the database. We do this for
  604.          * two reasons: one, it allows us to ensure that the filters being used
  605.          * are real; two, we need to sort the filters by taxonomy branch.
  606.          */
  607.         $query->select('t1.id, t1.title, t2.title AS branch')
  608.             ->from($db->quoteName('#__finder_taxonomy') . ' AS t1')
  609.             ->join('INNER', $db->quoteName('#__finder_taxonomy') . ' AS t2 ON t2.id = t1.parent_id')
  610.             ->where('t1.state = 1')
  611.             ->where('t1.access IN (' . $groups . ')')
  612.             ->where('t1.id IN (' . implode(',', $filters) . ')')
  613.             ->where('t2.state = 1')
  614.             ->where('t2.access IN (' . $groups . ')');
  615.  
  616.         // Load the filters.
  617.         $db->setQuery($query);
  618.         $results = $db->loadObjectList();
  619.  
  620.         // Cleared filter branches.
  621.         $cleared = array();
  622.  
  623.         /*
  624.          * Sort the filter ids by branch. Because these filters are designed to
  625.          * override and further narrow the items selected in the static filter,
  626.          * we will clear the values from the static filter on a branch by
  627.          * branch basis before adding the dynamic filters. So, if the static
  628.          * filter defines a type filter of "articles" and three "category"
  629.          * filters but the user only limits the category further, the category
  630.          * filters will be flushed but the type filters will not.
  631.          */
  632.         foreach ($results as $result)
  633.         {
  634.             // Check if the branch has been cleared.
  635.             if (!in_array($result->branch, $cleared))
  636.             {
  637.                 // Clear the branch.
  638.                 $this->filters[$result->branch] = array();
  639.  
  640.                 // Add the branch to the cleared list.
  641.                 $cleared[] = $result->branch;
  642.             }
  643.  
  644.             // Add the filter to the list.
  645.             $this->filters[$result->branch][$result->title] = (int) $result->id;
  646.         }
  647.  
  648.         return true;
  649.     }
  650.  
  651.     /**
  652.      * Method to process the query date filters to determine start and end
  653.      * date limitations.
  654.      *
  655.      * @param   string  $date1  The first date filter.
  656.      * @param   string  $date2  The second date filter.
  657.      * @param   string  $when1  The first date modifier.
  658.      * @param   string  $when2  The second date modifier.
  659.      *
  660.      * @return  boolean  True on success.
  661.      *
  662.      * @since   2.5
  663.      */
  664.     protected function processDates($date1, $date2, $when1, $when2)
  665.     {
  666.         // Clean up the inputs.
  667.         $date1 = JString::trim(JString::strtolower($date1));
  668.         $date2 = JString::trim(JString::strtolower($date2));
  669.         $when1 = JString::trim(JString::strtolower($when1));
  670.         $when2 = JString::trim(JString::strtolower($when2));
  671.  
  672.         // Get the time offset.
  673.         $offset = JFactory::getApplication()->getCfg('offset');
  674.  
  675.         // Array of allowed when values.
  676.         $whens = array('before', 'after', 'exact');
  677.  
  678.         // The value of 'today' is a special case that we need to handle.
  679.         if ($date1 === JString::strtolower(JText::_('COM_FINDER_QUERY_FILTER_TODAY')))
  680.         {
  681.             $today = JFactory::getDate('now', $offset);
  682.             $date1 = $today->format('%Y-%m-%d');
  683.         }
  684.  
  685.         // Try to parse the date string.
  686.         $date = JFactory::getDate($date1, $offset);
  687.  
  688.         // Check if the date was parsed successfully.
  689.         if ($date->toUnix() !== null)
  690.         {
  691.             // Set the date filter.
  692.             $this->date1 = $date->toSQL();
  693.             $this->when1 = in_array($when1, $whens) ? $when1 : 'before';
  694.         }
  695.  
  696.         // The value of 'today' is a special case that we need to handle.
  697.         if ($date2 === JString::strtolower(JText::_('COM_FINDER_QUERY_FILTER_TODAY')))
  698.         {
  699.             $today = JFactory::getDate('now', $offset);
  700.             $date2 = $today->format('%Y-%m-%d');
  701.         }
  702.  
  703.         // Try to parse the date string.
  704.         $date = JFactory::getDate($date2, $offset);
  705.  
  706.         // Check if the date was parsed successfully.
  707.         if ($date->toUnix() !== null)
  708.         {
  709.             // Set the date filter.
  710.             $this->date2 = $date->toSQL();
  711.             $this->when2 = in_array($when2, $whens) ? $when2 : 'before';
  712.         }
  713.  
  714.         return true;
  715.     }
  716.  
  717.     /**
  718.      * Method to process the query input string and extract required, optional,
  719.      * and excluded tokens; taxonomy filters; and date filters.
  720.      *
  721.      * @param   string  $input  The query input string.
  722.      * @param   string  $lang   The query input language.
  723.      * @param   string  $mode   The query matching mode.
  724.      *
  725.      * @return  boolean  True on success.
  726.      *
  727.      * @since   2.5
  728.      * @throws  Exception on database error.
  729.      */
  730.     protected function processString($input, $lang, $mode)
  731.     {
  732.         // Clean up the input string.
  733.         $input = html_entity_decode($input, ENT_QUOTES, 'UTF-8');
  734.         $input = JString::strtolower($input);
  735.         $input = preg_replace('#\s+#mi', ' ', $input);
  736.         $input = JString::trim($input);
  737.         $debug = JFactory::getConfig()->get('debug_lang');
  738.  
  739.         /*
  740.          * First, we need to handle string based modifiers. String based
  741.          * modifiers could potentially include things like "category:blah" or
  742.          * "before:2009-10-21" or "type:article", etc.
  743.          */
  744.         $patterns = array(
  745.             'before' => JText::_('COM_FINDER_FILTER_WHEN_BEFORE'),
  746.             'after' => JText::_('COM_FINDER_FILTER_WHEN_AFTER')
  747.         );
  748.  
  749.         // Add the taxonomy branch titles to the possible patterns.
  750.         foreach (FinderIndexerTaxonomy::getBranchTitles() as $branch)
  751.         {
  752.             // Add the pattern.
  753.             $patterns[$branch] = JString::strtolower(JText::_(FinderHelperLanguage::branchSingular($branch)));
  754.         }
  755.  
  756.         // Container for search terms and phrases.
  757.         $terms = array();
  758.         $phrases = array();
  759.  
  760.         // Cleared filter branches.
  761.         $cleared = array();
  762.  
  763.         /*
  764.          * Compile the suffix pattern. This is used to match the values of the
  765.          * filter input string. Single words can be input directly, multi-word
  766.          * values have to be wrapped in double quotes.
  767.          */
  768.         $quotes = html_entity_decode('&#8216;&#8217;&#39;', ENT_QUOTES, 'UTF-8');
  769.         $suffix = '(([\w\d' . $quotes . '-]+)|\"([\w\d\s' . $quotes . '-]+)\")';
  770.  
  771.         /*
  772.          * Iterate through the possible filter patterns and search for matches.
  773.          * We need to match the key, colon, and a value pattern for the match
  774.          * to be valid.
  775.          */
  776.         foreach ($patterns as $modifier => $pattern)
  777.         {
  778.             $matches = array();
  779.  
  780.             if ($debug)
  781.             {
  782.                 $pattern = substr($pattern, 2, -2);
  783.             }
  784.  
  785.             // Check if the filter pattern is in the input string.
  786.             if (preg_match('#' . $pattern . '\s*:\s*' . $suffix . '#mi', $input, $matches))
  787.             {
  788.                 // Get the value given to the modifier.
  789.                 $value = isset($matches[3]) ? $matches[3] : $matches[1];
  790.  
  791.                 // Now we have to handle the filter string.
  792.                 switch ($modifier)
  793.                 {
  794.                     // Handle a before and after date filters.
  795.                     case 'before':
  796.                     case 'after':
  797.                     {
  798.                         // Get the time offset.
  799.                         $offset = JFactory::getApplication()->getCfg('offset');
  800.  
  801.                         // Array of allowed when values.
  802.                         $whens = array('before', 'after', 'exact');
  803.  
  804.                         // The value of 'today' is a special case that we need to handle.
  805.                         if ($value === JString::strtolower(JText::_('COM_FINDER_QUERY_FILTER_TODAY')))
  806.                         {
  807.                             $today = JFactory::getDate('now', $offset);
  808.                             $value = $today->format('%Y-%m-%d');
  809.                         }
  810.  
  811.                         // Try to parse the date string.
  812.                         $date = JFactory::getDate($value, $offset);
  813.  
  814.                         // Check if the date was parsed successfully.
  815.                         if ($date->toUnix() !== null)
  816.                         {
  817.                             // Set the date filter.
  818.                             $this->date1 = $date->toSQL();
  819.                             $this->when1 = in_array($modifier, $whens) ? $modifier : 'before';
  820.                         }
  821.  
  822.                         break;
  823.                     }
  824.  
  825.                     // Handle a taxonomy branch filter.
  826.                     default:
  827.                         {
  828.                         // Try to find the node id.
  829.                         $return = FinderIndexerTaxonomy::getNodeByTitle($modifier, $value);
  830.  
  831.                         // Check if the node id was found.
  832.                         if ($return)
  833.                         {
  834.                             // Check if the branch has been cleared.
  835.                             if (!in_array($modifier, $cleared))
  836.                             {
  837.                                 // Clear the branch.
  838.                                 $this->filters[$modifier] = array();
  839.  
  840.                                 // Add the branch to the cleared list.
  841.                                 $cleared[] = $modifier;
  842.                             }
  843.  
  844.                             // Add the filter to the list.
  845.                             $this->filters[$modifier][$return->title] = (int) $return->id;
  846.                         }
  847.  
  848.                         break;
  849.                         }
  850.                 }
  851.  
  852.                 // Clean up the input string again.
  853.                 $input = str_replace($matches[0], '', $input);
  854.                 $input = preg_replace('#\s+#mi', ' ', $input);
  855.                 $input = JString::trim($input);
  856.             }
  857.         }
  858.  
  859.         /*
  860.          * Extract the tokens enclosed in double quotes so that we can handle
  861.          * them as phrases.
  862.          */
  863.         if (JString::strpos($input, '"') !== false)
  864.         {
  865.             $matches = array();
  866.  
  867.             // Extract the tokens enclosed in double quotes.
  868.             if (preg_match_all('#\"([^"]+)\"#mi', $input, $matches))
  869.             {
  870.                 /*
  871.                  * One or more phrases were found so we need to iterate through
  872.                  * them, tokenize them as phrases, and remove them from the raw
  873.                  * input string before we move on to the next processing step.
  874.                  */
  875.                 foreach ($matches[1] as $key => $match)
  876.                 {
  877.                     // Find the complete phrase in the input string.
  878.                     $pos = JString::strpos($input, $matches[0][$key]);
  879.                     $len = JString::strlen($matches[0][$key]);
  880.  
  881.                     // Add any terms that are before this phrase to the stack.
  882.                     if (JString::trim(JString::substr($input, 0, $pos)))
  883.                     {
  884.                         $terms = array_merge($terms, explode(' ', JString::trim(JString::substr($input, 0, $pos))));
  885.                     }
  886.  
  887.                     // Strip out everything up to and including the phrase.
  888.                     $input = JString::substr($input, $pos + $len);
  889.  
  890.                     // Clean up the input string again.
  891.                     $input = preg_replace('#\s+#mi', ' ', $input);
  892.                     $input = JString::trim($input);
  893.  
  894.                     // Get the number of words in the phrase.
  895.                     $parts = explode(' ', $match);
  896.  
  897.                     // Check if the phrase is longer than three words.
  898.                     if (count($parts) > 3)
  899.                     {
  900.                         /*
  901.                          * If the phrase is longer than three words, we need to
  902.                          * break it down into smaller chunks of phrases that
  903.                          * are less than or equal to three words. We overlap
  904.                          * the chunks so that we can ensure that a match is
  905.                          * found for the complete phrase and not just portions
  906.                          * of it.
  907.                          */
  908.                         for ($i = 0, $c = count($parts); $i < $c; $i += 2)
  909.                         {
  910.                             // Set up the chunk.
  911.                             $chunk = array();
  912.  
  913.                             // The chunk has to be assembled based on how many
  914.                             // pieces are available to use.
  915.                             switch ($c - $i)
  916.                             {
  917.                                 /*
  918.                                  * If only one word is left, we can break from
  919.                                  * the switch and loop because the last word
  920.                                  * was already used at the end of the last
  921.                                  * chunk.
  922.                                  */
  923.                                 case 1:
  924.                                     break 2;
  925.  
  926.                                 // If there words are left, we use them both as
  927.                                 // the last chunk of the phrase and we're done.
  928.                                 case 2:
  929.                                     $chunk[] = $parts[$i];
  930.                                     $chunk[] = $parts[$i + 1];
  931.                                     break;
  932.  
  933.                                 // If there are three or more words left, we
  934.                                 // build a three word chunk and continue on.
  935.                                 default:
  936.                                     $chunk[] = $parts[$i];
  937.                                     $chunk[] = $parts[$i + 1];
  938.                                     $chunk[] = $parts[$i + 2];
  939.                                     break;
  940.                             }
  941.  
  942.                             // If the chunk is not empty, add it as a phrase.
  943.                             if (count($chunk))
  944.                             {
  945.                                 $phrases[] = implode(' ', $chunk);
  946.                                 $terms[] = implode(' ', $chunk);
  947.                             }
  948.                         }
  949.                     }
  950.                     else
  951.                     {
  952.                         // The phrase is <= 3 words so we can use it as is.
  953.                         $phrases[] = $match;
  954.                         $terms[] = $match;
  955.                     }
  956.                 }
  957.             }
  958.         }
  959.  
  960.         // Add the remaining terms if present.
  961.         if (!empty($input))
  962.         {
  963.             $terms = array_merge($terms, explode(' ', $input));
  964.         }
  965.  
  966.         // An array of our boolean operators. $operator => $translation
  967.         $operators = array(
  968.             'AND' => JString::strtolower(JText::_('COM_FINDER_QUERY_OPERATOR_AND')),
  969.             'OR' => JString::strtolower(JText::_('COM_FINDER_QUERY_OPERATOR_OR')),
  970.             'NOT' => JString::strtolower(JText::_('COM_FINDER_QUERY_OPERATOR_NOT'))
  971.         );
  972.  
  973.         // If language debugging is enabled you need to ignore the debug strings in matching.
  974.         if (JDEBUG)
  975.         {
  976.             $debugStrings = array('**', '??');
  977.             $operators = str_replace($debugStrings, '', $operators);
  978.         }
  979.  
  980.         /*
  981.          * Iterate through the terms and perform any sorting that needs to be
  982.          * done based on boolean search operators. Terms that are before an
  983.          * and/or/not modifier have to be handled in relation to their operator.
  984.          */
  985.         for ($i = 0, $c = count($terms); $i < $c; $i++)
  986.         {
  987.             // Check if the term is followed by an operator that we understand.
  988.             if (isset($terms[$i + 1]) && in_array($terms[$i + 1], $operators))
  989.             {
  990.                 // Get the operator mode.
  991.                 $op = array_search($terms[$i + 1], $operators);
  992.  
  993.                 // Handle the AND operator.
  994.                 if ($op === 'AND' && isset($terms[$i + 2]))
  995.                 {
  996.                     // Tokenize the current term.
  997.                     $token = FinderIndexerHelper::tokenize($terms[$i], $lang, true);
  998.                     $token = $this->getTokenData($token);
  999.  
  1000.                     // Set the required flag.
  1001.                     $token->required = true;
  1002.  
  1003.                     // Add the current token to the stack.
  1004.                     $this->included[] = $token;
  1005.                     $this->highlight = array_merge($this->highlight, array_keys($token->matches));
  1006.  
  1007.                     // Skip the next token (the mode operator).
  1008.                     $this->operators[] = $terms[$i + 1];
  1009.  
  1010.                     // Tokenize the term after the next term (current plus two).
  1011.                     $other = FinderIndexerHelper::tokenize($terms[$i + 2], $lang, true);
  1012.                     $other = $this->getTokenData($other);
  1013.  
  1014.                     // Set the required flag.
  1015.                     $other->required = true;
  1016.  
  1017.                     // Add the token after the next token to the stack.
  1018.                     $this->included[] = $other;
  1019.                     $this->highlight = array_merge($this->highlight, array_keys($other->matches));
  1020.  
  1021.                     // Remove the processed phrases if possible.
  1022.                     if (($pk = array_search($terms[$i], $phrases)) !== false)
  1023.                     {
  1024.                         unset($phrases[$pk]);
  1025.                     }
  1026.                     if (($pk = array_search($terms[$i + 2], $phrases)) !== false)
  1027.                     {
  1028.                         unset($phrases[$pk]);
  1029.                     }
  1030.  
  1031.                     // Remove the processed terms.
  1032.                     unset($terms[$i]);
  1033.                     unset($terms[$i + 1]);
  1034.                     unset($terms[$i + 2]);
  1035.  
  1036.                     // Adjust the loop.
  1037.                     $i += 2;
  1038.                     continue;
  1039.                 }
  1040.                 // Handle the OR operator.
  1041.                 elseif ($op === 'OR' && isset($terms[$i + 2]))
  1042.                 {
  1043.                     // Tokenize the current term.
  1044.                     $token = FinderIndexerHelper::tokenize($terms[$i], $lang, true);
  1045.                     $token = $this->getTokenData($token);
  1046.  
  1047.                     // Set the required flag.
  1048.                     $token->required = false;
  1049.  
  1050.                     // Add the current token to the stack.
  1051.                     if (count($token->matches))
  1052.                     {
  1053.                         $this->included[] = $token;
  1054.                         $this->highlight = array_merge($this->highlight, array_keys($token->matches));
  1055.                     }
  1056.                     else
  1057.                     {
  1058.                         $this->ignored[] = $token;
  1059.                     }
  1060.  
  1061.                     // Skip the next token (the mode operator).
  1062.                     $this->operators[] = $terms[$i + 1];
  1063.  
  1064.                     // Tokenize the term after the next term (current plus two).
  1065.                     $other = FinderIndexerHelper::tokenize($terms[$i + 2], $lang, true);
  1066.                     $other = $this->getTokenData($other);
  1067.  
  1068.                     // Set the required flag.
  1069.                     $other->required = false;
  1070.  
  1071.                     // Add the token after the next token to the stack.
  1072.                     if (count($other->matches))
  1073.                     {
  1074.                         $this->included[] = $other;
  1075.                         $this->highlight = array_merge($this->highlight, array_keys($other->matches));
  1076.                     }
  1077.                     else
  1078.                     {
  1079.                         $this->ignored[] = $other;
  1080.                     }
  1081.  
  1082.                     // Remove the processed phrases if possible.
  1083.                     if (($pk = array_search($terms[$i], $phrases)) !== false)
  1084.                     {
  1085.                         unset($phrases[$pk]);
  1086.                     }
  1087.                     if (($pk = array_search($terms[$i + 2], $phrases)) !== false)
  1088.                     {
  1089.                         unset($phrases[$pk]);
  1090.                     }
  1091.  
  1092.                     // Remove the processed terms.
  1093.                     unset($terms[$i]);
  1094.                     unset($terms[$i + 1]);
  1095.                     unset($terms[$i + 2]);
  1096.  
  1097.                     // Adjust the loop.
  1098.                     $i += 2;
  1099.                     continue;
  1100.                 }
  1101.             }
  1102.             // Handle an orphaned OR operator.
  1103.             elseif (isset($terms[$i + 1]) && array_search($terms[$i], $operators) === 'OR')
  1104.             {
  1105.                 // Skip the next token (the mode operator).
  1106.                 $this->operators[] = $terms[$i];
  1107.  
  1108.                 // Tokenize the next term (current plus one).
  1109.                 $other = FinderIndexerHelper::tokenize($terms[$i + 1], $lang, true);
  1110.                 $other = $this->getTokenData($other);
  1111.  
  1112.                 // Set the required flag.
  1113.                 $other->required = false;
  1114.  
  1115.                 // Add the token after the next token to the stack.
  1116.                 if (count($other->matches))
  1117.                 {
  1118.                     $this->included[] = $other;
  1119.                     $this->highlight = array_merge($this->highlight, array_keys($other->matches));
  1120.                 }
  1121.                 else
  1122.                 {
  1123.                     $this->ignored[] = $other;
  1124.                 }
  1125.  
  1126.                 // Remove the processed phrase if possible.
  1127.                 if (($pk = array_search($terms[$i + 1], $phrases)) !== false)
  1128.                 {
  1129.                     unset($phrases[$pk]);
  1130.                 }
  1131.  
  1132.                 // Remove the processed terms.
  1133.                 unset($terms[$i]);
  1134.                 unset($terms[$i + 1]);
  1135.  
  1136.                 // Adjust the loop.
  1137.                 $i += 1;
  1138.                 continue;
  1139.             }
  1140.             // Handle the NOT operator.
  1141.             elseif (isset($terms[$i + 1]) && array_search($terms[$i], $operators) === 'NOT')
  1142.             {
  1143.                 // Skip the next token (the mode operator).
  1144.                 $this->operators[] = $terms[$i];
  1145.  
  1146.                 // Tokenize the next term (current plus one).
  1147.                 $other = FinderIndexerHelper::tokenize($terms[$i + 1], $lang, true);
  1148.                 $other = $this->getTokenData($other);
  1149.  
  1150.                 // Set the required flag.
  1151.                 $other->required = false;
  1152.  
  1153.                 // Add the next token to the stack.
  1154.                 if (count($other->matches))
  1155.                 {
  1156.                     $this->excluded[] = $other;
  1157.                 }
  1158.                 else
  1159.                 {
  1160.                     $this->ignored[] = $other;
  1161.                 }
  1162.  
  1163.                 // Remove the processed phrase if possible.
  1164.                 if (($pk = array_search($terms[$i + 1], $phrases)) !== false)
  1165.                 {
  1166.                     unset($phrases[$pk]);
  1167.                 }
  1168.  
  1169.                 // Remove the processed terms.
  1170.                 unset($terms[$i]);
  1171.                 unset($terms[$i + 1]);
  1172.  
  1173.                 // Adjust the loop.
  1174.                 $i += 1;
  1175.                 continue;
  1176.             }
  1177.         }
  1178.  
  1179.         /*
  1180.          * Iterate through any search phrases and tokenize them. We handle
  1181.          * phrases as autonomous units and do not break them down into two and
  1182.          * three word combinations.
  1183.          */
  1184.         for ($i = 0, $c = count($phrases); $i < $c; $i++)
  1185.         {
  1186.             // Tokenize the phrase.
  1187.             $token = FinderIndexerHelper::tokenize($phrases[$i], $lang, true);
  1188.             $token = $this->getTokenData($token);
  1189.  
  1190.             // Set the required flag.
  1191.             $token->required = true;
  1192.  
  1193.             // Add the current token to the stack.
  1194.             $this->included[] = $token;
  1195.             $this->highlight = array_merge($this->highlight, array_keys($token->matches));
  1196.  
  1197.             // Remove the processed term if possible.
  1198.             if (($pk = array_search($phrases[$i], $terms)) !== false)
  1199.             {
  1200.                 unset($terms[$pk]);
  1201.             }
  1202.  
  1203.             // Remove the processed phrase.
  1204.             unset($phrases[$i]);
  1205.         }
  1206.  
  1207.         /*
  1208.          * Handle any remaining tokens using the standard processing mechanism.
  1209.          */
  1210.         if (!empty($terms))
  1211.         {
  1212.             // Tokenize the terms.
  1213.             $terms = implode(' ', $terms);
  1214.             $tokens = FinderIndexerHelper::tokenize($terms, $lang, false);
  1215.  
  1216.             // Make sure we are working with an array.
  1217.             $tokens = is_array($tokens) ? $tokens : array($tokens);
  1218.  
  1219.             // Get the token data and required state for all the tokens.
  1220.             foreach ($tokens as $token)
  1221.             {
  1222.                 // Get the token data.
  1223.                 $token = $this->getTokenData($token);
  1224.  
  1225.                 // Set the required flag for the token.
  1226.                 $token->required = $mode === 'AND' ? ($token->phrase ? false : true) : false;
  1227.  
  1228.                 // Add the token to the appropriate stack.
  1229.                 if (count($token->matches) || $token->required)
  1230.                 {
  1231.                     $this->included[] = $token;
  1232.                     $this->highlight = array_merge($this->highlight, array_keys($token->matches));
  1233.                 }
  1234.                 else
  1235.                 {
  1236.                     $this->ignored[] = $token;
  1237.                 }
  1238.             }
  1239.         }
  1240.  
  1241.         return true;
  1242.     }
  1243.  
  1244.     /**
  1245.      * Method to get the base and similar term ids and, if necessary, suggested
  1246.      * term data from the database. The terms ids are identified based on a
  1247.      * 'like' match in MySQL and/or a common stem. If no term ids could be
  1248.      * found, then we know that we will not be able to return any results for
  1249.      * that term and we should try to find a similar term to use that we can
  1250.      * match so that we can suggest the alternative search query to the user.
  1251.      *
  1252.      * @param   FinderIndexerToken  $token  A FinderIndexerToken object.
  1253.      *
  1254.      * @return  FinderIndexerToken  A FinderIndexerToken object.
  1255.      *
  1256.      * @since   2.5
  1257.      * @throws  Exception on database error.
  1258.      */
  1259.     protected function getTokenData($token)
  1260.     {
  1261.         // Get the database object.
  1262.         $db = JFactory::getDbo();
  1263.  
  1264.         // Create a database query to build match the token.
  1265.         $query = $db->getQuery(true)
  1266.             ->select('t.term, t.term_id')
  1267.             ->from('#__finder_terms AS t');
  1268.  
  1269.         /*
  1270.          * If the token is a phrase, the lookup process is fairly simple. If
  1271.          * the token is a word, it is a little more complicated. We have to
  1272.          * create two queries to lookup the term and the stem respectively,
  1273.          * then union the result sets together. This is MUCH faster than using
  1274.          * an or condition in the database query.
  1275.          */
  1276.         if ($token->phrase)
  1277.         {
  1278.             // Add the phrase to the query.
  1279.             $query->where('t.term = ' . $db->quote($token->term))
  1280.                 ->where('t.phrase = 1');
  1281.         }
  1282.         else
  1283.         {
  1284.             // Add the term to the query.
  1285.             $query->where('t.term = ' . $db->quote($token->term))
  1286.                 ->where('t.phrase = 0');
  1287.  
  1288.             // Clone the query, replace the WHERE clause.
  1289.             $sub = clone($query);
  1290.             $sub->clear('where');
  1291.             $sub->where('t.stem = ' . $db->quote($token->stem));
  1292.             $sub->where('t.phrase = 0');
  1293.  
  1294.             // Union the two queries.
  1295.             $query->union($sub);
  1296.         }
  1297.  
  1298.         // Get the terms.
  1299.         $db->setQuery($query);
  1300.         $matches = $db->loadObjectList();
  1301.  
  1302.         // Setup the container.
  1303.         $token->matches = array();
  1304.  
  1305.         // Check the matching terms.
  1306.         if (!empty($matches))
  1307.         {
  1308.             // Add the matches to the token.
  1309.             for ($i = 0, $c = count($matches); $i < $c; $i++)
  1310.             {
  1311.                 $token->matches[$matches[$i]->term] = (int) $matches[$i]->term_id;
  1312.             }
  1313.         }
  1314.  
  1315.         // If no matches were found, try to find a similar but better token.
  1316.         if (empty($token->matches))
  1317.         {
  1318.             // Create a database query to get the similar terms.
  1319.             // TODO: PostgreSQL doesn't support SOUNDEX out of the box
  1320.             $query->clear()
  1321.                 ->select('DISTINCT t.term_id AS id, t.term AS term')
  1322.                 ->from('#__finder_terms AS t')
  1323.                 // ->where('t.soundex = ' . soundex($db->quote($token->term)))
  1324.                 ->where('t.soundex = SOUNDEX(' . $db->quote($token->term) . ')')
  1325.                 ->where('t.phrase = ' . (int) $token->phrase);
