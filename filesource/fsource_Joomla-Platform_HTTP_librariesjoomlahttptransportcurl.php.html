  16.  * @subpackage  HTTP

  19. class JHttpTransportCurl implements JHttpTransport
  20. {
  21.     /**
  22.      * @var    JRegistry  The client options.
  23.      * @since  11.3
  24.      */
  25.     protected $options;
  26.  
  27.     /**
  28.      * Constructor. CURLOPT_FOLLOWLOCATION must be disabled when open_basedir or safe_mode are enabled.
  29.      *
  30.      * @param   JRegistry  $options  Client options object.
  31.      *
  32.      * @see     http://www.php.net/manual/en/function.curl-setopt.php
  33.      * @since   11.3
  34.      * @throws  RuntimeException
  35.      */
  36.     public function __construct(JRegistry $options)
  37.     {
  38.         if (!function_exists('curl_init') || !is_callable('curl_init'))
  39.         {
  40.             throw new RuntimeException('Cannot use a cURL transport when curl_init() is not available.');
  41.         }
  42.  
  43.         $this->options = $options;
  44.     }
  45.  
  46.     /**
  47.      * Send a request to the server and return a JHttpResponse object with the response.
  48.      *
  49.      * @param   string   $method     The HTTP method for sending the request.
  50.      * @param   JUri     $uri        The URI to the resource to request.
  51.      * @param   mixed    $data       Either an associative array or a string to be sent with the request.
  52.      * @param   array    $headers    An array of request headers to send with the request.
  53.      * @param   integer  $timeout    Read timeout in seconds.
  54.      * @param   string   $userAgent  The optional user agent string to send with the request.
  55.      *
  56.      * @return  JHttpResponse 
  57.      *
  58.      * @since   11.3
  59.      */
  60.     public function request($method, JUri $uri, $data = null, array $headers = null, $timeout = null, $userAgent = null)
  61.     {
  62.         // Setup the cURL handle.
  63.         $ch = curl_init();
  64.  
  65.         // Set the request method.
  66.         $options[CURLOPT_CUSTOMREQUEST] = strtoupper($method);
  67.  
  68.         // Don't wait for body when $method is HEAD
  69.         $options[CURLOPT_NOBODY] = ($method === 'HEAD');
  70.  
  71.         // Initialize the certificate store
  72.         $options[CURLOPT_CAINFO] = $this->options->get('curl.certpath', __DIR__ . '/cacert.pem');
  73.  
  74.         // If data exists let's encode it and make sure our Content-type header is set.
  75.         if (isset($data))
  76.         {
  77.             // If the data is a scalar value simply add it to the cURL post fields.
  78.             if (is_scalar($data) || (isset($headers['Content-Type']) && strpos($headers['Content-Type'], 'multipart/form-data') === 0))
  79.             {
  80.                 $options[CURLOPT_POSTFIELDS] = $data;
  81.             }
  82.             // Otherwise we need to encode the value first.
  83.             else
  84.             {
  85.                 $options[CURLOPT_POSTFIELDS] = http_build_query($data);
  86.             }
  87.  
  88.             if (!isset($headers['Content-Type']))
  89.             {
  90.                 $headers['Content-Type'] = 'application/x-www-form-urlencoded; charset=utf-8';
  91.             }
  92.  
  93.             // Add the relevant headers.
  94.             if (is_scalar($options[CURLOPT_POSTFIELDS]))
  95.             {
  96.                 $headers['Content-Length'] = strlen($options[CURLOPT_POSTFIELDS]);
  97.             }
  98.         }
  99.  
  100.         // Build the headers string for the request.
  101.         $headerArray = array();
  102.         if (isset($headers))
  103.         {
  104.             foreach ($headers as $key => $value)
  105.             {
  106.                 $headerArray[] = $key . ': ' . $value;
  107.             }
  108.  
  109.             // Add the headers string into the stream context options array.
  110.             $options[CURLOPT_HTTPHEADER] = $headerArray;
  111.         }
  112.  
  113.         // If an explicit timeout is given user it.
  114.         if (isset($timeout))
  115.         {
  116.             $options[CURLOPT_TIMEOUT] = (int) $timeout;
  117.             $options[CURLOPT_CONNECTTIMEOUT] = (int) $timeout;
  118.         }
  119.  
  120.         // If an explicit user agent is given use it.
  121.         if (isset($userAgent))
  122.         {
  123.             $headers[CURLOPT_USERAGENT] = $userAgent;
  124.         }
  125.  
  126.         // Set the request URL.
  127.         $options[CURLOPT_URL] = (string) $uri;
  128.  
  129.         // We want our headers. :-)
  130.         $options[CURLOPT_HEADER] = true;
  131.  
  132.         // Return it... echoing it would be tacky.
  133.         $options[CURLOPT_RETURNTRANSFER] = true;
  134.  
  135.         // Override the Expect header to prevent cURL from confusing itself in its own stupidity.
  136.         // Link: http://the-stickman.com/web-development/php-and-curl-disabling-100-continue-header/
  137.         $options[CURLOPT_HTTPHEADER][] = 'Expect:';
  138.  
  139.         // Follow redirects.
  140.         $options[CURLOPT_FOLLOWLOCATION] = (bool) $this->options->get('follow_location', true);
  141.  
  142.         // Set the cURL options.
  143.         curl_setopt_array($ch, $options);
  144.  
  145.         // Execute the request and close the connection.
  146.         $content = curl_exec($ch);
  147.  
  148.         // Get the request information.
  149.         $info = curl_getinfo($ch);
  150.  
  151.         // Close the connection.
  152.         curl_close($ch);
  153.  
  154.         return $this->getResponse($content, $info);
  155.     }

  157.     /**
  158.      * Method to get a response object from a server response.
  159.      *
  160.      * @param   string  $content  The complete server response, including headers.
  161.      * @param   array   $info     The cURL request information.
  162.      *
  163.      * @return  JHttpResponse 
  164.      *
  165.      * @since   11.3
  166.      * @throws  UnexpectedValueException
  167.      */
  168.     protected function getResponse($content, $info)
  169.     {
  170.         // Create the response object.
  171.         $return = new JHttpResponse;
  172.  
  173.         // Check if the content is actually a string.
  174.         if (!is_string($content))
  175.         {
  176.             throw new UnexpectedValueException('No HTTP response received.');
  177.         }
  178.  
  179.         // Get the number of redirects that occurred.
  180.         $redirects = isset($info['redirect_count']) ? $info['redirect_count'] : 0;
  181.  
  182.         /*
  183.          * Split the response into headers and body. If cURL encountered redirects, the headers for the redirected requests will
  184.          * also be included. So we split the response into header + body + the number of redirects and only use the last two
  185.          * sections which should be the last set of headers and the actual body.
  186.          */
  187.         $response = explode("\r\n\r\n", $content, 2 + $redirects);
  188.  
  189.         // Set the body for the response.
  190.         $return->body = array_pop($response);
  191.  
  192.         // Get the last set of response headers as an array.
  193.         $headers = explode("\r\n", array_pop($response));
  194.  
  195.         // Get the response code from the first offset of the response headers.
  196.         preg_match('/[0-9]{3}/', array_shift($headers), $matches);
  197.  
  198.         $code = count($matches) ? $matches[0] : null;
  199.         if (is_numeric($code))
  200.         {
  201.             $return->code = (int) $code;
  202.         }
  203.         // No valid response code was detected.
  204.         else
  205.         {
  206.             throw new UnexpectedValueException('No HTTP response code found.');
  207.         }
  208.  
  209.         // Add the response headers to the response object.
  210.         foreach ($headers as $header)
  211.         {
  212.             $pos = strpos($header, ':');
  213.             $return->headers[trim(substr($header, 0, $pos))] = trim(substr($header, ($pos + 1)));
  214.         }
  215.  
  216.         return $return;
  217.     }

  220.      * Method to check if HTTP transport cURL is available for use

  226.     static public function isSupported()

  229.     }
