  22.      * Find the ftp filesystem root for a given user/pass pair.
  23.      *
  24.      * @param   array  $options  Configuration options.
  25.      *
  26.      * @return  mixed  FTP root for given FTP user, or boolean false if not found.
  27.      *
  28.      * @since   3.1

  30.     public function detectFtpRoot($options)
  31.     {
  32.         // Get the application
  33.         /* @var InstallationApplicationWeb $app */
  34.         $app = JFactory::getApplication();
  35.  
  36.         // Get the options as a object for easier handling.
  37.         $options = JArrayHelper::toObject($options);
  38.  
  39.         // Connect and login to the FTP server.
  40.         // Use binary transfer mode to be able to compare files.
  41.         @$ftp = JClientFtp::getInstance($options->get('ftp_host'), $options->get('ftp_port'), array('type' => FTP_BINARY));
  42.  
  43.         // Check to make sure FTP is connected and authenticated.
  44.         if (!$ftp->isConnected())
  45.         {
  46.             $app->enqueueMessage($options->get('ftp_host') . ':' . $options->get('ftp_port') . ' ' . JText::_('INSTL_FTP_NOCONNECT'), 'error');
  47.             return false;
  48.         }
  49.  
  50.         if (!$ftp->login($options->get('ftp_user'), $options->get('ftp_pass')))
  51.         {
  52.             $app->enqueueMessage(JText::_('INSTL_FTP_NOLOGIN'), 'error');
  53.             return false;
  54.         }
  55.  
  56.         // Get the current working directory from the FTP server.
  57.         $cwd = $ftp->pwd();
  58.         if ($cwd === false)
  59.         {
  60.             $app->enqueueMessage(JText::_('INSTL_FTP_NOPWD'), 'error');
  61.             return false;
  62.         }
  63.         $cwd = rtrim($cwd, '/');
  64.  
  65.         // Get a list of folders in the current working directory.
  66.         $cwdFolders = $ftp->listDetails(null, 'folders');
  67.         if ($cwdFolders === false || count($cwdFolders) == 0)
  68.         {
  69.             $app->enqueueMessage(JText::_('INSTL_FTP_NODIRECTORYLISTING'), 'error');
  70.             return false;
  71.         }
  72.  
  73.         // Get just the folder names from the list of folder data.
  74.         for ($i = 0, $n = count($cwdFolders); $i < $n; $i++)
  75.         {
  76.             $cwdFolders[$i] = $cwdFolders[$i]['name'];
  77.         }
  78.  
  79.         // Check to see if Joomla is installed at the FTP current working directory.
  80.         $paths = array();
  81.         $known = array('administrator', 'components', 'installation', 'language', 'libraries', 'plugins');
  82.         if (count(array_diff($known, $cwdFolders)) == 0)
  83.         {
  84.             $paths[] = $cwd . '/';
  85.         }
  86.  
  87.         // Search through the segments of JPATH_SITE looking for root possibilities.
  88.         $parts = explode(DIRECTORY_SEPARATOR, JPATH_SITE);
  89.         $tmp = '';
  90.         for ($i = count($parts) - 1; $i >= 0; $i--)
  91.         {
  92.             $tmp = '/' . $parts[$i] . $tmp;
  93.             if (in_array($parts[$i], $cwdFolders))
  94.             {
  95.                 $paths[] = $cwd . $tmp;
  96.             }
  97.         }
  98.  
  99.         // Check all possible paths for the real Joomla installation by comparing version files.
  100.         $rootPath = false;
  101.         $checkValue = file_get_contents(JPATH_LIBRARIES . '/cms/version/version.php');
  102.         foreach ($paths as $tmp)
  103.         {
  104.             $filePath = rtrim($tmp, '/') . '/libraries/cms/version/version.php';
  105.             $buffer = null;
  106.             @ $ftp->read($filePath, $buffer);
  107.             if ($buffer == $checkValue)
  108.             {
  109.                 $rootPath = $tmp;
  110.                 break;
  111.             }
  112.         }
  113.  
  114.         // Close the FTP connection.
  115.         $ftp->quit();
  116.  
  117.         // Return an error if no root path was found.
  118.         if ($rootPath === false)
  119.         {
  120.             $app->enqueueMessage(JText::_('INSTL_FTP_UNABLE_DETECT_ROOT_FOLDER'), 'error');
  121.             return false;
  122.         }
  123.  
  124.         return $rootPath;
  125.     }

  127.     /**
  128.      * Verify the FTP settings as being functional and correct.
  129.      *
  130.      * @param   array  $options  Configuration options.
  131.      *
  132.      * @return  mixed  FTP root for given FTP user, or boolean false if not found.
  133.      *
  134.      * @since   3.1
  135.      */
  136.     public function verifyFtpSettings($options)
  137.     {
  138.         // Get the application
  139.         /* @var InstallationApplicationWeb $app */
  140.         $app = JFactory::getApplication();
  141.  
  142.         // Get the options as a object for easier handling.
  143.         $options = JArrayHelper::toObject($options);
  144.  
  145.         // Connect and login to the FTP server.
  146.         @$ftp = JClientFtp::getInstance($options->get('ftp_host'), $options->get('ftp_port'));
  147.  
  148.         // Check to make sure FTP is connected and authenticated.
  149.         if (!$ftp->isConnected())
  150.         {
  151.             $app->enqueueMessage(JText::_('INSTL_FTP_NOCONNECT'), 'error');
  152.             return false;
  153.         }
  154.         if (!$ftp->login($options->get('ftp_user'), $options->get('ftp_pass')))
  155.         {
  156.             $ftp->quit();
  157.             $app->enqueueMessage(JText::_('INSTL_FTP_NOLOGIN'), 'error');
  158.             return false;
  159.         }
  160.  
  161.         // Since the root path will be trimmed when it gets saved to configuration.php,
  162.         // we want to test with the same value as well.
  163.         $root = rtrim($options->get('ftp_root'), '/');
  164.  
  165.         // Verify PWD function
  166.         if ($ftp->pwd() === false)
  167.         {
  168.             $ftp->quit();
  169.             $app->enqueueMessage(JText::_('INSTL_FTP_NOPWD'), 'error');
  170.             return false;
  171.         }
  172.  
  173.         // Verify root path exists
  174.         if (!$ftp->chdir($root))
  175.         {
  176.             $ftp->quit();
  177.             $app->enqueueMessage(JText::_('INSTL_FTP_NOROOT'), 'error');
  178.             return false;
  179.         }
  180.  
  181.         // Verify NLST function
  182.         if (($rootList = $ftp->listNames()) === false)
  183.         {
  184.             $ftp->quit();
  185.             $app->enqueueMessage(JText::_('INSTL_FTP_NONLST'), 'error');
  186.             return false;
  187.         }
  188.  
  189.         // Verify LIST function
  190.         if ($ftp->listDetails() === false)
  191.         {
  192.             $ftp->quit();
  193.             $app->enqueueMessage(JText::_('INSTL_FTP_NOLIST'), 'error');
  194.             return false;
  195.         }
  196.  
  197.         // Verify SYST function
  198.         if ($ftp->syst() === false)
  199.         {
  200.             $ftp->quit();
  201.             $app->enqueueMessage(JText::_('INSTL_FTP_NOSYST'), 'error');
  202.             return false;
  203.         }
  204.  
  205.         // Verify valid root path, part one
  206.         $checkList = array('robots.txt', 'index.php');
  207.         if (count(array_diff($checkList, $rootList)))
  208.         {
  209.             $ftp->quit();
  210.             $app->enqueueMessage(JText::_('INSTL_FTP_INVALIDROOT'), 'error');
  211.             return false;
  212.         }
  213.  
  214.         // Verify RETR function
  215.         $buffer = null;
  216.         if ($ftp->read($root . '/libraries/cms/version/version.php', $buffer) === false)
  217.         {
  218.             $ftp->quit();
  219.             $app->enqueueMessage(JText::_('INSTL_FTP_NORETR'), 'error');
  220.             return false;
  221.         }
  222.  
  223.         // Verify valid root path, part two
  224.         $checkValue = file_get_contents(JPATH_ROOT . '/libraries/cms/version/version.php');
  225.         if ($buffer !== $checkValue)
  226.         {
  227.             $ftp->quit();
  228.             $app->enqueueMessage(JText::_('INSTL_FTP_INVALIDROOT'), 'error');
  229.             return false;
  230.         }
  231.  
  232.         // Verify STOR function
  233.         if ($ftp->create($root . '/ftp_testfile') === false)
  234.         {
  235.             $ftp->quit();
  236.             $app->enqueueMessage(JText::_('INSTL_FTP_NOSTOR'), 'error');
  237.             return false;
  238.         }
  239.  
  240.         // Verify DELE function
  241.         if ($ftp->delete($root . '/ftp_testfile') === false)
  242.         {
  243.             $ftp->quit();
  244.             $app->enqueueMessage(JText::_('INSTL_FTP_NODELE'), 'error');
  245.             return false;
  246.         }
  247.  
  248.         // Verify MKD function
  249.         if ($ftp->mkdir($root . '/ftp_testdir') === false)
  250.         {
  251.             $ftp->quit();
  252.             $app->enqueueMessage(JText::_('INSTL_FTP_NOMKD'), 'error');
  253.             return false;
  254.         }
  255.  
  256.         // Verify RMD function
  257.         if ($ftp->delete($root . '/ftp_testdir') === false)
  258.         {
  259.             $ftp->quit();
  260.             $app->enqueueMessage(JText::_('INSTL_FTP_NORMD'), 'error');
  261.             return false;
  262.         }
  263.  
  264.         $ftp->quit();
  265.         return true;
  266.     }
