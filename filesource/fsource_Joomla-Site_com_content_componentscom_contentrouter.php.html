  69.         {

  72.  
  73.         unset($query['id']);

  76.     }
  77.  
  78.     if ($view == 'category' || $view == 'article')

  101.                     $alias = $db->loadResult();

  104.             }
  105.             else
  106.             {
  107.                 // we should have these two set for this view.  If we don't, it is an error

  109.             }
  110.         }
  111.         else
  112.         {
  113.             if (isset($query['id']))

  116.             }
  117.             else
  118.             {
  119.                 // we should have id set for this view.  If we don't, it is an error
  120.                 return $segments;
  121.             }
  122.         }
  123.  
  124.         if ($menuItemGiven && isset($menuItem->query['id']))
  125.         {
  126.             $mCatid = $menuItem->query['id'];

  128.         else
  129.         {
  130.             $mCatid = 0;
  131.         }
  132.  
  133.         $categories = JCategories::getInstance('Content');
  134.         $category = $categories->get($catid);
  135.  
  136.         if (!$category)
  137.         {
  138.             // we couldn't find the category we were given.  Bail.
  139.             return $segments;
  140.         }
  141.  
  142.         $path = array_reverse($category->getPath());
  143.  
  144.         $array = array();
  145.  
  146.         foreach ($path as $id)
  147.         {

  172.             }
  173.             else
  174.             {
  175.                 $id = $query['id'];
  176.             }
  177.             $segments[] = $id;
  178.         }
  179.         unset($query['id']);
  180.         unset($query['catid']);
  181.     }
  182.  
  183.     if ($view == 'archive')

  187.             $segments[] = $view;
  188.             unset($query['view']);
  189.         }
  190.  
  191.         if (isset($query['year']))
  192.         {
  193.             if ($menuItemGiven)
  194.             {
  195.                 $segments[] = $query['year'];
  196.                 unset($query['year']);
  197.             }
  198.         }
  199.  
  200.         if (isset($query['year']) && isset($query['month']))
  201.         {
  202.             if ($menuItemGiven)
  203.             {
  204.                 $segments[] = $query['month'];
  205.                 unset($query['month']);

  207.         }
  208.     }

  210.     // if the layout is specified and it is the same as the layout in the menu item, we

  212.     if (isset($query['layout']))

  214.         if ($menuItemGiven && isset($menuItem->query['layout']))
  215.         {
  216.             if ($query['layout'] == $menuItem->query['layout'])
  217.             {

  219.             }
  220.         }
  221.         else
  222.         {
  223.             if ($query['layout'] == 'default')

  226.             }
  227.         }
  228.     }
  229.  
  230.     return $segments;

  233. /**
  234.  * Parse the segments of a URL.
  235.  *
  236.  * @return  array  The segments of the URL to parse.
  237.  *
  238.  * @return  array  The URL attributes to be used by the application.
  239.  * @since    1.5
  240.  */
  241. function ContentParseRoute($segments)
  242. {
  243.     $vars = array();
  244.  
  245.     //Get the active menu item.
  246.     $app = JFactory::getApplication();
  247.     $menu = $app->getMenu();
  248.     $item = $menu->getActive();
  249.     $params = JComponentHelper::getParams('com_content');
  250.     $advanced = $params->get('sef_advanced_link', 0);
  251.     $db = JFactory::getDbo();
  252.  
  253.     // Count route segments
  254.     $count = count($segments);
  255.  
  256.     // Standard routing for articles.  If we don't pick up an Itemid then we get the view from the segments
  257.     // the first segment is the view and the last segment is the id of the article or category.
  258.     if (!isset($item))
  259.     {
  260.         $vars['view'] = $segments[0];
  261.         $vars['id'] = $segments[$count - 1];
  262.  
  263.         return $vars;
  264.     }
  265.  
  266.     // if there is only one segment, then it points to either an article or a category
  267.     // we test it first to see if it is a category.  If the id and alias match a category
  268.     // then we assume it is a category.  If they don't we assume it is an article
  269.     if ($count == 1)
  270.     {
  271.         // we check to see if an alias is given.  If not, we assume it is an article
  272.         if (strpos($segments[0], ':') === false)
  273.         {
  274.             $vars['view'] = 'article';
  275.             $vars['id'] = (int) $segments[0];
  276.             return $vars;
  277.         }
  278.  
  279.         list($id, $alias) = explode(':', $segments[0], 2);
  280.  
  281.         // first we check if it is a category
  282.         $category = JCategories::getInstance('Content')->get($id);
  283.  
  284.         if ($category && $category->alias == $alias)
  285.         {
  286.             $vars['view'] = 'category';
  287.             $vars['id'] = $id;
  288.  
  289.             return $vars;
  290.         }
  291.         else
  292.         {
  293.             $query = 'SELECT alias, catid FROM #__content WHERE id = ' . (int) $id;
  294.             $db->setQuery($query);
  295.             $article = $db->loadObject();
  296.  
  297.             if ($article)
  298.             {
  299.                 if ($article->alias == $alias)
  300.                 {
  301.                     $vars['view'] = 'article';
  302.                     $vars['catid'] = (int) $article->catid;
  303.                     $vars['id'] = (int) $id;
  304.  
  305.                     return $vars;
  306.                 }
  307.             }
  308.         }
  309.     }
  310.  
  311.     // if there was more than one segment, then we can determine where the URL points to
  312.     // because the first segment will have the target category id prepended to it.  If the
  313.     // last segment has a number prepended, it is an article, otherwise, it is a category.
  314.     if (!$advanced)
  315.     {
  316.         $cat_id = (int) $segments[0];
  317.  
  318.         $article_id = (int) $segments[$count - 1];
  319.  
  320.         if ($article_id > 0)
  321.         {
  322.             $vars['view'] = 'article';
  323.             $vars['catid'] = $cat_id;
  324.             $vars['id'] = $article_id;
  325.         }
  326.         else
  327.         {
  328.             $vars['view'] = 'category';
  329.             $vars['id'] = $cat_id;
  330.         }
  331.  
  332.         return $vars;
  333.     }
  334.  
  335.     // we get the category id from the menu item and search from there
  336.     $id = $item->query['id'];
  337.     $category = JCategories::getInstance('Content')->get($id);
  338.  
  339.     if (!$category)
  340.     {
  341.         JError::raiseError(404, JText::_('COM_CONTENT_ERROR_PARENT_CATEGORY_NOT_FOUND'));
  342.         return $vars;
  343.     }
  344.  
  345.     $categories = $category->getChildren();
  346.     $vars['catid'] = $id;
  347.     $vars['id'] = $id;
  348.     $found = 0;
  349.  
  350.     foreach ($segments as $segment)
  351.     {
  352.         $segment = str_replace(':', '-', $segment);
  353.  
  354.         foreach ($categories as $category)
  355.         {
  356.             if ($category->alias == $segment)
  357.             {
  358.                 $vars['id'] = $category->id;
  359.                 $vars['catid'] = $category->id;
  360.                 $vars['view'] = 'category';
  361.                 $categories = $category->getChildren();
  362.                 $found = 1;
  363.                 break;
  364.             }
  365.         }
  366.  
  367.         if ($found == 0)
  368.         {
  369.             if ($advanced)
  370.             {
  371.                 $db = JFactory::getDbo();
  372.                 $query = $db->getQuery(true)
  373.                     ->select($db->quoteName('id'))
  374.                     ->from('#__content')
  375.                     ->where($db->quoteName('catid') . ' = ' . (int) $vars['catid'])
  376.                     ->where($db->quoteName('alias') . ' = ' . $db->quote($db->quote($segment)));
  377.                 $db->setQuery($query);
  378.                 $cid = $db->loadResult();
  379.             }
  380.             else
  381.             {
  382.                 $cid = $segment;
  383.             }
  384.  
  385.             $vars['id'] = $cid;
  386.  
  387.             if ($item->query['view'] == 'archive' && $count != 1)
  388.             {
  389.                 $vars['year'] = $count >= 2 ? $segments[$count - 2] : null;
  390.                 $vars['month'] = $segments[$count - 1];
  391.                 $vars['view'] = 'archive';
  392.             }
  393.             else
  394.             {
  395.                 $vars['view'] = 'article';
  396.             }
  397.         }
  398.  
  399.         $found = 0;
  400.     }
  401.  
  402.     return $vars;
  403. }
