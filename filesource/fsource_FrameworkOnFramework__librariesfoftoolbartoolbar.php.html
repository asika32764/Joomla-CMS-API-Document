  218.     public function renderToolbar($view = null, $task = null, $input = null)

  272.  
  273.         // Check for an onViewTask method
  274.         $methodName = 'on' . ucfirst($view) . ucfirst($task);
  275.

  494.      * @param   string|null $icon    Icon class (used by some renderers, like the Bootstrap renderer)
  495.      * @param   string|null $parent  The parent element (referenced by name)) Thsi will create a dropdown list
  496.      *
  497.      * @return  void 
  498.      */
  499.     public function appendLink($name, $link = null, $active = false, $icon = null, $parent = '')
  500.     {
  501.         $linkDefinition = array(
  502.             'name'     => $name,
  503.             'link'     => $link,
  504.             'active' => $active,
  505.             'icon'     => $icon
  506.         );
  507.  
  508.         if (empty($parent))
  509.         {
  510.             $this->linkbar[$name] = $linkDefinition;
  511.         }
  512.         else
  513.         {
  514.             if (!array_key_exists($parent, $this->linkbar))
  515.             {
  516.                 $parentElement = $linkDefinition;
  517.                 $parentElement['link'] = null;
  518.                 $this->linkbar[$parent] = $parentElement;
  519.                 $parentElement['items'] = array();
  520.             }
  521.             else
  522.             {
  523.                 $parentElement = $this->linkbar[$parent];
  524.  
  525.                 if (!array_key_exists('dropdown', $parentElement) && !empty($parentElement['link']))
  526.                 {
  527.                     $newSubElement = $parentElement;
  528.                     $parentElement['items'] = array($newSubElement);
  529.                 }
  530.             }
  531.  
  532.             $parentElement['items'][] = $linkDefinition;
  533.             $parentElement['dropdown'] = true;
  534.  
  535.             $this->linkbar[$parent] = $parentElement;
  536.         }
  537.     }
  538.  
  539.     /**
  540.      * Prefixes (some people erroneously call this "prepend" – there is no such word) a link to the link bar
  541.      *
  542.      * @param   string       $name    The text of the link
  543.      * @param   string|null $link    The link to render; set to null to render a separator
  544.      * @param   boolean      $active  True if it's an active link
  545.      * @param   string|null $icon    Icon class (used by some renderers, like the Bootstrap renderer)
  546.      *
  547.      * @return  void 
  548.      */
  549.     public function prefixLink($name, $link = null, $active = false, $icon = null)
  550.     {
  551.         $linkDefinition = array(
  552.             'name'     => $name,

  560.     /**
  561.      * Renders the submenu (toolbar links) for all detected views of this component
  562.      *
  563.      * @return  void 
  564.      */
  565.     public function renderSubmenu()
  566.     {
  567.         $views = $this->getMyViews();
  568.  
  569.         if (empty($views))
  570.         {
  571.             return;
  572.         }
  573.  
  574.         $activeView = $this->input->getCmd('view', 'cpanel');
  575.  
  576.         foreach ($views as $view)
  577.         {
  578.             // Get the view name
  579.             $key = strtoupper($this->component) . '_TITLE_' . strtoupper($view);
  580.  
  581.             if (strtoupper(JText::_($key)) == $key)
  582.             {
  583.                 $altview = FOFInflector::isPlural($view) ? FOFInflector::singularize($view) : FOFInflector::pluralize($view);
  584.                 $key2 = strtoupper($this->component) . '_TITLE_' . strtoupper($altview);
  585.  
  586.                 if (strtoupper(JText::_($key2)) == $key2)
  587.                 {
  588.                     $name = ucfirst($view);
  589.                 }
  590.                 else
  591.                 {
  592.                     $name = JText::_($key2);
  593.                 }
  594.             }
  595.             else
  596.             {
  597.                 $name = JText::_($key);
  598.             }
  599.  
  600.             $link = 'index.php?option=' . $this->component . '&view=' . $view;
  601.  
  602.             $active = $view == $activeView;
  603.  
  604.             $this->appendLink($name, $link, $active);
  605.         }
  606.     }
  607.  
  608.     /**
  609.      * Automatically detects all views of the component
  610.      *
  611.      * @return  array  A list of all views, in the order to be displayed in the toolbar submenu
  612.      */
  613.     protected function getMyViews()
  614.     {
  615.         $views = array();
  616.         $t_views = array();
  617.         $using_meta = false;
  618.  
  619.         $componentPaths = FOFPlatform::getInstance()->getComponentBaseDirs($this->component);
  620.         $searchPath = $componentPaths['main'] . '/views';
  621.  
  622.         JLoader::import('joomla.filesystem.folder');
  623.         JLoader::import('joomla.utilities.arrayhelper');
  624.  
  625.         $allFolders = JFolder::folders($searchPath);
  626.  
  627.         if (!empty($allFolders))
  628.         {
  629.             foreach ($allFolders as $folder)
  630.             {
  631.                 $view = $folder;
  632.  
  633.                 // View already added
  634.                 if (in_array(FOFInflector::pluralize($view), $t_views))
  635.                 {
  636.                     continue;
  637.                 }
  638.  
  639.                 // Do we have a 'skip.xml' file in there?
  640.                 $files = JFolder::files($searchPath . '/' . $view, '^skip\.xml$');
  641.  
  642.                 if (!empty($files))
  643.                 {
  644.                     continue;
  645.                 }
  646.  
  647.                 // Do we have extra information about this view? (ie. ordering)
  648.                 $meta = JFolder::files($searchPath . '/' . $view, '^metadata\.xml$');
  649.  
  650.                 // Not found, do we have it inside the plural one?
  651.                 if (!$meta)
  652.                 {
  653.                     $plural = FOFInflector::pluralize($view);
  654.  
  655.                     if (in_array($plural, $allFolders))
  656.                     {
  657.                         $view = $plural;
  658.                         $meta = JFolder::files($searchPath . '/' . $view, '^metadata\.xml$');
  659.                     }
  660.                 }
  661.  
  662.                 if (!empty($meta))
  663.                 {
  664.                     $using_meta = true;
  665.                     $xml = simplexml_load_file($searchPath . '/' . $view . '/' . $meta[0]);
  666.                     $order = (int) $xml->foflib->ordering;
  667.                 }
  668.                 else
  669.                 {
  670.                     // Next place. It's ok since the index are 0-based and count is 1-based
  671.  
  672.                     if (!isset($to_order))
  673.                     {
  674.                         $to_order = array();
  675.                     }
  676.  
  677.                     $order = count($to_order);
  678.                 }
  679.  
  680.                 $view = FOFInflector::pluralize($view);
  681.  
  682.                 $t_view = new stdClass;
  683.                 $t_view->ordering = $order;
  684.                 $t_view->view = $view;
  685.  
  686.                 $to_order[] = $t_view;
  687.                 $t_views[] = $view;
  688.             }
  689.         }
  690.  
  691.         JArrayHelper::sortObjects($to_order, 'ordering');
  692.         $views = JArrayHelper::getColumn($to_order, 'view');
  693.  
  694.         // If not using the metadata file, let's put the cpanel view on top
  695.         if (!$using_meta)
  696.         {
  697.             $cpanel = array_search('cpanels', $views);
  698.  
  699.             if ($cpanel !== false)
  700.             {
  701.                 unset($views[$cpanel]);
  702.                 array_unshift($views, 'cpanels');
  703.             }
  704.         }
  705.  
  706.         return $views;
  707.     }

  831.  
  832.             case 'assign':

  834.                 $alt = isset($attributes['alt']) ? $attributes['alt'] : 'JTOOLBAR_ASSIGN';
  835.

  841.                 {

  880.                     $alt = isset($attributes['alt']) ? $attributes['alt'] : 'JTOOLBAR_UNPUBLISH';
