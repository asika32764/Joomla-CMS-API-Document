  216.     /**
  217.      * Calculate the size of a resized image
  218.      *
  219.      * @param   integer  $width   Image width
  220.      * @param   integer  $height  Image height

  225.      * @since   3.2

  227.     public static function imageResize($width, $height, $target)
  228.     {
  229.         /*
  230.          * Takes the larger size of the width and height and applies the
  231.          * formula accordingly. This is so this script will work
  232.          * dynamically with any size image
  233.          */
  234.         if ($width > $height)
  235.         {
  236.             $percentage = ($target / $width);
  237.         }
  238.         else
  239.         {
  240.             $percentage = ($target / $height);
  241.         }
  242.  
  243.         // Gets the new value and applies the percentage, then rounds the value
  244.         $width  = round($width * $percentage);
  245.         $height = round($height * $percentage);
  246.  
  247.         return array($width, $height);
  248.     }

  277.                 {
