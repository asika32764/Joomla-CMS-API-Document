  81. /**
  82.  * Renders the pagination list
  83.  *
  84.  * @param   array  $list  Array containing pagination information
  85.  *
  86.  * @return  string  HTML markup for the full pagination object

  89.  */
  90. function pagination_list_render($list)
  91. {
  92.     // Calculate to display range of pages
  93.     $currentPage = 1;
  94.     $range = 1;
  95.     $step = 5;
  96.     foreach ($list['pages'] as $k => $page)
  97.     {
  98.         if (!$page['active'])
  99.         {
  100.             $currentPage = $k;
  101.         }
  102.     }
  103.     if ($currentPage >= $step)
  104.     {
  105.         if ($currentPage % $step == 0)
  106.         {
  107.             $range = ceil($currentPage / $step) + 1;
  108.         }
  109.         else
  110.         {
  111.             $range = ceil($currentPage / $step);
  112.         }
  113.     }
  114.  
  115.     $html = '<ul class="pagination-list">';
  116.     $html .= $list['start']['data'];
  117.     $html .= $list['previous']['data'];
  118.  
  119.     foreach ($list['pages'] as $k => $page)
  120.     {
  121.         if (in_array($k, range($range * $step - ($step + 1), $range * $step)))
  122.         {
  123.             if (($k % $step == 0 || $k == $range * $step - ($step + 1)) && $k != $currentPage && $k != $range * $step - $step)
  124.             {
  125.                 $page['data'] = preg_replace('#(<a.*?>).*?(</a>)#', '$1...$2', $page['data']);
  126.             }
  127.         }
  128.  
  129.         $html .= $page['data'];
  130.     }
  131.  
  132.     $html .= $list['next']['data'];
  133.     $html .= $list['end']['data'];
  134.  
  135.     $html .= '</ul>';
  136.     return $html;
  137. }
