  14. /**
  15.  * Checks if the plugin is enabled. If not it returns true, meaning that the
  16.  * message concerning two factor authentication should be displayed.
  17.  *
  18.  * @return  integer 
  19.  *
  20.  * @since   3.2
  21.  */
  22. function twofactorauth_postinstall_condition()
  23. {
  24.     $db = JFactory::getDbo();
  25.  
  26.     $query = $db->getQuery(true)
  27.         ->select('*')
  28.         ->from($db->qn('#__extensions'))
  29.         ->where($db->qn('type') . ' = ' . $db->q('plugin'))
  30.         ->where($db->qn('enabled') . ' = ' . $db->q('1'))
  31.         ->where($db->qn('folder') . ' = ' . $db->q('twofactorauth'));
  32.     $db->setQuery($query);
  33.     $enabled_plugins = $db->loadObjectList();
  34.  
  35.     return count($enabled_plugins) == 0;
  36. }
  37.  
  38. /**
  39.  * Enables the two factor authentication plugin and redirects the user to their
  40.  * user profile page so that they can enable two factor authentication on their
  41.  * account.
  42.  *
  43.  * @return  void 
  44.  *
  45.  * @since   3.2
  46.  */
  47. function twofactorauth_postinstall_action()
  48. {
  49.     // Enable the plugin
  50.     $db = JFactory::getDbo();
  51.  
  52.     $query = $db->getQuery(true)
  53.         ->update($db->qn('#__extensions'))
  54.         ->set($db->qn('enabled') . ' = ' . $db->q(1))
  55.         ->where($db->qn('type') . ' = ' . $db->q('plugin'))
  56.         ->where($db->qn('folder') . ' = ' . $db->q('twofactorauth'));
  57.     $db->setQuery($query);
  58.     $db->execute();
  59.  
  60.     // Redirect the user to their profile editor page
  61.     $url = 'index.php?option=com_users&task=user.edit&id=' . JFactory::getUser()->id;
  62.     JFactory::getApplication()->redirect($url);
  63. }
