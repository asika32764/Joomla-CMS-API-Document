  63.     public $phrase;

  71.     public $length;

  99.     public function __construct($term, $lang, $spacer = ' ')
  100.     {
  101.         $this->language = $lang;
  102.  
  103.         // Tokens can be a single word or an array of words representing a phrase.
  104.         if (is_array($term))
  105.         {
  106.             // Populate the token instance.
  107.             $this->term = implode($spacer, $term);
  108.             $this->stem = implode($spacer, array_map(array('FinderIndexerHelper', 'stem'), $term, array($lang)));
  109.             $this->numeric = false;
  110.             $this->common = false;
  111.             $this->phrase = true;
  112.             $this->length = JString::strlen($this->term);
  113.  
  114.             /*
  115.              * Calculate the weight of the token.
  116.              *
  117.              * 1. Length of the token up to 30 and divide by 30, add 1.
  118.              * 2. Round weight to 4 decimal points.
  119.              */
  120.             $this->weight = (($this->length >= 30 ? 30 : $this->length) / 30) + 1;
  121.             $this->weight = round($this->weight, 4);
  122.         }
  123.         else
  124.         {
  125.             // Populate the token instance.
  126.             $this->term = $term;
  127.             $this->stem = FinderIndexerHelper::stem($this->term, $lang);
  128.             $this->numeric = (is_numeric($this->term) || (bool) preg_match('#^[0-9,.\-\+]+$#', $this->term));
  129.             $this->common = $this->numeric ? false : FinderIndexerHelper::isCommon($this->term, $lang);
  130.             $this->phrase = false;
  131.             $this->length = JString::strlen($this->term);
  132.  
  133.             /*
  134.              * Calculate the weight of the token.
  135.              *
  136.              * 1. Length of the token up to 15 and divide by 15.
  137.              * 2. If common term, divide weight by 8.
  138.              * 3. If numeric, multiply weight by 1.5.
  139.              * 4. Round weight to 4 decimal points.
  140.              */
  141.             $this->weight = (($this->length >= 15 ? 15 : $this->length) / 15);
  142.             $this->weight = ($this->common == true ? $this->weight / 8 : $this->weight);
  143.             $this->weight = ($this->numeric == true ? $this->weight * 1.5 : $this->weight);
  144.             $this->weight = round($this->weight, 4);
  145.         }
  146.     }
