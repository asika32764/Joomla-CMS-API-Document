  320.         if ($msg)

  327.             switch ($step['type'])

  329.                 case 'file':

  331.                     $stepval = JFile::delete($step['path']);

  333.  
  334.                 case 'folder':
  335.                     // Remove the folder
  336.                     $stepval = JFolder::delete($step['path']);
  337.                     break;
  338.  
  339.                 case 'query':
  340.                     // Placeholder in case this is necessary in the future
  341.                     // $stepval is always false because if this step was called it invariably failed
  342.                     $stepval = false;
  343.                     break;
  344.  
  345.                 case 'extension':

  348.                     $query = $db->getQuery(true);
  349.  
  350.                     // Remove the entry from the #__extensions table
  351.                     $query->delete($db->quoteName('#__extensions'))
  352.                         ->where($db->quoteName('extension_id') . ' = ' . (int) $step['id']);
  353.                     $db->setQuery($query);
  354.                     $stepval = $db->execute();
  355.  
  356.                     break;
  357.  
  358.                 default:
  359.                     if ($type && is_object($this->_adapters[$type]))
  360.                     {
  361.                         // Build the name of the custom rollback method for the type
  362.                         $method = '_rollback_' . $step['type'];
  363.  
  364.                         // Custom rollback method handler
  365.                         if (method_exists($this->_adapters[$type], $method))
  366.                         {
  367.                             $stepval = $this->_adapters[$type]->$method($step);
  368.                         }
  369.                     }
  370.                     else
  371.                     {
  372.                         // Set it to false
  373.                         $stepval = false;
  374.                     }
  375.                     break;
  376.             }
  377.  
  378.             // Only set the return value if it is false
  379.             if ($stepval === false)
  380.             {
  381.                 $retval = false;
  382.             }
  383.  
  384.             // Get the next step and continue
  385.             $step = array_pop($this->stepStack);
  386.         }
  387.  
  388.         $conf = JFactory::getConfig();
  389.         $debug = $conf->get('debug');
  390.  
  391.         if ($debug)
  392.         {
  393.             throw new RuntimeException('Installation unexpectedly terminated: ' . $msg, 500);
  394.         }
  395.  
  396.         return $retval;
  397.     }

  529.                     );
  530.  
  531.                     // Run the install

  538.                     );
  539.  
  540.                     if ($result !== false)
  541.                     {

  547.                     }
  548.                 }

  553.                     return false;

  555.             }
  556.  
  557.             return false;
  558.         }
  559.         else
