  57.     /**
  58.      * Custom loadLanguage method
  59.      *

  66.     public function loadLanguage($path = null)
  67.     {
  68.         $source = $this->parent->getPath('source');
  69.  
  70.         if (!$source)
  71.         {
  72.             $this->parent
  73.                 ->setPath(
  74.                 'source',
  75.                 ($this->parent->extension->client_id ? JPATH_ADMINISTRATOR : JPATH_SITE) . '/templates/' . $this->parent->extension->element
  76.             );
  77.         }
  78.  
  79.         $this->manifest = $this->parent->getManifest();
  80.         $name = strtolower(JFilterInput::getInstance()->clean((string) $this->manifest->name, 'cmd'));
  81.         $client = (string) $this->manifest->attributes()->client;
  82.  
  83.         // Load administrator language if not set.
  84.         if (!$client)
  85.         {
  86.             $client = 'ADMINISTRATOR';
  87.         }
  88.  
  89.         $extension = "tpl_$name";
  90.         $lang = JFactory::getLanguage();
  91.         $source = $path ? $path : ($this->parent->extension->client_id ? JPATH_ADMINISTRATOR : JPATH_SITE) . '/templates/' . $name;
  92.         $lang->load($extension . '.sys', $source, null, false, true)
  93.             || $lang->load($extension . '.sys', constant('JPATH_' . strtoupper($client)), null, false, true);
  94.     }
  95.  
  96.     /**
  97.      * Custom install method
  98.      *
  99.      * @return  boolean  True on success
  100.      *
  101.      * @since   3.1
  102.      */
  103.     public function install()
  104.     {
  105.         // Get a database connector object
  106.         $db = $this->parent->getDbo();
  107.  
  108.         $lang = JFactory::getLanguage();
  109.         $xml = $this->parent->getManifest();
  110.  
  111.         // Get the client application target
  112.         if ($cname = (string) $xml->attributes()->client)
  113.         {
  114.             // Attempt to map the client to a base path
  115.             $client = JApplicationHelper::getClientInfo($cname, true);
  116.  
  117.             if ($client === false)
  118.             {
  119.                 $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_TPL_INSTALL_UNKNOWN_CLIENT', $cname));
  120.  
  121.                 return false;
  122.             }
  123.             $basePath = $client->path;
  124.             $clientId = $client->id;
  125.         }
  126.         else
  127.         {
  128.             // No client attribute was found so we assume the site as the client
  129.             $basePath = JPATH_SITE;
  130.             $clientId = 0;
  131.         }
  132.  
  133.         // Set the extension's name
  134.         $name = JFilterInput::getInstance()->clean((string) $xml->name, 'cmd');
  135.  
  136.         $element = strtolower(str_replace(" ", "_", $name));
  137.         $this->set('name', $name);
  138.         $this->set('element', $element);
  139.  
  140.         // Check to see if a template by the same name is already installed.
  141.         $query = $db->getQuery(true)
  142.             ->select($db->quoteName('extension_id'))
  143.             ->from($db->quoteName('#__extensions'))
  144.             ->where($db->quoteName('type') . ' = ' . $db->quote('template'))
  145.             ->where($db->quoteName('element') . ' = ' . $db->quote($element));
  146.         $db->setQuery($query);
  147.  
  148.         try
  149.         {
  150.             $id = $db->loadResult();
  151.         }
  152.         catch (RuntimeException $e)
  153.         {
  154.             // Install failed, roll back changes
  155.             $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_TPL_INSTALL_ROLLBACK'), $e->getMessage());
  156.  
  157.             return false;
  158.         }
  159.  
  160.         // Set the template root path
  161.         $this->parent->setPath('extension_root', $basePath . '/templates/' . $element);
  162.  
  163.         // If it's on the fs...
  164.         if (file_exists($this->parent->getPath('extension_root')) && (!$this->parent->isOverwrite() || $this->parent->isUpgrade()))
  165.         {
  166.             $updateElement = $xml->update;
  167.  
  168.             // Upgrade manually set or update tag detected
  169.             if ($this->parent->isUpgrade() || $updateElement)
  170.             {
  171.                 // Force this one
  172.                 $this->parent->setOverwrite(true);
  173.                 $this->parent->setUpgrade(true);
  174.  
  175.                 if ($id)
  176.                 {
  177.                     // If there is a matching extension mark this as an update; semantics really
  178.                     $this->route = 'update';
  179.                 }
  180.             }
  181.             elseif (!$this->parent->isOverwrite())
  182.             {
  183.                 // Overwrite is not set
  184.                 // If we didn't have overwrite set, find an update function or find an update tag so let's call it safe
  185.                 $this->parent
  186.                     ->abort(
  187.                     JText::sprintf(
  188.                         'JLIB_INSTALLER_ABORT_TPL_INSTALL_ANOTHER_TEMPLATE_USING_DIRECTORY', JText::_('JLIB_INSTALLER_' . $this->route),
  189.                         $this->parent->getPath('extension_root')
  190.                     )
  191.                 );
  192.  
  193.                 return false;
  194.             }
  195.         }
  196.  
  197.         /*
  198.          * If the template directory already exists, then we will assume that the template is already
  199.          * installed or another template is using that directory.
  200.          */
  201.         if (file_exists($this->parent->getPath('extension_root')) && !$this->parent->isOverwrite())
  202.         {
  203.             JLog::add(
  204.                 JText::sprintf('JLIB_INSTALLER_ABORT_TPL_INSTALL_ANOTHER_TEMPLATE_USING_DIRECTORY', $this->parent->getPath('extension_root')),
  205.                 JLog::WARNING, 'jerror'
  206.             );
  207.  
  208.             return false;
  209.         }
  210.  
  211.         // If the template directory does not exist, let's create it
  212.         $created = false;
  213.  
  214.         if (!file_exists($this->parent->getPath('extension_root')))
  215.         {
  216.             if (!$created = JFolder::create($this->parent->getPath('extension_root')))
  217.             {
  218.                 $this->parent
  219.                     ->abort(JText::sprintf('JLIB_INSTALLER_ABORT_TPL_INSTALL_FAILED_CREATE_DIRECTORY', $this->parent->getPath('extension_root')));
  220.  
  221.                 return false;
  222.             }
  223.         }
  224.  
  225.         // If we created the template directory and will want to remove it if we have to roll back
  226.         // the installation, let's add it to the installation step stack
  227.         if ($created)
  228.         {
  229.             $this->parent->pushStep(array('type' => 'folder', 'path' => $this->parent->getPath('extension_root')));
  230.         }
  231.  
  232.         // Copy all the necessary files
  233.         if ($this->parent->parseFiles($xml->files, -1) === false)
  234.         {
  235.             // Install failed, rollback changes
  236.             $this->parent->abort();
  237.  
  238.             return false;
  239.         }
  240.  
  241.         if ($this->parent->parseFiles($xml->images, -1) === false)
  242.         {
  243.             // Install failed, rollback changes
  244.             $this->parent->abort();
  245.  
  246.             return false;
  247.         }
  248.  
  249.         if ($this->parent->parseFiles($xml->css, -1) === false)
  250.         {
  251.             // Install failed, rollback changes
  252.             $this->parent->abort();
  253.  
  254.             return false;
  255.         }
  256.  
  257.         // Parse optional tags
  258.         $this->parent->parseMedia($xml->media);
  259.         $this->parent->parseLanguages($xml->languages, $clientId);
  260.  
  261.         // Get the template description
  262.         $this->parent->set('message', JText::_((string) $xml->description));
  263.  
  264.         // Lastly, we will copy the manifest file to its appropriate place.
  265.         if (!$this->parent->copyManifest(-1))
  266.         {
  267.             // Install failed, rollback changes
  268.             $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_TPL_INSTALL_COPY_SETUP'));
  269.  
  270.             return false;
  271.         }
  272.  
  273.         /**
  274.          * ---------------------------------------------------------------------------------------------
  275.          * Extension Registration
  276.          * ---------------------------------------------------------------------------------------------
  277.          */
  278.  
  279.         $row = JTable::getInstance('extension');
  280.  
  281.         if ($this->route == 'update' && $id)
  282.         {
  283.             $row->load($id);
  284.         }
  285.         else
  286.         {
  287.             $row->type = 'template';
  288.             $row->element = $this->get('element');
  289.  
  290.             // There is no folder for templates
  291.             $row->folder = '';
  292.             $row->enabled = 1;
  293.             $row->protected = 0;
  294.             $row->access = 1;
  295.             $row->client_id = $clientId;
  296.             $row->params = $this->parent->getParams();
  297.  
  298.             // Custom data
  299.             $row->custom_data = '';
  300.         }
  301.  
  302.         // Name might change in an update
  303.         $row->name = $this->get('name');
  304.         $row->manifest_cache = $this->parent->generateManifestCache();
  305.  
  306.         if (!$row->store())
  307.         {
  308.             // Install failed, roll back changes
  309.             $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_TPL_INSTALL_ROLLBACK', $db->stderr(true)));
  310.  
  311.             return false;
  312.         }
  313.  
  314.         if ($this->route == 'install')
  315.         {
  316.             $debug = $lang->setDebug(false);
  317.  
  318.             $columns = array($db->quoteName('template'),
  319.                 $db->quoteName('client_id'),
  320.                 $db->quoteName('home'),
  321.                 $db->quoteName('title'),
  322.                 $db->quoteName('params')
  323.             );
  324.  
  325.             $values = array(
  326.                 $db->quote($row->element), $clientId, $db->quote(0),
  327.                 $db->quote(JText::sprintf('JLIB_INSTALLER_DEFAULT_STYLE', JText::_($this->get('name')))),
  328.                 $db->quote($row->params) );
  329.  
  330.             $lang->setDebug($debug);
  331.  
  332.             // Insert record in #__template_styles
  333.             $query->clear()
  334.                 ->insert($db->quoteName('#__template_styles'))
  335.                 ->columns($columns)
  336.                 ->values(implode(',', $values));
  337.  
  338.             $db->setQuery($query);
  339.  
  340.             // There is a chance this could fail but we don't care...
  341.             $db->execute();
  342.         }
  343.  
  344.         return $row->get('extension_id');
  345.     }
  346.  
  347.     /**
  348.      * Custom update method for components
  349.      *
  350.      * @return  boolean  True on success
  351.      *
  352.      * @since   3.1
  353.      */
  354.     public function update()
  355.     {
  356.         $this->route = 'update';
  357.  
  358.         return $this->install();
  359.     }

  361.     /**
  362.      * Custom uninstall method
  363.      *
  364.      * @param   integer  $id  The extension ID
  365.      *
  366.      * @return  boolean  True on success
  367.      *
  368.      * @since   3.1
  369.      */
  370.     public function uninstall($id)
  371.     {
  372.         // First order of business will be to load the template object table from the database.
  373.         // This should give us the necessary information to proceed.
  374.         $row = JTable::getInstance('extension');
  375.  
  376.         if (!$row->load((int) $id) || !strlen($row->element))
  377.         {
  378.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_TPL_UNINSTALL_ERRORUNKOWNEXTENSION'), JLog::WARNING, 'jerror');
  379.  
  380.             return false;
  381.         }
  382.  
  383.         // Is the template we are trying to uninstall a core one?
  384.         // Because that is not a good idea...
  385.         if ($row->protected)
  386.         {
  387.             JLog::add(JText::sprintf('JLIB_INSTALLER_ERROR_TPL_UNINSTALL_WARNCORETEMPLATE', $row->name), JLog::WARNING, 'jerror');
  388.  
  389.             return false;
  390.         }
  391.  
  392.         $name = $row->element;
  393.         $clientId = $row->client_id;
  394.  
  395.         // For a template the id will be the template name which represents the subfolder of the templates folder that the template resides in.
  396.         if (!$name)
  397.         {
  398.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_TPL_UNINSTALL_TEMPLATE_ID_EMPTY'), JLog::WARNING, 'jerror');
  399.  
  400.             return false;
  401.         }
  402.  
  403.         // Deny remove default template
  404.         $db = $this->parent->getDbo();
  405.         $query = "SELECT COUNT(*) FROM #__template_styles WHERE home = '1' AND template = " . $db->quote($name);
  406.         $db->setQuery($query);
  407.  
  408.         if ($db->loadResult() != 0)
  409.         {
  410.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_TPL_UNINSTALL_TEMPLATE_DEFAULT'), JLog::WARNING, 'jerror');
  411.  
  412.             return false;
  413.         }
  414.  
  415.         // Get the template root path
  416.         $client = JApplicationHelper::getClientInfo($clientId);
  417.  
  418.         if (!$client)
  419.         {
  420.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_TPL_UNINSTALL_INVALID_CLIENT'), JLog::WARNING, 'jerror');
  421.  
  422.             return false;
  423.         }
  424.  
  425.         $this->parent->setPath('extension_root', $client->path . '/templates/' . strtolower($name));
  426.         $this->parent->setPath('source', $this->parent->getPath('extension_root'));
  427.  
  428.         // We do findManifest to avoid problem when uninstalling a list of extensions: getManifest cache its manifest file
  429.         $this->parent->findManifest();
  430.         $manifest = $this->parent->getManifest();
  431.  
  432.         if (!($manifest instanceof SimpleXMLElement))
  433.         {
  434.             // Kill the extension entry
  435.             $row->delete($row->extension_id);
  436.             unset($row);
  437.  
  438.             // Make sure we delete the folders
  439.             JFolder::delete($this->parent->getPath('extension_root'));
  440.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_TPL_UNINSTALL_INVALID_NOTFOUND_MANIFEST'), JLog::WARNING, 'jerror');
  441.  
  442.             return false;
  443.         }
  444.  
  445.         // Remove files
  446.         $this->parent->removeFiles($manifest->media);
  447.         $this->parent->removeFiles($manifest->languages, $clientId);
  448.  
  449.         // Delete the template directory
  450.         if (JFolder::exists($this->parent->getPath('extension_root')))
  451.         {
  452.             $retval = JFolder::delete($this->parent->getPath('extension_root'));
  453.         }
  454.         else
  455.         {
  456.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_TPL_UNINSTALL_TEMPLATE_DIRECTORY'), JLog::WARNING, 'jerror');
  457.             $retval = false;
  458.         }
  459.  
  460.         // Set menu that assigned to the template back to default template
  461.         $query = 'UPDATE #__menu'
  462.             . ' SET template_style_id = 0'
  463.             . ' WHERE template_style_id in ('
  464.             . '    SELECT s.id FROM #__template_styles s'
  465.             . ' WHERE s.template = ' . $db->quote(strtolower($name)) . ' AND s.client_id = ' . $clientId . ')';
  466.  
  467.         $db->setQuery($query);
  468.         $db->execute();
  469.  
  470.         $query = 'DELETE FROM #__template_styles WHERE template = ' . $db->quote($name) . ' AND client_id = ' . $clientId;
  471.         $db->setQuery($query);
  472.         $db->execute();
  473.
