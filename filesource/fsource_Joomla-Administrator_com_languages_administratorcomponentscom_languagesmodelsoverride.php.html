  118.     public function save($data, $opposite_client = false)
  119.     {
  120.         $app = JFactory::getApplication();
  121.         require_once JPATH_COMPONENT.'/helpers/languages.php';
  122.         jimport('joomla.filesystem.file');
  123.  
  124.         $client        = $app->getUserState('com_languages.overrides.filter.client', 0);
  125.         $language    = $app->getUserState('com_languages.overrides.filter.language', 'en-GB');
  126.  
  127.         // If the override should be created for both
  128.         if ($opposite_client)
  129.         {
  130.             $client = 1 - $client;
  131.         }
  132.  
  133.         $client = $client ? 'administrator' : 'site';
  134.  
  135.         // Parse the override.ini file in oder to get the keys and strings
  136.         $filename    = constant('JPATH_'.strtoupper($client)) . '/language/overrides/' . $language . '.override.ini';
  137.         $strings    = LanguagesHelper::parseFile($filename);
  138.  
  139.         if (isset($strings[$data['id']]))
  140.         {
  141.             // If an existent string was edited check whether
  142.             // the name of the constant is still the same
  143.             if ($data['key'] == $data['id'])
  144.             {
  145.                 // If yes, simply override it
  146.                 $strings[$data['key']] = $data['override'];
  147.             }
  148.             else
  149.             {
  150.                 // If no, delete the old string and prepend the new one
  151.                 unset($strings[$data['id']]);
  152.                 $strings = array($data['key'] => $data['override']) + $strings;
  153.             }
  154.         }
  155.         else
  156.         {
  157.             // If it is a new override simply prepend it
  158.             $strings = array($data['key'] => $data['override']) + $strings;
  159.         }
  160.  
  161.         foreach ($strings as $key => $string)
  162.         {
  163.             $strings[$key] = str_replace('"', '"_QQ_"', $string);
  164.         }
  165.  
  166.         // Write override.ini file with the strings
  167.         $registry = new JRegistry;
  168.         $registry->loadObject($strings);
  169.         $reg = $registry->toString('INI');
  170.  
  171.         if (!JFile::write($filename, $reg))
  172.         {
  173.             return false;
  174.         }
  175.  
  176.         // If the override should be stored for both clients save
  177.         // it also for the other one and prevent endless recursion
  178.         if (isset($data['both']) && $data['both'] && !$opposite_client)
  179.         {
  180.             return $this->save($data, true);
  181.         }
  182.  
  183.         return true;
