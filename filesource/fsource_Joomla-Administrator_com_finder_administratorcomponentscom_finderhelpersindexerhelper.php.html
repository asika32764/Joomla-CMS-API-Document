  51.     /**
  52.      * Method to tokenize a text string.
  53.      *
  54.      * @param   string   $input   The input to tokenize.
  55.      * @param   string   $lang    The language of the input.
  56.      * @param   boolean  $phrase  Flag to indicate whether input could be a phrase. [optional]
  57.      *
  58.      * @return  array  An array of FinderIndexerToken objects.
  59.      *
  60.      * @since   2.5
  61.      */
  62.     public static function tokenize($input, $lang, $phrase = false)
  63.     {
  64.         static $cache;
  65.         $store = JString::strlen($input) < 128 ? md5($input . '::' . $lang . '::' . $phrase) : null;
  66.  
  67.         // Check if the string has been tokenized already.
  68.         if ($store && isset($cache[$store]))
  69.         {
  70.             return $cache[$store];
  71.         }
  72.  
  73.         $tokens = array();
  74.         $quotes = html_entity_decode('&#8216;&#8217;&#39;', ENT_QUOTES, 'UTF-8');
  75.  
  76.         // Get the simple language key.
  77.         $lang = self::getPrimaryLanguage($lang);
  78.  
  79.         /*
  80.          * Parsing the string input into terms is a multi-step process.
  81.          *
  82.          * Regexes:
  83.          *  1. Remove everything except letters, numbers, quotes, apostrophe, plus, dash, period, and comma.
  84.          *  2. Remove plus, dash, period, and comma characters located before letter characters.
  85.          *  3. Remove plus, dash, period, and comma characters located after other characters.
  86.          *  4. Remove plus, period, and comma characters enclosed in alphabetical characters. Ungreedy.
  87.          *  5. Remove orphaned apostrophe, plus, dash, period, and comma characters.
  88.          *  6. Remove orphaned quote characters.
  89.          *  7. Replace the assorted single quotation marks with the ASCII standard single quotation.
  90.          *  8. Remove multiple space characters and replaces with a single space.
  91.          */
  92.         $input = JString::strtolower($input);
  93.         $input = preg_replace('#[^\pL\pM\pN\p{Pi}\p{Pf}\'+-.,]+#mui', ' ', $input);
  94.         $input = preg_replace('#(^|\s)[+-.,]+([\pL\pM]+)#mui', ' $1', $input);
  95.         $input = preg_replace('#([\pL\pM\pN]+)[+-.,]+(\s|$)#mui', '$1 ', $input);
  96.         $input = preg_replace('#([\pL\pM]+)[+.,]+([\pL\pM]+)#muiU', '$1 $2', $input);
  97.         $input = preg_replace('#(^|\s)[\'+-.,]+(\s|$)#mui', ' ', $input);
  98.         $input = preg_replace('#(^|\s)[\p{Pi}\p{Pf}]+(\s|$)#mui', ' ', $input);
  99.         $input = preg_replace('#[' . $quotes . ']+#mui', '\'', $input);
  100.         $input = preg_replace('#\s+#mui', ' ', $input);
  101.         $input = JString::trim($input);
  102.  
  103.         // Explode the normalized string to get the terms.
  104.         $terms = explode(' ', $input);
  105.  
  106.         /*
  107.          * If we have Unicode support and are dealing with Chinese text, Chinese
  108.          * has to be handled specially because there are not necessarily any spaces
  109.          * between the "words". So, we have to test if the words belong to the Chinese
  110.          * character set and if so, explode them into single glyphs or "words".
  111.          */
  112.         if ($lang === 'zh')
  113.         {
  114.             // Iterate through the terms and test if they contain Chinese.
  115.             for ($i = 0, $n = count($terms); $i < $n; $i++)
  116.             {
  117.                 $charMatches = array();
  118.                 $charCount = preg_match_all('#[\p{Han}]#mui', $terms[$i], $charMatches);
  119.  
  120.                 // Split apart any groups of Chinese characters.
  121.                 for ($j = 0; $j < $charCount; $j++)
  122.                 {
  123.                     $tSplit = JString::str_ireplace($charMatches[0][$j], '', $terms[$i], false);
  124.                     if (!empty($tSplit))
  125.                     {
  126.                         $terms[$i] = $tSplit;
  127.                     }
  128.                     else
  129.                     {
  130.                         unset($terms[$i]);
  131.                     }
  132.  
  133.                     $terms[] = $charMatches[0][$j];
  134.                 }
  135.             }
  136.  
  137.             // Reset array keys.
  138.             $terms = array_values($terms);
  139.         }
  140.  
  141.         /*
  142.          * If we have to handle the input as a phrase, that means we don't
  143.          * tokenize the individual terms and we do not create the two and three
  144.          * term combinations. The phrase must contain more than one word!
  145.          */
  146.         if ($phrase === true && count($terms) > 1)
  147.         {
  148.             // Create tokens from the phrase.
  149.             $tokens[] = new FinderIndexerToken($terms, $lang);
  150.         }
  151.         else
  152.         {
  153.             // Create tokens from the terms.
  154.             for ($i = 0, $n = count($terms); $i < $n; $i++)
  155.             {
  156.                 $tokens[] = new FinderIndexerToken($terms[$i], $lang);
  157.             }
  158.  
  159.             // Create two and three word phrase tokens from the individual words.
  160.             for ($i = 0, $n = count($tokens); $i < $n; $i++)
  161.             {
  162.                 // Setup the phrase positions.
  163.                 $i2 = $i + 1;
  164.                 $i3 = $i + 2;
  165.  
  166.                 // Create the two word phrase.
  167.                 if ($i2 < $n && isset($tokens[$i2]))
  168.                 {
  169.                     // Tokenize the two word phrase.
  170.                     $token = new FinderIndexerToken(array($tokens[$i]->term, $tokens[$i2]->term), $lang, $lang === 'zh' ? '' : ' ');
  171.                     $token->derived = true;
  172.  
  173.                     // Add the token to the stack.
  174.                     $tokens[] = $token;
  175.                 }
  176.  
  177.                 // Create the three word phrase.
  178.                 if ($i3 < $n && isset($tokens[$i3]))
  179.                 {
  180.                     // Tokenize the three word phrase.
  181.                     $token = new FinderIndexerToken(array($tokens[$i]->term, $tokens[$i2]->term, $tokens[$i3]->term), $lang, $lang === 'zh' ? '' : ' ');
  182.                     $token->derived = true;
  183.  
  184.                     // Add the token to the stack.
  185.                     $tokens[] = $token;
  186.                 }
  187.             }
  188.         }
  189.  
  190.         if ($store)
  191.         {
  192.             $cache[$store] = count($tokens) > 1 ? $tokens : array_shift($tokens);
  193.             return $cache[$store];
  194.         }
  195.         else
  196.         {
  197.             return count($tokens) > 1 ? $tokens : array_shift($tokens);
  198.         }
  199.     }

  201.     /**
  202.      * Method to get the base word of a token. This method uses the public
  203.      * {@link FinderIndexerHelper::$stemmer} object if it is set. If no stemmer is set,
  204.      * the original token is returned.
  205.      *
  206.      * @param   string  $token  The token to stem.
  207.      * @param   string  $lang   The language of the token.
  208.      *
  209.      * @return  string  The root token.
  210.      *
  211.      * @since   2.5
  212.      */
  213.     public static function stem($token, $lang)
  214.     {
  215.         // Trim apostrophes at either end of the token.
  216.         $token = JString::trim($token, '\'');
  217.  
  218.         // Trim everything after any apostrophe in the token.
  219.         if (($pos = JString::strpos($token, '\'')) !== false)
  220.         {
  221.             $token = JString::substr($token, 0, $pos);
  222.         }
  223.  
  224.         // Stem the token if we have a valid stemmer to use.
  225.         if (self::$stemmer instanceof FinderIndexerStemmer)
  226.         {
  227.             return self::$stemmer->stem($token, $lang);
  228.         }
  229.         else
  230.         {
  231.             return $token;
  232.         }
  233.     }
