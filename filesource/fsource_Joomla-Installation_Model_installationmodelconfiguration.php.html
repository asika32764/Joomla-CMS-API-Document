  230.     private function _createRootUser($options)

  247.

  261.  
  262.         // Take the admin user id

  264.

  267.  
  268.         // Create the admin user
  269.         date_default_timezone_set('UTC');

  295.                 ->where($db->quoteName('id') . ' = ' . $db->quote($userId));
  296.         }
  297.         else
