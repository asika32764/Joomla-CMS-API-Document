  16.  * @subpackage  Content.pagenavigation

  19. class PlgContentPagenavigation extends JPlugin

  21.     /**
  22.      * If in the article view and the parameter is enabled shows the page navigation
  23.      *
  24.      * @param   string   $context  The context of the content being passed to the plugin
  25.      * @param   object   &$row     The article object
  26.      * @param   mixed    &$params  The article params
  27.      * @param   integer  $page     The 'page' number
  28.      *
  29.      * @return  mixed  void or true
  30.      *
  31.      * @since   1.6
  32.      */
  33.     public function onContentBeforeDisplay($context, &$row, &$params, $page = 0)
  34.     {
  35.         $app = JFactory::getApplication();
  36.         $view = $app->input->get('view');
  37.         $print = $app->input->getBool('print');
  38.  
  39.         if ($print)
  40.         {
  41.             return false;
  42.         }
  43.  
  44.         if (($context == 'com_content.article') && ($view == 'article') && $params->get('show_item_navigation'))
  45.         {
  46.             $db = JFactory::getDbo();
  47.             $user = JFactory::getUser();
  48.             $lang = JFactory::getLanguage();
  49.             $nullDate = $db->getNullDate();
  50.  
  51.             $date = JFactory::getDate();
  52.             $now = $date->toSql();
  53.  
  54.             $uid = $row->id;
  55.             $option = 'com_content';
  56.             $canPublish = $user->authorise('core.edit.state', $option . '.article.' . $row->id);
  57.  
  58.             /**
  59.              * The following is needed as different menu items types utilise a different param to control ordering.
  60.              * For Blogs the `orderby_sec` param is the order controlling param.
  61.              * For Table and List views it is the `orderby` param.
  62.             **/
  63.             $params_list = $params->toArray();
  64.  
  65.             if (array_key_exists('orderby_sec', $params_list))
  66.             {
  67.                 $order_method = $params->get('orderby_sec', '');
  68.             }
  69.             else
  70.             {
  71.                 $order_method = $params->get('orderby', '');
  72.             }
  73.  
  74.             // Additional check for invalid sort ordering.
  75.             if ($order_method == 'front')
  76.             {
  77.                 $order_method = '';
  78.             }
  79.  
  80.             // Determine sort order.
  81.             switch ($order_method)
  82.             {
  83.                 case 'date' :
  84.                     $orderby = 'a.created';
  85.                     break;
  86.                 case 'rdate' :
  87.                     $orderby = 'a.created DESC';
  88.                     break;
  89.                 case 'alpha' :
  90.                     $orderby = 'a.title';
  91.                     break;
  92.                 case 'ralpha' :
  93.                     $orderby = 'a.title DESC';
  94.                     break;
  95.                 case 'hits' :
  96.                     $orderby = 'a.hits';
  97.                     break;
  98.                 case 'rhits' :
  99.                     $orderby = 'a.hits DESC';
  100.                     break;
  101.                 case 'order' :
  102.                     $orderby = 'a.ordering';
  103.                     break;
  104.                 case 'author' :
  105.                     $orderby = 'a.created_by_alias, u.name';
  106.                     break;
  107.                 case 'rauthor' :
  108.                     $orderby = 'a.created_by_alias DESC, u.name DESC';
  109.                     break;
  110.                 case 'front' :
  111.                     $orderby = 'f.ordering';
  112.                     break;
  113.                 default :
  114.                     $orderby = 'a.ordering';
  115.                     break;
  116.             }
  117.  
  118.             $xwhere = ' AND (a.state = 1 OR a.state = -1)' .
  119.                 ' AND (publish_up = ' . $db->quote($nullDate) . ' OR publish_up <= ' . $db->quote($now) . ')' .
  120.                 ' AND (publish_down = ' . $db->quote($nullDate) . ' OR publish_down >= ' . $db->quote($now) . ')';
  121.  
  122.             // Array of articles in same category correctly ordered.
  123.             $query = $db->getQuery(true);
  124.  
  125.             // Sqlsrv changes
  126.             $case_when = ' CASE WHEN ';
  127.             $case_when .= $query->charLength('a.alias', '!=', '0');
  128.             $case_when .= ' THEN ';
  129.             $a_id = $query->castAsChar('a.id');
  130.             $case_when .= $query->concatenate(array($a_id, 'a.alias'), ':');
  131.             $case_when .= ' ELSE ';
  132.             $case_when .= $a_id . ' END as slug';
  133.  
  134.             $case_when1 = ' CASE WHEN ';
  135.             $case_when1 .= $query->charLength('cc.alias', '!=', '0');
  136.             $case_when1 .= ' THEN ';
  137.             $c_id = $query->castAsChar('cc.id');
  138.             $case_when1 .= $query->concatenate(array($c_id, 'cc.alias'), ':');
  139.             $case_when1 .= ' ELSE ';
  140.             $case_when1 .= $c_id . ' END as catslug';
  141.             $query->select('a.id,' . $case_when . ',' . $case_when1)
  142.                 ->from('#__content AS a')
  143.                 ->join('LEFT', '#__categories AS cc ON cc.id = a.catid')
  144.                 ->where(
  145.                     'a.catid = ' . (int) $row->catid . ' AND a.state = ' . (int) $row->state
  146.                         . ($canPublish ? '' : ' AND a.access = ' . (int) $row->access) . $xwhere
  147.                 );
  148.             $query->order($orderby);
  149.  
  150.             if ($app->isSite() && $app->getLanguageFilter())
  151.             {
  152.                 $query->where('a.language in (' . $db->quote($lang->getTag()) . ',' . $db->quote('*') . ')');
  153.             }
  154.  
  155.             $db->setQuery($query);
  156.             $list = $db->loadObjectList('id');
  157.  
  158.             // This check needed if incorrect Itemid is given resulting in an incorrect result.
  159.             if (!is_array($list))
  160.             {
  161.                 $list = array();
  162.             }
  163.  
  164.             reset($list);
  165.  
  166.             // Location of current content item in array list.
  167.             $location = array_search($uid, array_keys($list));
  168.  
  169.             $rows = array_values($list);
  170.  
  171.             $row->prev = null;
  172.             $row->next = null;
  173.  
  174.             if ($location - 1 >= 0)
  175.             {
  176.                 // The previous content item cannot be in the array position -1.
  177.                 $row->prev = $rows[$location - 1];
  178.             }
  179.  
  180.             if (($location + 1) < count($rows))
  181.             {
  182.                 // The next content item cannot be in an array position greater than the number of array postions.
  183.                 $row->next = $rows[$location + 1];
  184.             }
  185.  
  186.             // $pnSpace is/can be used in the include file
  187.             $pnSpace = "";
  188.  
  189.             if (JText::_('JGLOBAL_LT') || JText::_('JGLOBAL_GT'))
  190.             {
  191.                 $pnSpace = " ";
  192.             }
  193.  
  194.             if ($row->prev)
  195.             {
  196.                 $row->prev = JRoute::_(ContentHelperRoute::getArticleRoute($row->prev->slug, $row->prev->catslug));
  197.             }
  198.             else
  199.             {
  200.                 $row->prev = '';
  201.             }
  202.  
  203.             if ($row->next)
  204.             {
  205.                 $row->next = JRoute::_(ContentHelperRoute::getArticleRoute($row->next->slug, $row->next->catslug));
  206.             }
  207.             else
  208.             {
  209.                 $row->next = '';
  210.             }
  211.  
  212.             // Output.
  213.             if ($row->prev || $row->next)
  214.             {
  215.                 // Get the path for the layout file
  216.                 $path = JPluginHelper::getLayoutPath('content', 'pagenavigation');
  217.  
  218.                 // Render the pagenav
  219.                 ob_start();
  220.                 include $path;
  221.                 $row->pagination = ob_get_clean();
  222.  
  223.                 $row->paginationposition = $this->params->get('position', 1);
  224.  
  225.                 // This will default to the 1.5 and 1.6-1.7 behavior.
  226.                 $row->paginationrelative = $this->params->get('relative', 0);
  227.             }
  228.         }
  229.  
  230.         return;
  231.     }
  232. }
