  111.     /**
  112.      * Returns an object list

  119.     protected function _getList($query, $limitstart = 0, $limit = 0)
  120.     {
  121.         $search = $this->getState('filter.search');
  122.         $ordering = $this->getState('list.ordering', 'ordering');
  123.         if ($ordering == 'name' || (!empty($search) && stripos($search, 'id:') !== 0))
  124.         {
  125.             $this->_db->setQuery($query);
  126.             $result = $this->_db->loadObjectList();
  127.             $this->translate($result);
  128.             if (!empty($search))
  129.             {
  130.                 foreach ($result as $i => $item)
  131.                 {
  132.                     if (!preg_match("/$search/i", $item->name))
  133.                     {
  134.                         unset($result[$i]);
  135.                     }
  136.                 }
  137.             }
  138.  
  139.             $direction = ($this->getState('list.direction') == 'desc') ? -1 : 1;
  140.             JArrayHelper::sortObjects($result, $ordering, $direction, true, true);
  141.  
  142.             $total = count($result);
  143.             $this->cache[$this->getStoreId('getTotal')] = $total;
  144.             if ($total < $limitstart)
  145.             {
  146.                 $limitstart = 0;
  147.                 $this->setState('list.start', 0);
  148.             }
  149.             return array_slice($result, $limitstart, $limit ? $limit : null);
  150.         }
  151.         else
  152.         {
  153.             if ($ordering == 'ordering')
  154.             {
  155.                 $query->order('a.folder ASC');
  156.                 $ordering = 'a.ordering';
  157.             }
  158.             $query->order($this->_db->quoteName($ordering) . ' ' . $this->getState('list.direction'));
  159.  
  160.             if ($ordering == 'folder')
  161.             {
  162.                 $query->order('a.ordering ASC');
  163.             }
  164.             $result = parent::_getList($query, $limitstart, $limit);
  165.             $this->translate($result);

  167.         }

  220.         // Filter by access level.

  226.         // Filter by published state

  228.         if (is_numeric($published))

  232.         elseif ($published === '')

  236.  
  237.         // Filter by state

  239.  
  240.         // Filter by folder.
  241.         if ($folder = $this->getState('filter.folder'))

  244.         }
  245.  
  246.         // Filter by search in name or id
  247.         $search = $this->getState('filter.search');
  248.         if (!empty($search))
  249.         {
  250.             if (stripos($search, 'id:') === 0)
