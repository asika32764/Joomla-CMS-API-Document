  164.     public static function imageResize($width, $height, $target)
  165.     {
  166.         //takes the larger size of the width and height and applies the
  167.         //formula accordingly...this is so this script will work
  168.         //dynamically with any size image
  169.         if ($width > $height)
  170.         {
  171.             $percentage = ($target / $width);
  172.         }
  173.         else
  174.         {
  175.             $percentage = ($target / $height);
  176.         }
  177.  
  178.         //gets the new value and applies the percentage, then rounds the value
  179.         $width = round($width * $percentage);
  180.         $height = round($height * $percentage);
  181.  
  182.         return array($width, $height);
  183.     }
