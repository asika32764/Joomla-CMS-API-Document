  19. class JSchemaChangeitemSqlsrv extends JSchemaChangeitem

  21.     /**
  22.      * Checks a DDL query to see if it is a known type
  23.      * If yes, build a check query to see if the DDL has been run on the database.
  24.      * If successful, the $msgElements, $queryType, $checkStatus and $checkQuery fields are populated.
  25.      * The $msgElements contains the text to create the user message.
  26.      * The $checkQuery contains the SQL query to check whether the schema change has
  27.      * been run against the current database. The $queryType contains the type of
  28.      * DDL query that was run (for example, CREATE_TABLE, ADD_COLUMN, CHANGE_COLUMN_TYPE, ADD_INDEX).
  29.      * The $checkStatus field is set to zero if the query is created
  30.      *
  31.      * If not successful, $checkQuery is empty and , and $checkStatus is -1.
  32.      * For example, this will happen if the current line is a non-DDL statement.
  33.      *
  34.      * @return void 
  35.      *
  36.      * @since  2.5
  37.      */
  38.     protected function buildCheckQuery()
  39.     {
  40.         // Initialize fields in case we can't create a check query
  41.         $this->checkStatus = -1; // change status to skipped
  42.         $result = null;
  43.  
  44.         // Remove any newlines
  45.         $this->updateQuery = str_replace("\n", '', $this->updateQuery);
  46.  
  47.         // Fix up extra spaces around () and in general
  48.         $find = array('#((\s*)\(\s*([^)\s]+)\s*)(\))#', '#(\s)(\s*)#');
  49.         $replace = array('($3)', '$1');
  50.         $updateQuery = preg_replace($find, $replace, $this->updateQuery);
  51.         $wordArray = explode(' ', $updateQuery);
  52.  
  53.         // First, make sure we have an array of at least 6 elements
  54.         // if not, we can't make a check query for this one
  55.         if (count($wordArray) < 6)
  56.         {
  57.             // Done with method
  58.             return;
  59.         }
  60.  
  61.         // We can only make check queries for alter table and create table queries
  62.         $command = strtoupper($wordArray[0] . ' ' . $wordArray[1]);
  63.         if ($command === 'ALTER TABLE')
  64.         {
  65.             $alterCommand = strtoupper($wordArray[3] . ' ' . $wordArray[4]);
  66.             if ($alterCommand == 'ADD')
  67.             {
  68.                 $result = 'SELECT * FROM INFORMATION_SCHEMA.Columns ' . $wordArray[2] . ' WHERE COLUMN_NAME = ' . $this->fixQuote($wordArray[5]);
  69.                 $this->queryType = 'ADD';
  70.                 $this->msgElements = array($this->fixQuote($wordArray[2]), $this->fixQuote($wordArray[5]));
  71.             }
  72.             elseif ($alterCommand == 'CREATE INDEX')
  73.             {
  74.                 $index = $this->fixQuote(substr($wordArray[5], 0, strpos($wordArray[5], '(')));
  75.                 $result = 'SELECT * FROM SYS.INDEXES ' . $wordArray[2] . ' WHERE name = ' . $index;
  76.                 $this->queryType = 'CREATE INDEX';
  77.                 $this->msgElements = array($this->fixQuote($wordArray[2]), $index);
  78.             }
  79.             elseif (strtoupper($wordArray[3]) == 'MODIFY' || strtoupper($wordArray[3]) == 'CHANGE')
  80.             {
  81.                 $result = 'SELECT COLUMN_NAME FROM INFORMATION_SCHEMA.COLUMNS  WHERE table_name = ' . $this->fixQuote($wordArray[2]);
  82.                 $this->queryType = 'ALTER COLUMN COLUMN_NAME =' . $this->fixQuote($wordArray[4]);
  83.                 $this->msgElements = array($this->fixQuote($wordArray[2]), $this->fixQuote($wordArray[4]));
  84.             }
  85.         }
  86.  
  87.         if ($command == 'CREATE TABLE')
  88.         {
  89.             $table = $wordArray[5];
  90.             $result = 'SELECT * FROM sys.TABLES WHERE NAME = ' . $this->fixQuote($table);
  91.             $this->queryType = 'CREATE_TABLE';
  92.             $this->msgElements = array($this->fixQuote($table));
  93.         }
  94.  
  95.         // Set fields based on results
  96.         if ($this->checkQuery = $result)
  97.         {
  98.             // Unchecked status
  99.             $this->checkStatus = 0;
  100.         }
  101.         else
  102.         {
  103.             // Skipped
  104.             $this->checkStatus = -1;
  105.         }
  106.     }
  107.  
  108.     /**
  109.      * Fix up integer. Fixes problem with MySQL integer descriptions.
  110.      * If you change a column to "integer unsigned" it shows
  111.      * as "int(10) unsigned" in the check query.
  112.      *
  113.      * @param   string  $type1  the column type
  114.      * @param   string  $type2  the column attributes
  115.      *
  116.      * @return  string  The original or changed column type.
  117.      *
  118.      * @since   2.5
  119.      */
  120.     private function fixInteger($type1, $type2)
  121.     {
  122.         $result = $type1;
  123.         if (strtolower($type1) == 'integer' && strtolower(substr($type2, 0, 8)) == 'unsigned')
  124.         {
  125.             $result = 'int';
  126.         }
  127.         return $result;
  128.     }
  129.  
  130.     /**
  131.      * Fixes up a string for inclusion in a query.
  132.      * Replaces name quote character with normal quote for literal.
  133.      * Drops trailing semi-colon. Injects the database prefix.
  134.      *
  135.      * @param   string  $string  The input string to be cleaned up.
  136.      *
  137.      * @return  string  The modified string.
  138.      *
  139.      * @since   2.5
  140.      */
  141.     private function fixQuote($string)
  142.     {
  143.         $string = str_replace('`', '', $string);
  144.         $string = str_replace(';', '', $string);
  145.         $string = str_replace('#__', $this->db->getPrefix(), $string);
  146.         return $this->db->quote($string);
  147.     }
  148. }
