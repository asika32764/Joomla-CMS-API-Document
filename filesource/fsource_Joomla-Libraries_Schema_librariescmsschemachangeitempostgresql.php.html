  18.  */
  19. class JSchemaChangeitemPostgresql extends JSchemaChangeitem
  20. {
  21.     /**
  22.      * Checks a DDL query to see if it is a known type
  23.      * If yes, build a check query to see if the DDL has been run on the database.
  24.      * If successful, the $msgElements, $queryType, $checkStatus and $checkQuery fields are populated.
  25.      * The $msgElements contains the text to create the user message.
  26.      * The $checkQuery contains the SQL query to check whether the schema change has
  27.      * been run against the current database. The $queryType contains the type of
  28.      * DDL query that was run (for example, CREATE_TABLE, ADD_COLUMN, CHANGE_COLUMN_TYPE, ADD_INDEX).
  29.      * The $checkStatus field is set to zero if the query is created
  30.      *
  31.      * If not successful, $checkQuery is empty and , and $checkStatus is -1.
  32.      * For example, this will happen if the current line is a non-DDL statement.
  33.      *
  34.      * @return void 
  35.      *
  36.      * @since  3.0
  37.      */
  38.     protected function buildCheckQuery()
  39.     {
  40.         // Initialize fields in case we can't create a check query
  41.         $this->checkStatus = -1; // change status to skipped
  42.         $result = null;
  43.  
  44.         // Remove any newlines
  45.         $this->updateQuery = str_replace("\n", '', $this->updateQuery);
  46.  
  47.         // Fix up extra spaces around () and in general
  48.         $find = array('#((\s*)\(\s*([^)\s]+)\s*)(\))#', '#(\s)(\s*)#');
  49.         $replace = array('($3)', '$1');
  50.         $updateQuery = preg_replace($find, $replace, $this->updateQuery);
  51.         $wordArray = explode(' ', $updateQuery);
  52.  
  53.         // First, make sure we have an array of at least 6 elements
  54.         // if not, we can't make a check query for this one
  55.         if (count($wordArray) < 6)
  56.         {
  57.             // Done with method
  58.             return;
  59.         }
  60.  
  61.         // We can only make check queries for alter table and create table queries
  62.         $command = strtoupper($wordArray[0] . ' ' . $wordArray[1]);
  63.         if ($command === 'ALTER TABLE')
  64.         {
  65.             $alterCommand = strtoupper($wordArray[3] . ' ' . $wordArray[4]);
  66.             if ($alterCommand === 'ADD COLUMN')
  67.             {
  68.                 $result = 'SELECT column_name FROM information_schema.columns WHERE table_name='
  69.                 . $this->fixQuote($wordArray[2]) . ' AND column_name=' . $this->fixQuote($wordArray[5]);
  70.  
  71.                 $this->queryType = 'ADD_COLUMN';
  72.                 $this->msgElements = array($this->fixQuote($wordArray[2]), $this->fixQuote($wordArray[5]));
  73.             }
  74.             elseif ($alterCommand === 'ALTER COLUMN')
  75.             {
  76.                 if (strtoupper($wordArray[6]) == 'TYPE')
  77.                 {
  78.                     $type = '';
  79.                     for ($i = 7; $i < count($wordArray); $i++)
  80.                     {
  81.                         $type .= $wordArray[$i] . ' ';
  82.                     }
  83.  
  84.                     if ($pos = strpos($type, '('))
  85.                     {
  86.                         $type = substr($type, 0, $pos);
  87.                     }
  88.  
  89.                     if ($pos = strpos($type, ';'))
  90.                     {
  91.                         $type = substr($type, 0, $pos);
  92.                     }
  93.  
  94.                     $result = 'SELECT column_name, data_type FROM information_schema.columns WHERE table_name='
  95.                         . $this->fixQuote($wordArray[2]) . ' AND column_name=' . $this->fixQuote($wordArray[5])
  96.                         . ' AND data_type=' . $this->fixQuote($type);
  97.  
  98.                     $this->queryType = 'CHANGE_COLUMN_TYPE';
  99.                     $this->msgElements = array($this->fixQuote($wordArray[2]), $this->fixQuote($wordArray[5]), $type);
  100.                 }
  101.                 elseif (strtoupper($wordArray[7] . ' ' . $wordArray[8]) == 'NOT NULL')
  102.                 {
  103.                     if (strtoupper($wordArray[6]) == 'SET')
  104.                     {
  105.                         // SET NOT NULL
  106.                         $isNullable = $this->fixQuote('NO');
  107.                     }
  108.                     else
  109.                     {
  110.                         // DROP NOT NULL
  111.                         $isNullable = $this->fixQuote('YES');
  112.                     }
  113.                     $result = 'SELECT column_name, data_type, is_nullable FROM information_schema.columns WHERE table_name='
  114.                         . $this->fixQuote($wordArray[2]) . ' AND column_name=' . $this->fixQuote($wordArray[5])
  115.                         . ' AND is_nullable=' . $isNullable;
  116.  
  117.                     $this->queryType = 'CHANGE_COLUMN_TYPE';
  118.                     $this->checkQueryExpected = 1;
  119.                     $this->msgElements = array($this->fixQuote($wordArray[2]), $this->fixQuote($wordArray[5]), $isNullable);
  120.                 }
  121.                 elseif (strtoupper($wordArray[7]) === 'DEFAULT')
  122.                 {
  123.                     if (strtoupper($wordArray[6]) == 'SET')
  124.                     {
  125.                         $isNullDef = 'IS NOT NULL';
  126.                     }
  127.                     else
  128.                     {
  129.                         // DROP DEFAULT
  130.                         $isNullDef = 'IS NULL';
  131.                     }
  132.                     $result = 'SELECT column_name, data_type, column_default FROM information_schema.columns WHERE table_name='
  133.                         . $this->fixQuote($wordArray[2]) . ' AND column_name=' . $this->fixQuote($wordArray[5])
  134.                         . ' AND column_default ' . $isNullDef;
  135.  
  136.                     $this->queryType = 'CHANGE_COLUMN_TYPE';
  137.                     $this->checkQueryExpected = 1;
  138.                     $this->msgElements = array($this->fixQuote($wordArray[2]), $this->fixQuote($wordArray[5]), $isNullDef);
  139.                 }
  140.             }
  141.         }
  142.         elseif ($command === 'DROP INDEX')
  143.         {
  144.             if (strtoupper($wordArray[2] . $wordArray[3]) == 'IFEXISTS')
  145.             {
  146.                 $idx = $this->fixQuote($wordArray[4]);
  147.             }
  148.             else
  149.             {
  150.                 $idx = $this->fixQuote($wordArray[2]);
  151.             }
  152.  
  153.             $result = 'SELECT * FROM pg_indexes WHERE indexname=' . $idx;
  154.             $this->queryType = 'DROP_INDEX';
  155.             $this->checkQueryExpected = 0;
  156.             $this->msgElements = array($this->fixQuote($idx));
  157.         }
  158.         elseif ($command == 'CREATE INDEX' || (strtoupper($command . $wordArray[2]) == 'CREATE UNIQUE INDEX'))
  159.         {
  160.             if ($wordArray[1] === 'UNIQUE')
  161.             {
  162.                 $idx = $this->fixQuote($wordArray[3]);
  163.                 $table = $this->fixQuote($wordArray[5]);
  164.             }
  165.             else
  166.             {
  167.                 $idx = $this->fixQuote($wordArray[2]);
  168.                 $table = $this->fixQuote($wordArray[4]);
  169.             }
  170.  
  171.             $result = 'SELECT * FROM pg_indexes WHERE indexname=' . $idx . ' AND tablename=' . $table;
  172.             $this->queryType = 'ADD_INDEX';
  173.             $this->checkQueryExpected = 1;
  174.             $this->msgElements = array($table, $idx);
  175.         }
  176.  
  177.         if ($command == 'CREATE TABLE')
  178.         {
  179.             if (strtoupper($wordArray[2] . $wordArray[3] . $wordArray[4]) == 'IFNOTEXISTS')
  180.             {
  181.                 $table = $this->fixQuote($wordArray[5]);
  182.             }
  183.             else
  184.             {
  185.                 $table = $this->fixQuote($wordArray[2]);
  186.             }
  187.             $result = 'SELECT table_name FROM information_schema.tables WHERE table_name=' . $table;
  188.             $this->queryType = 'CREATE_TABLE';
  189.             $this->checkQueryExpected = 1;
  190.             $this->msgElements = array($table);
  191.         }
  192.  
  193.         // Set fields based on results
  194.         if ($this->checkQuery = $result)
  195.         {
  196.             // Unchecked status
  197.             $this->checkStatus = 0;
  198.         }
  199.         else
  200.         {
  201.             // Skipped
  202.             $this->checkStatus = -1;
  203.         }
  204.     }
  205.  
  206.     /**
  207.      * Fix up integer. Fixes problem with PostgreSQL integer descriptions.
  208.      * If you change a column to "integer unsigned" it shows
  209.      * as "int(10) unsigned" in the check query.
  210.      *
  211.      * @param   string  $type1  the column type
  212.      * @param   string  $type2  the column attributes
  213.      *
  214.      * @return  string  The original or changed column type.
  215.      *
  216.      * @since   3.0
  217.      */
  218.     private function fixInteger($type1, $type2)
  219.     {
  220.         $result = $type1;
  221.         if (strtolower($type1) == 'integer' && strtolower(substr($type2, 0, 8)) == 'unsigned')
  222.         {
  223.             $result = 'unsigned int(10)';
  224.         }
  225.         return $result;
  226.     }
  227.  
  228.     /**
  229.      * Fixes up a string for inclusion in a query.
  230.      * Replaces name quote character with normal quote for literal.
  231.      * Drops trailing semi-colon. Injects the database prefix.
  232.      *
  233.      * @param   string  $string  The input string to be cleaned up.
  234.      *
  235.      * @return  string  The modified string.
  236.      *
  237.      * @since   3.0
  238.      */
  239.     private function fixQuote($string)
  240.     {
  241.         $string = str_replace('"', '', $string);
  242.         $string = str_replace(';', '', $string);
  243.         $string = str_replace('#__', $this->db->getPrefix(), $string);
  244.         return $this->db->quote($string);
  245.     }
