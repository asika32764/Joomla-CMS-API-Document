  26.      * @param   object  &$response    Authentication response object

  32.     public function onUserAuthenticate($credentials, $options, &$response)

  34.         $userdetails = null;

  49.  
  50.         // Load plugin params info

  53.         $ldap_uid        = $this->params->get('ldap_uid');
  54.         $auth_method    = $this->params->get('auth_method');
  55.  
  56.         $ldap = new JClientLdap($this->params);
  57.  
  58.         if (!$ldap->connect())

  65.  
  66.         switch ($auth_method)

  69.             {
  70.                 // Bind using Connect Username/password
  71.                 // Force anon bind to mitigate misconfiguration like [#7119]
  72.                 if (strlen($this->params->get('username')))

  75.                 }

  101.                 {
  102.                     $response->status = JAuthentication::STATUS_FAILURE;

  106.  
  107.             case 'bind':

  112.                 if ($success)

  115.                 }
  116.                 else
  117.                 {

  122.         }
  123.  
  124.         if (!$success)

  126.             $response->status = JAuthentication::STATUS_FAILURE;
  127.  
  128.             if (!strlen($response->error_message))
  129.             {
  130.                 $response->error_message = JText::_('JGLOBAL_AUTH_INCORRECT');
  131.             }
  132.         }
  133.         else
  134.         {
  135.             // Grab some details from LDAP and return them
  136.             if (isset($userdetails[0][$ldap_uid][0]))

  138.                 $response->username = $userdetails[0][$ldap_uid][0];
  139.             }
  140.  
  141.             if (isset($userdetails[0][$ldap_email][0]))

  144.             }
  145.  
  146.             if (isset($userdetails[0][$ldap_fullname][0]))
  147.             {
  148.                 $response->fullname = $userdetails[0][$ldap_fullname][0];
  149.             } else {
  150.                 $response->fullname = $credentials['username'];
  151.             }
  152.  
  153.             // Were good - So say so.
  154.             $response->status        = JAuthentication::STATUS_SUCCESS;

  159.     }
  160. }
