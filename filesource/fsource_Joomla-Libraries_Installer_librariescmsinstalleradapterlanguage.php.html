  124.         // Do we have a meta file in the file list?  In other words... is this a core language pack?

  135.                 }
  136.             }

  138.  
  139.         // If the language directory does not exist, let's create it

  142.         if (!file_exists($this->parent->getPath('extension_site')))

  148.                     JText::sprintf(
  149.                         'JLIB_INSTALLER_ABORT',
  150.                         JText::sprintf('JLIB_INSTALLER_ERROR_CREATE_FOLDER_FAILED', $this->parent->getPath('extension_site'))
  151.                     )
  152.                 );

  168.             elseif (!$this->parent->isOverwrite())
  169.             {

  176.                         JText::sprintf('JLIB_INSTALLER_ABORT', JText::sprintf('JLIB_INSTALLER_ERROR_FOLDER_IN_USE', $this->parent->getPath('extension_site'))),

  180.                 else

  185.                         JLog::WARNING, 'jerror'
  186.                     );

  188.                 return false;
  189.             }

  191.  
  192.         /*

  201.  
  202.         // Copy all the necessary files
  203.         if ($this->parent->parseFiles($element) === false)
  204.         {
  205.             // Install failed, rollback changes
  206.             $this->parent->abort();
  207.  
  208.             return false;
  209.         }
  210.  
  211.         // Parse optional tags
  212.         $this->parent->parseMedia($this->manifest->media);
  213.  
  214.         // Copy all the necessary font files to the common pdf_fonts directory

  238.  
  239.         // Add an entry to the extension table with a whole heap of defaults
  240.         $row = JTable::getInstance('extension');
  241.         $row->set('name', $this->get('name'));
  242.         $row->set('type', 'language');
  243.         $row->set('element', $this->get('tag'));
  244.  
  245.         // There is no folder for languages
  246.         $row->set('folder', '');
  247.         $row->set('enabled', 1);
  248.         $row->set('protected', 0);
  249.         $row->set('access', 0);
  250.         $row->set('client_id', $clientId);

  259.             return false;

  261.  
  262.         // Clobber any possible pending updates
  263.         $update = JTable::getInstance('update');
  264.         $uid = $update->find(array('element' => $this->get('tag'), 'type' => 'language', 'client_id' => '', 'folder' => ''));
  265.  
  266.         if ($uid)
  267.         {
  268.             $update->delete($uid);
  269.         }

  272.     }

  274.     /**

  281.     public function update()
  282.     {

  306.  
  307.         // Get the Language tag [ISO tag, eg. en-GB]

  312.         {
  313.             $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT', JText::_('JLIB_INSTALLER_ERROR_NO_LANGUAGE_TAG')));

  324.         if (count($xml->files->children()))

  415.     /**
  416.      * Custom uninstall method

  424.     public function uninstall($eid)
  425.     {
  426.         // Load up the extension details
  427.         $extension = JTable::getInstance('extension');
  428.         $extension->load($eid);
  429.  
  430.         // Grab a copy of the client details
  431.         $client = JApplicationHelper::getClientInfo($extension->get('client_id'));
  432.  
  433.         // Check the element isn't blank to prevent nuking the languages directory...just in case
  434.         $element = $extension->get('element');
  435.  
  436.         if (empty($element))
  437.         {
  438.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_LANG_UNINSTALL_ELEMENT_EMPTY'), JLog::WARNING, 'jerror');
  439.  
  440.             return false;
  441.         }
  442.  
  443.         // Check that the language is not protected, Normally en-GB.
  444.         $protected = $extension->get('protected');
  445.  
  446.         if ($protected == 1)
  447.         {
  448.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_LANG_UNINSTALL_PROTECTED'), JLog::WARNING, 'jerror');
  449.  
  450.             return false;
  451.         }
  452.  
  453.         // Verify that it's not the default language for that client
  454.         $params = JComponentHelper::getParams('com_languages');
  455.  
  456.         if ($params->get($client->name) == $element)
  457.         {
  458.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_LANG_UNINSTALL_DEFAULT'), JLog::WARNING, 'jerror');
  459.  
  460.             return false;
  461.         }
  462.  
  463.         // Construct the path from the client, the language and the extension element name
  464.         $path = $client->path . '/language/' . $element;
  465.  
  466.         // Get the package manifest object and remove media
  467.         $this->parent->setPath('source', $path);
  468.  
  469.         // We do findManifest to avoid problem when uninstalling a list of extension: getManifest cache its manifest file
  470.         $this->parent->findManifest();
  471.         $this->manifest = $this->parent->getManifest();
  472.         $this->parent->removeFiles($this->manifest->media);
  473.  
  474.         // Check it exists
  475.         if (!JFolder::exists($path))
  476.         {
  477.             // If the folder doesn't exist lets just nuke the row as well and presume the user killed it for us
  478.             $extension->delete();

  480.  
  481.             return false;
  482.         }
  483.  
  484.         if (!JFolder::delete($path))
  485.         {
  486.             // If deleting failed we'll leave the extension entry in tact just in case
  487.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_LANG_UNINSTALL_DIRECTORY'), JLog::WARNING, 'jerror');
  488.  
  489.             return false;
  490.         }
  491.  
  492.         // Remove the extension table entry
  493.         $extension->delete();
  494.  
  495.         // Setting the language of users which have this language as the default language
  496.         $db = JFactory::getDbo();
  497.         $query = $db->getQuery(true)
  498.             ->from('#__users')
  499.             ->select('*');
  500.         $db->setQuery($query);
  501.         $users = $db->loadObjectList();
  502.  
  503.         if ($client->name == 'administrator')
  504.         {
  505.             $param_name = 'admin_language';
  506.         }
  507.         else
  508.         {
  509.             $param_name = 'language';
  510.         }
  511.  
  512.         $count = 0;
  513.  
  514.         foreach ($users as $user)
  515.         {
  516.             $registry = new JRegistry;
  517.             $registry->loadString($user->params);
  518.  
  519.             if ($registry->get($param_name) == $element)
  520.             {
  521.                 $registry->set($param_name, '');
  522.                 $query->clear()
  523.                     ->update('#__users')
  524.                     ->set('params=' . $db->quote($registry))
  525.                     ->where('id=' . (int) $user->id);
  526.                 $db->setQuery($query);
  527.                 $db->execute();
  528.                 $count++;
  529.             }
  530.         }
  531.         if (!empty($count))
  532.         {
  533.             JLog::add(JText::plural('JLIB_INSTALLER_NOTICE_LANG_RESET_USERS', $count), JLog::NOTICE, 'jerror');
  534.         }
  535.  
  536.         // All done!
  537.         return true;
  538.     }
