  120.         $args = $_SERVER['argv'];
  121.

  124.

  127.         {

  130.  
  131.             // First let's tackle the long argument case.  eg. --foo
  132.             if (strlen($arg) > 2 && substr($arg, 0, 2) == '--')
  133.             {
  134.  
  135.                 // Attempt to split the thing over equals so we can get the key/value pair if an = was used.
  136.                 $arg = substr($arg, 2);
  137.                 $parts = explode('=', $arg);
  138.                 $this->data[$parts[0]] = true;
  139.  
  140.                 // Does not have an =, so let's look ahead to the next argument for the value.
  141.                 if (count($parts) == 1 && isset($args[$i + 1]) && preg_match('/^--?.+/', $args[$i + 1]) == 0)
  142.                 {
  143.                     $this->data[$parts[0]] = $args[$i + 1];
  144.  
  145.                     // Since we used the next argument, increment the counter so we don't use it again.
  146.                     $i++;
  147.                 }
  148.                 // We have an equals sign so take the second "part" of the argument as the value.
  149.                 elseif (count($parts) == 2)
  150.                 {
  151.                     $this->data[$parts[0]] = $parts[1];
  152.                 }
  153.             }
  154.  
  155.             // Next let's see if we are dealing with a "bunch" of short arguments.  eg. -abc
  156.             elseif (strlen($arg) > 2 && $arg[0] == '-')
  157.             {
  158.  
  159.                 // For each of these arguments set the value to TRUE since the flag has been set.
  160.                 for ($j = 1; $j < strlen($arg); $j++)
  161.                 {
  162.                     $this->data[$arg[$j]] = true;
  163.                 }
  164.             }
  165.  
  166.             // OK, so it isn't a long argument or bunch of short ones, so let's look and see if it is a single
  167.             // short argument.  eg. -h
  168.             elseif (strlen($arg) == 2 && $arg[0] == '-')
  169.             {
  170.

  172.                 $this->data[$arg[1]] = true;
  173.  
  174.                 // Let's look ahead to see if the next argument is a "value".  If it is, use it for this value.
  175.                 if (isset($args[$i + 1]) && preg_match('/^--?.+/', $args[$i + 1]) == 0)
  176.                 {

  179.                     // Since we used the next argument, increment the counter so we don't use it again.
  180.                     $i++;
  181.                 }
  182.             }
  183.
