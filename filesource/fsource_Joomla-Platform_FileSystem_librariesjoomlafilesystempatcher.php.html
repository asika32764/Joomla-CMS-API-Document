  126.      *

  129.     public function apply()
  130.     {
  131.         foreach ($this->patches as $patch)
  132.         {
  133.             // Separate the input into lines
  134.             $lines = self::splitLines($patch['udiff']);
  135.  
  136.             // Loop for each header
  137.             while (self::findHeader($lines, $src, $dst))
  138.             {
  139.                 $done = false;
  140.  
  141.                 if ($patch['strip'] === null)
  142.                 {
  143.                     $src = $patch['root'] . preg_replace('#^([^/]*/)*#', '', $src);
  144.                     $dst = $patch['root'] . preg_replace('#^([^/]*/)*#', '', $dst);
  145.                 }
  146.                 else
  147.                 {
  148.                     $src = $patch['root'] . preg_replace('#^([^/]*/){' . (int) $patch['strip'] . '}#', '', $src);
  149.                     $dst = $patch['root'] . preg_replace('#^([^/]*/){' . (int) $patch['strip'] . '}#', '', $dst);
  150.                 }
  151.  
  152.                 // Loop for each hunk of differences
  153.                 while (self::findHunk($lines, $src_line, $src_size, $dst_line, $dst_size))
  154.                 {
  155.                     $done = true;
  156.  
  157.                     // Apply the hunk of differences
  158.                     $this->applyHunk($lines, $src, $dst, $src_line, $src_size, $dst_line, $dst_size);
  159.                 }
  160.  
  161.                 // If no modifications were found, throw an exception
  162.                 if (!$done)
  163.                 {
  164.                     throw new RuntimeException('Invalid Diff');
  165.                 }
  166.             }
  167.         }
  168.  
  169.         // Initialize the counter
  170.         $done = 0;
  171.  
  172.         // Patch each destination file
  173.         foreach ($this->destinations as $file => $content)
  174.         {
  175.             if (JFile::write($file, implode("\n", $content)))
  176.             {
  177.                 if (isset($this->sources[$file]))
  178.                 {
  179.                     $this->sources[$file] = $content;
  180.                 }
  181.                 $done++;
  182.             }
  183.         }
  184.  
  185.         // Remove each removed file
  186.         foreach ($this->removals as $file)
  187.         {
  188.             if (JFile::delete($file))
  189.             {
  190.                 if (isset($this->sources[$file]))
  191.                 {
  192.                     unset($this->sources[$file]);
  193.                 }
  194.                 $done++;
  195.             }
  196.         }
  197.  
  198.         // Clear the destinations cache
  199.         $this->destinations = array();
  200.  
  201.         // Clear the removals
  202.         $this->removals = array();
  203.  
  204.         // Clear the patches
  205.         $this->patches = array();
  206.         return $done;
  207.     }

  260.     /**
  261.      * Find the diff header
  262.      *
  263.      * The internal array pointer of $lines is on the next line after the finding
  264.      *
  265.      * @param   array   &$lines  The udiff array of lines
  266.      * @param   string  &$src    The source file
  267.      * @param   string  &$dst    The destination file
  268.      *
  269.      * @return  boolean  TRUE in case of success, FALSE in case of failure
  270.      *
  271.      * @throw  RuntimeException
  272.      */
  273.     protected static function findHeader(&$lines, &$src, &$dst)
  274.     {
  275.         // Get the current line
  276.         $line = current($lines);
  277.  
  278.         // Search for the header
  279.         while ($line !== false && !preg_match(self::SRC_FILE, $line, $m))
  280.         {
  281.             $line = next($lines);
  282.         }
  283.         if ($line === false)
  284.         {
  285.             // No header found, return false
  286.             return false;
  287.         }
  288.         else
  289.         {
  290.             // Set the source file
  291.             $src = $m[1];
  292.  
  293.             // Advance to the next line
  294.             $line = next($lines);
  295.             if ($line === false)
  296.             {
  297.                 throw new RuntimeException('Unexpected EOF');
  298.             }
  299.  
  300.             // Search the destination file
  301.             if (!preg_match(self::DST_FILE, $line, $m))
  302.             {
  303.                 throw new RuntimeException('Invalid Diff file');
  304.             }
  305.  
  306.             // Set the destination file
  307.             $dst = $m[1];
  308.  
  309.             // Advance to the next line
  310.             if (next($lines) === false)
  311.             {
  312.                 throw new RuntimeException('Unexpected EOF');
  313.             }
  314.             return true;
  315.         }
  316.     }
  317.  
  318.     /**
  319.      * Find the next hunk of difference
  320.      *
  321.      * The internal array pointer of $lines is on the next line after the finding
  322.      *
  323.      * @param   array   &$lines     The udiff array of lines
  324.      * @param   string  &$src_line  The beginning of the patch for the source file
  325.      * @param   string  &$src_size  The size of the patch for the source file
  326.      * @param   string  &$dst_line  The beginning of the patch for the destination file
  327.      * @param   string  &$dst_size  The size of the patch for the destination file
  328.      *
  329.      * @return  boolean  TRUE in case of success, false in case of failure
  330.      *
  331.      * @throw  RuntimeException
  332.      */
  333.     protected static function findHunk(&$lines, &$src_line, &$src_size, &$dst_line, &$dst_size)
  334.     {
  335.         $line = current($lines);
  336.         if (preg_match(self::HUNK, $line, $m))
  337.         {
  338.             $src_line = (int) $m[1];
  339.             if ($m[3] === '')
  340.             {
  341.                 $src_size = 1;
  342.             }
  343.             else
  344.             {
  345.                 $src_size = (int) $m[3];
  346.             }
  347.  
  348.             $dst_line = (int) $m[4];
  349.             if ($m[6] === '')
  350.             {
  351.                 $dst_size = 1;
  352.             }
  353.             else
  354.             {
  355.                 $dst_size = (int) $m[6];
  356.             }
  357.  
  358.             if (next($lines) === false)
  359.             {
  360.                 throw new RuntimeException('Unexpected EOF');
  361.             }
  362.  
  363.             return true;
  364.         }
  365.         else
  366.         {
  367.             return false;
  368.         }
  369.     }

  371.     /**
  372.      * Apply the patch
  373.      *
  374.      * @param   array   &$lines    The udiff array of lines
  375.      * @param   string  $src       The source file
  376.      * @param   string  $dst       The destination file
  377.      * @param   string  $src_line  The beginning of the patch for the source file
  378.      * @param   string  $src_size  The size of the patch for the source file
  379.      * @param   string  $dst_line  The beginning of the patch for the destination file
  380.      * @param   string  $dst_size  The size of the patch for the destination file
  381.      *
  382.      * @return  void 
  383.      *
  384.      * @throw  RuntimeException
  385.      */
  386.     protected function applyHunk(&$lines, $src, $dst, $src_line, $src_size, $dst_line, $dst_size)
  387.     {
  388.         $src_line--;
  389.         $dst_line--;
  390.         $line = current($lines);
  391.  
  392.         // Source lines (old file)
  393.         $source = array();
  394.  
  395.         // New lines (new file)
  396.         $destin = array();
  397.         $src_left = $src_size;
  398.         $dst_left = $dst_size;
  399.         do
  400.         {
  401.             if (!isset($line[0]))
  402.             {
  403.                 $source[] = '';
  404.                 $destin[] = '';
  405.                 $src_left--;
  406.                 $dst_left--;
  407.             }
  408.             elseif ($line[0] == '-')
  409.             {
  410.                 if ($src_left == 0)
  411.                 {
  412.                     throw new RuntimeException(JText::sprintf('JLIB_FILESYSTEM_PATCHER_REMOVE_LINE', key($lines)));
  413.                 }
  414.                 $source[] = substr($line, 1);
  415.                 $src_left--;
  416.             }
  417.             elseif ($line[0] == '+')
  418.             {
  419.                 if ($dst_left == 0)
  420.                 {
  421.                     throw new RuntimeException(JText::sprintf('JLIB_FILESYSTEM_PATCHER_ADD_LINE', key($lines)));
  422.                 }
  423.                 $destin[] = substr($line, 1);
  424.                 $dst_left--;
  425.             }
  426.             elseif ($line != '\\ No newline at end of file')
  427.             {
  428.                 $line = substr($line, 1);
  429.                 $source[] = $line;
  430.                 $destin[] = $line;
  431.                 $src_left--;
  432.                 $dst_left--;
  433.             }
  434.             if ($src_left == 0 && $dst_left == 0)
  435.             {
  436.  
  437.                 // Now apply the patch, finally!
  438.                 if ($src_size > 0)
  439.                 {
  440.                     $src_lines = & $this->getSource($src);
  441.                     if (!isset($src_lines))
  442.                     {
  443.                         throw new RuntimeException(JText::sprintf('JLIB_FILESYSTEM_PATCHER_UNEXISING_SOURCE', $src));
  444.                     }
  445.                 }
  446.                 if ($dst_size > 0)
  447.                 {
  448.                     if ($src_size > 0)
  449.                     {
  450.                         $dst_lines = & $this->getDestination($dst, $src);
  451.                         $src_bottom = $src_line + count($source);
  452.                         for ($l = $src_line;$l < $src_bottom;$l++)
  453.                         {
  454.                             if ($src_lines[$l] != $source[$l - $src_line])
  455.                             {
  456.                                 throw new RuntimeException(JText::sprintf('JLIB_FILESYSTEM_PATCHER_FAILED_VERIFY', $src, $l));
  457.                             }
  458.                         }
  459.                         array_splice($dst_lines, $dst_line, count($source), $destin);
  460.                     }
  461.                     else
  462.                     {
  463.                         $this->destinations[$dst] = $destin;
  464.                     }
  465.                 }
  466.                 else
  467.                 {
  468.                     $this->removals[] = $src;
  469.                 }
  470.                 next($lines);
  471.                 return;
  472.             }
  473.             $line = next($lines);
  474.         }
  475.         while ($line !== false);
  476.         throw new RuntimeException('Unexpected EOF');
  477.     }

  479.     /**
  480.      * Get the lines of a source file
  481.      *
  482.      * @param   string  $src  The path of a file
  483.      *
  484.      * @return  array  The lines of the source file
  485.      *
  486.      * @since   12.1
  487.      */
  488.     protected function &getSource($src)
