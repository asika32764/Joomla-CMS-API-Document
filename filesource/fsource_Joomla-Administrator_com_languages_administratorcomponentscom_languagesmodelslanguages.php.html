  109.     protected function getListQuery()
  110.     {
  111.         // Create a new query object.
  112.         $db = $this->getDbo();
  113.         $query = $db->getQuery(true);
  114.  
  115.         // Select all fields from the languages table.
  116.         $query->select($this->getState('list.select', 'a.*', 'l.home'))
  117.             ->from($db->quoteName('#__languages') . ' AS a');

  119.         // Join over the asset groups.
  120.         $query->select('ag.title AS access_level')

  123.         // Select the language home pages

  125.             ->join('LEFT', $db->quoteName('#__menu') . ' AS l  ON  l.language = a.lang_code AND l.home=1  AND l.language <> ' . $db->quote('*'));
  126.  
  127.         // Filter on the published state.
  128.         $published = $this->getState('filter.published');
  129.         if (is_numeric($published))

  131.             $query->where('a.published = ' . (int) $published);
  132.         }
  133.         elseif ($published === '')
  134.         {
  135.             $query->where('(a.published IN (0, 1))');

  137.  
  138.         // Filter by search in title
  139.         $search = $this->getState('filter.search');
  140.         if (!empty($search))
  141.         {
  142.             $search = $db->quote('%' . $db->escape($search, true) . '%', false);
  143.             $query->where('(a.title LIKE ' . $search . ')');

  147.         if ($access = $this->getState('filter.access'))
