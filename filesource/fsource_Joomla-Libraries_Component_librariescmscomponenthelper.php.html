  106.     {
  107.         // Filter settings
  108.         $config     = static::getParams('com_config');
  109.         $user       = JFactory::getUser();
  110.         $userGroups = JAccess::getGroupsByUser($user->get('id'));
  111.  
  112.         $filters = $config->get('filters');
  113.  
  114.         $blackListTags       = array();

  117.         $customListTags       = array();
  118.         $customListAttributes = array();
  119.  
  120.         $whiteListTags       = array();

  123.         $whiteList  = false;
  124.         $blackList  = false;
  125.         $customList = false;
  126.         $unfiltered = false;
  127.  
  128.         // Cycle through each of the user groups the user is in.
  129.         // Remember they are included in the Public group as well.
  130.         foreach ($userGroups as $groupId)
  131.         {
  132.             // May have added a group by not saved the filters.
  133.             if (!isset($filters->$groupId))
  134.             {
  135.                 continue;
  136.             }
  137.  
  138.             // Each group the user is in could have different filtering properties.
  139.             $filterData = $filters->$groupId;
  140.             $filterType = strtoupper($filterData->filter_type);
  141.  
  142.             if ($filterType == 'NH')
  143.             {
  144.                 // Maximum HTML filtering.
  145.             }
  146.             elseif ($filterType == 'NONE')
  147.             {
  148.                 // No HTML filtering.
  149.                 $unfiltered = true;
  150.             }
  151.             else
  152.             {
  153.                 // Black or white list.
  154.                 // Preprocess the tags and attributes.
  155.                 $tags           = explode(',', $filterData->filter_tags);
  156.                 $attributes     = explode(',', $filterData->filter_attributes);
  157.                 $tempTags       = array();
  158.                 $tempAttributes = array();
  159.  
  160.                 foreach ($tags as $tag)
  161.                 {
  162.                     $tag = trim($tag);
  163.  
  164.                     if ($tag)
  165.                     {
  166.                         $tempTags[] = $tag;
  167.                     }
  168.                 }
  169.  
  170.                 foreach ($attributes as $attribute)
  171.                 {
  172.                     $attribute = trim($attribute);
  173.  
  174.                     if ($attribute)
  175.                     {
  176.                         $tempAttributes[] = $attribute;
  177.                     }
  178.                 }
  179.  
  180.                 // Collect the black or white list tags and attributes.
  181.                 // Each list is cummulative.
  182.                 if ($filterType == 'BL')
  183.                 {
  184.                     $blackList           = true;
  185.                     $blackListTags       = array_merge($blackListTags, $tempTags);
  186.                     $blackListAttributes = array_merge($blackListAttributes, $tempAttributes);
  187.                 }
  188.                 elseif ($filterType == 'CBL')
  189.                 {
  190.                     // Only set to true if Tags or Attributes were added
  191.                     if ($tempTags || $tempAttributes)
  192.                     {
  193.                         $customList           = true;
  194.                         $customListTags       = array_merge($customListTags, $tempTags);
  195.                         $customListAttributes = array_merge($customListAttributes, $tempAttributes);
  196.                     }
  197.                 }
  198.                 elseif ($filterType == 'WL')
  199.                 {
  200.                     $whiteList           = true;
  201.                     $whiteListTags       = array_merge($whiteListTags, $tempTags);
  202.                     $whiteListAttributes = array_merge($whiteListAttributes, $tempAttributes);
  203.                 }
  204.             }
  205.         }
  206.  
  207.         // Remove duplicates before processing (because the black list uses both sets of arrays).
  208.         $blackListTags        = array_unique($blackListTags);
  209.         $blackListAttributes  = array_unique($blackListAttributes);
  210.         $customListTags       = array_unique($customListTags);
  211.         $customListAttributes = array_unique($customListAttributes);
  212.         $whiteListTags        = array_unique($whiteListTags);
  213.         $whiteListAttributes  = array_unique($whiteListAttributes);
  214.  
  215.         // Unfiltered assumes first priority.
  216.         if ($unfiltered)
  217.         {
  218.             // Dont apply filtering.
  219.         }
  220.         else
  221.         {
  222.             // Custom blacklist precedes Default blacklist
  223.             if ($customList)
  224.             {
  225.                 $filter = JFilterInput::getInstance(array(), array(), 1, 1);
  226.  
  227.                 // Override filter's default blacklist tags and attributes
  228.                 if ($customListTags)
  229.                 {
  230.                     $filter->tagBlacklist = $customListTags;
  231.                 }
  232.                 if ($customListAttributes)
  233.                 {
  234.                     $filter->attrBlacklist = $customListAttributes;
  235.                 }
  236.             }
  237.             // Black lists take second precedence.
  238.             elseif ($blackList)
  239.             {
  240.                 // Remove the white-listed tags and attributes from the black-list.
  241.                 $blackListTags       = array_diff($blackListTags, $whiteListTags);
  242.                 $blackListAttributes = array_diff($blackListAttributes, $whiteListAttributes);
  243.  
  244.                 $filter = JFilterInput::getInstance($blackListTags, $blackListAttributes, 1, 1);
  245.  
  246.                 // Remove white listed tags from filter's default blacklist
  247.                 if ($whiteListTags)
  248.                 {
  249.                     $filter->tagBlacklist = array_diff($filter->tagBlacklist, $whiteListTags);
  250.                 }
  251.                 // Remove white listed attributes from filter's default blacklist
  252.                 if ($whiteListAttributes)
  253.                 {
  254.                     $filter->attrBlacklist = array_diff($filter->attrBlacklist);
  255.                 }
  256.             }
  257.             // White lists take third precedence.
  258.             elseif ($whiteList)
  259.             {
  260.                 // Turn off XSS auto clean
  261.                 $filter = JFilterInput::getInstance($whiteListTags, $whiteListAttributes, 0, 0, 0);
  262.             }
  263.             // No HTML takes last place.
  264.             else
  265.             {
  266.                 $filter = JFilterInput::getInstance();
  267.             }
  268.  
  269.             $text = $filter->clean($text, 'html');
  270.         }
  271.  
  272.         return $text;
  273.     }

  295.

  337.     }

  413.  
  414.         // Convert the params to an object.

  423.     }
