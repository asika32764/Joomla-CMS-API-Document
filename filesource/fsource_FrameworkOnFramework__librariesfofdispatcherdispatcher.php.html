  255.     public function dispatch()
  256.     {
  257.         if (!FOFPlatform::getInstance()->authorizeAdmin($this->input->getCmd('option', 'com_foobar')))
  258.         {
  259.             if (FOFPlatform::getInstance()->checkVersion(JVERSION, '3.0', 'ge'))
  260.             {
  261.                 throw new Exception(JText::_('JLIB_APPLICATION_ERROR_ACCESS_FORBIDDEN'), 403);
  262.             }
  263.             else

  266.             }
  267.         }
  268.  
  269.         $this->transparentAuthentication();
  270.  
  271.         // Merge English and local translations
  272.         FOFPlatform::getInstance()->loadTranslations($this->component);
  273.  
  274.         $canDispatch = true;
  275.  
  276.         if (FOFPlatform::getInstance()->isCli())
  277.         {
  278.             $canDispatch = $canDispatch && $this->onBeforeDispatchCLI();
  279.         }
  280.  
  281.         $canDispatch = $canDispatch && $this->onBeforeDispatch();
  282.  
  283.         if (!$canDispatch)
  284.         {
  285.             JResponse::setHeader('Status', '403 Forbidden', true);
  286.  
  287.             if (FOFPlatform::getInstance()->checkVersion(JVERSION, '3.0', 'ge'))
  288.             {
  289.                 throw new Exception(JText::_('JLIB_APPLICATION_ERROR_ACCESS_FORBIDDEN'), 403);
  290.             }
  291.             else
  292.             {
  293.                 return JError::raiseError('403', JText::_('JLIB_APPLICATION_ERROR_ACCESS_FORBIDDEN'));
  294.             }
  295.         }
  296.  
  297.         // Get and execute the controller
  298.         $option = $this->input->getCmd('option', 'com_foobar');
  299.         $view = $this->input->getCmd('view', $this->defaultView);
  300.         $task = $this->input->getCmd('task', null);
  301.  
  302.         if (empty($task))
  303.         {
  304.             $task = $this->getTask($view);
  305.         }
  306.  
  307.         // Pluralise/sungularise the view name for typical tasks
  308.  
  309.         if (in_array($task, array('edit', 'add', 'read')))
  310.         {
  311.             $view = FOFInflector::singularize($view);
  312.         }
  313.         elseif (in_array($task, array('browse')))
  314.         {
  315.             $view = FOFInflector::pluralize($view);
  316.         }
  317.  
  318.         $this->input->set('view', $view);
  319.         $this->input->set('task', $task);
  320.  
  321.         $config = $this->config;
  322.         $config['input'] = $this->input;
  323.  
  324.         $controller = FOFController::getTmpInstance($option, $view, $config);
  325.         $status = $controller->execute($task);
  326.  
  327.         if (!$this->onAfterDispatch())
  328.         {
  329.             JResponse::setHeader('Status', '403 Forbidden', true);
  330.  
  331.             if (FOFPlatform::getInstance()->checkVersion(JVERSION, '3.0', 'ge'))
  332.             {
  333.                 throw new Exception(JText::_('JLIB_APPLICATION_ERROR_ACCESS_FORBIDDEN'), 403);

  335.             else
  336.             {
  337.                 return JError::raiseError('403', JText::_('JLIB_APPLICATION_ERROR_ACCESS_FORBIDDEN'));

  361.     }

  363.     /**
  364.      * Tries to guess the controller task to execute based on the view name and
  365.      * the HTTP request method.
  366.      *
  367.      * @param   string  $view  The name of the view
  368.      *
  369.      * @return  string  The best guess of the task to execute
  370.      */
  371.     protected function getTask($view)
  372.     {
  373.         // Get a default task based on plural/singular view
  374.         $request_task = $this->input->getCmd('task', null);
  375.         $task = FOFInflector::isPlural($view) ? 'browse' : 'edit';
  376.  
  377.         // Get a potential ID, we might need it later
  378.         $id = $this->input->get('id', null, 'int');
  379.  
  380.         if ($id == 0)
  381.         {
  382.             $ids = $this->input->get('ids', array(), 'array');
  383.  
  384.             if (!empty($ids))
  385.             {
  386.                 $id = array_shift($ids);
  387.             }
  388.         }
  389.  
  390.         // Check the request method
  391.  
  392.         if (!isset($_SERVER['REQUEST_METHOD']))
  393.         {
  394.             $_SERVER['REQUEST_METHOD'] = 'GET';
  395.         }
  396.  
  397.         $requestMethod = strtoupper($_SERVER['REQUEST_METHOD']);
  398.  
  399.         switch ($requestMethod)
  400.         {
  401.             case 'POST':
  402.             case 'PUT':
  403.                 if (!is_null($id))
  404.                 {
  405.                     $task = 'save';
  406.                 }
  407.                 break;
  408.  
  409.             case 'DELETE':
  410.                 if ($id != 0)
  411.                 {
  412.                     $task = 'delete';
  413.                 }
  414.                 break;
  415.  
  416.             case 'GET':
  417.             default:
  418.                 // If it's an edit without an ID or ID=0, it's really an add
  419.                 if (($task == 'edit') && ($id == 0))
  420.                 {
  421.                     $task = 'add';
  422.                 }
  423.  
  424.                 // If it's an edit in the frontend, it's really a read
  425.                 elseif (($task == 'edit') && FOFPlatform::getInstance()->isFrontend())
  426.                 {
  427.                     $task = 'read';
  428.                 }
  429.                 break;
  430.         }
  431.  
  432.         return $task;
  433.     }

  479.             }
  480.         }
  481.  
  482.         return true;
  483.     }

  502.     /**
  503.      * Transparently authenticates a user
  504.      *
  505.      * @return  void 
  506.      */
  507.     public function transparentAuthentication()
  508.     {
  509.         // Only run when there is no logged in user
  510.  
  511.         if (!FOFPlatform::getInstance()->getUser()->guest)
  512.         {
  513.             return;
  514.         }
  515.  
  516.         // @todo Check the format
  517.         $format = $this->input->getCmd('format', 'html');
  518.  
  519.         if (!in_array($format, $this->fofAuth_Formats))
  520.         {
  521.             return;
  522.         }
  523.  
  524.         foreach ($this->fofAuth_AuthMethods as $method)
  525.         {
  526.             // If we're already logged in, don't bother
  527.  
  528.             if ($this->_fofAuth_isLoggedIn)
  529.             {
  530.                 continue;
  531.             }
  532.  
  533.             // This will hold our authentication data array (username, password)
  534.             $authInfo = null;
  535.  
  536.             switch ($method)
  537.             {
  538.                 case 'HTTPBasicAuth_TOTP':
  539.  
  540.                     if (empty($this->fofAuth_Key))
  541.                     {
  542.                         continue;
  543.                     }
  544.  
  545.                     if (!isset($_SERVER['PHP_AUTH_USER']))
  546.                     {
  547.                         continue;
  548.                     }
  549.  
  550.                     if (!isset($_SERVER['PHP_AUTH_PW']))
  551.                     {
  552.                         continue;
  553.                     }
  554.  
  555.                     if ($_SERVER['PHP_AUTH_USER'] != '_fof_auth')
  556.                     {
  557.                         continue;
  558.                     }
  559.  
  560.                     $encryptedData = $_SERVER['PHP_AUTH_PW'];
  561.  
  562.                     $authInfo = $this->_decryptWithTOTP($encryptedData);
  563.                     break;
  564.  
  565.                 case 'QueryString_TOTP':
  566.                     $encryptedData = $this->input->get('_fofauthentication', '', 'raw');
  567.  
  568.                     if (empty($encryptedData))
  569.                     {
  570.                         continue;
  571.                     }
  572.  
  573.                     $authInfo = $this->_decryptWithTOTP($encryptedData);
  574.                     break;
  575.  
  576.                 case 'HTTPBasicAuth_Plaintext':
  577.                     if (!isset($_SERVER['PHP_AUTH_USER']))
  578.                     {
  579.                         continue;
  580.                     }
  581.  
  582.                     if (!isset($_SERVER['PHP_AUTH_PW']))
  583.                     {
  584.                         continue;
  585.                     }
  586.  
  587.                     $authInfo = array(
  588.                         'username'     => $_SERVER['PHP_AUTH_USER'],
  589.                         'password'     => $_SERVER['PHP_AUTH_PW']
  590.                     );
  591.                     break;
  592.  
  593.                 case 'QueryString_Plaintext':
  594.                     $jsonencoded = $this->input->get('_fofauthentication', '', 'raw');
  595.  
  596.                     if (empty($jsonencoded))
  597.                     {
  598.                         continue;
  599.                     }
  600.  
  601.                     $authInfo = json_decode($jsonencoded, true);
  602.  
  603.                     if (!is_array($authInfo))
  604.                     {
  605.                         $authInfo = null;
  606.                     }
  607.                     elseif (!array_key_exists('username', $authInfo) || !array_key_exists('password', $authInfo))
  608.                     {
  609.                         $authInfo = null;
  610.                     }
  611.                     break;
  612.  
  613.                 case 'SplitQueryString_Plaintext':
  614.                     $authInfo = array(
  615.                         'username'     => $this->input->get('_fofauthentication_username', '', 'raw'),
  616.                         'password'     => $this->input->get('_fofauthentication_password', '', 'raw'),
  617.                     );
  618.  
  619.                     if (empty($authInfo['username']))
  620.                     {
  621.                         $authInfo = null;
  622.                     }
  623.  
  624.                     break;
  625.  
  626.                 default:
  627.                     continue;
  628.  
  629.                     break;
  630.             }
  631.  
  632.             // No point trying unless we have a username and password
  633.             if (!is_array($authInfo))
  634.             {
  635.                 continue;
  636.             }
  637.  
  638.             $this->_fofAuth_isLoggedIn = FOFPlatform::getInstance()->loginUser($authInfo);
  639.         }
  640.     }
  641.  
  642.     /**
  643.      * Decrypts a transparent authentication message using a TOTP
  644.      *
  645.      * @param   string  $encryptedData  The encrypted data
  646.      *
  647.      * @return  array  The decrypted data
  648.      */
  649.     private function _decryptWithTOTP($encryptedData)
  650.     {
  651.         if (empty($this->fofAuth_Key))
  652.         {
  653.             $this->_fofAuth_CryptoKey = null;
  654.  
  655.             return null;
  656.         }
  657.  
  658.         $totp = new FOFEncryptTotp($this->fofAuth_timeStep);
  659.         $period = $totp->getPeriod();
  660.         $period--;
  661.  
  662.         for ($i = 0; $i <= 2; $i++)
  663.         {
  664.             $time = ($period + $i) * $this->fofAuth_timeStep;
  665.             $otp = $totp->getCode($this->fofAuth_Key, $time);
  666.             $this->_fofAuth_CryptoKey = hash('sha256', $this->fofAuth_Key . $otp);
  667.  
  668.             $aes = new FOFEncryptAes($this->_fofAuth_CryptoKey);
  669.             $ret = $aes->decryptString($encryptedData);
  670.             $ret = rtrim($ret, "\000");
  671.  
  672.             $ret = json_decode($ret, true);
  673.  
  674.             if (!is_array($ret))
  675.             {
  676.                 continue;
  677.             }
  678.  
  679.             if (!array_key_exists('username', $ret))
  680.             {
  681.                 continue;
  682.             }
  683.  
  684.             if (!array_key_exists('password', $ret))
  685.             {
  686.                 continue;
  687.             }
  688.  
  689.             // Successful decryption!
  690.             return $ret;
  691.         }
  692.  
  693.         // Obviously if we're here we could not decrypt anything. Bail out.
  694.         $this->_fofAuth_CryptoKey = null;
  695.  
  696.         return null;
  697.     }
  698.  
  699.     /**
  700.      * Creates a decryption key for use with the TOTP decryption method
  701.      *
  702.      * @param   integer  $time  The timestamp used for TOTP calculation, leave empty to use current timestamp
  703.      *
  704.      * @return  string  THe encryption key
  705.      */
  706.     private function _createDecryptionKey($time = null)
  707.     {
  708.         $totp = new FOFEncryptTotp($this->fofAuth_timeStep);
  709.         $otp = $totp->getCode($this->fofAuth_Key, $time);
  710.  
  711.         $key = hash('sha256', $this->fofAuth_Key . $otp);
  712.  
  713.         return $key;
  714.     }
