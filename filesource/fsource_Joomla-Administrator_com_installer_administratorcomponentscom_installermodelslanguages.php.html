  150.     public function install($lids)
  151.     {
  152.         $app       = JFactory::getApplication();
  153.         $installer = JInstaller::getInstance();
  154.  
  155.         // Loop through every selected language
  156.         foreach ($lids as $id)
  157.         {
  158.             // Loads the update database object that represents the language
  159.             $language = JTable::getInstance('update');
  160.             $language->load($id);
  161.  
  162.             // Get the url to the XML manifest file of the selected language
  163.             $remote_manifest = $this->_getLanguageManifest($id);
  164.             if (!$remote_manifest)
  165.             {
  166.                 // Could not find the url, the information in the update server may be corrupt
  167.                 $message  = JText::sprintf('COM_INSTALLER_MSG_LANGUAGES_CANT_FIND_REMOTE_MANIFEST', $language->name);
  168.                 $message .= ' ' . JText::_('COM_INSTALLER_MSG_LANGUAGES_TRY_LATER');
  169.                 $app->enqueueMessage($message);
  170.                 continue;
  171.             }
  172.  
  173.             // Based on the language XML manifest get the url of the package to download
  174.             $package_url = $this->_getPackageUrl($remote_manifest);
  175.             if (!$package_url)
  176.             {
  177.                 // Could not find the url , maybe the url is wrong in the update server, or there is not internet access
  178.                 $message  = JText::sprintf('COM_INSTALLER_MSG_LANGUAGES_CANT_FIND_REMOTE_PACKAGE', $language->name);
  179.                 $message .= ' ' . JText::_('COM_INSTALLER_MSG_LANGUAGES_TRY_LATER');
  180.                 $app->enqueueMessage($message);
  181.                 continue;
  182.             }
  183.  
  184.             // Download the package to the tmp folder
  185.             $package = $this->_downloadPackage($package_url);
  186.  
  187.             // Install the package
  188.             if (!$installer->install($package['dir']))
  189.             {
  190.                 // There was an error installing the package
  191.                 $message  = JText::sprintf('COM_INSTALLER_INSTALL_ERROR', $language->name);
  192.                 $message .= ' ' . JText::_('COM_INSTALLER_MSG_LANGUAGES_TRY_LATER');
  193.                 $app->enqueueMessage($message);
  194.                 continue;
  195.             }
  196.  
  197.             // Package installed successfully
  198.             $app->enqueueMessage(JText::sprintf('COM_INSTALLER_INSTALL_SUCCESS', $language->name));
  199.  
  200.             // Cleanup the install files in tmp folder
  201.             if (!is_file($package['packagefile']))
  202.             {
  203.                 $config = JFactory::getConfig();
  204.                 $package['packagefile'] = $config->get('tmp_path') . '/' . $package['packagefile'];
  205.             }
  206.             JInstallerHelper::cleanupInstall($package['packagefile'], $package['extractdir']);
  207.  
  208.             // Delete the installed language from the list
  209.             $language->delete($id);
  210.         }
  211.  
  212.     }
