  81.                     if ($filesize <= 0)

  84.                         $fp = @fopen($target, 'wb');

  86.                 }

  88.  
  89.             // If we have no download, close the file pointer
  90.             if ($result === false)

  93.             }

  95.  
  96.         if ($result === false)

  100.             if (file_exists($target))

  108.             // Download and write using JFile::write();

  111.  
  112.         return $result;
  113.     }

  159.     /**
  160.      * Downloads the contents of a URL and writes them to disk (if $fp is not null)
  161.      * or returns them as a string (if $fp is null)
  162.      *
  163.      * @param   string    $url       The URL to download from
  164.      * @param   resource  $fp        The file pointer to download to. Omit to return the contents.
  165.      * @param   boolean   $nofollow  Should we follow 301/302/307 redirection HTTP headers?
  166.      *
  167.      * @return   bool|stringFalse on failure, true on success ($fp not null) or the URL contents (if $fp is null)
  168.      *

  170.      */
  171.     private static function &getCURL($url, $fp = null, $nofollow = false)
  172.     {
  173.         $ch = curl_init($url);
  174.  
  175.         if ( !@curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1) && !$nofollow )
  176.         {
  177.  
  178.             // Safe Mode is enabled. We have to fetch the headers and
  179.             // parse any redirections present in there.
  180.             curl_setopt($ch, CURLOPT_AUTOREFERER, true);
  181.             curl_setopt($ch, CURLOPT_FAILONERROR, true);
  182.             curl_setopt($ch, CURLOPT_HEADER, true);
  183.             curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
  184.             curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
  185.             curl_setopt($ch, CURLOPT_CONNECTTIMEOUT, 10);
  186.             curl_setopt($ch, CURLOPT_TIMEOUT, 30);
  187.  
  188.             // Get the headers
  189.             $data = curl_exec($ch);
  190.             curl_close($ch);
  191.  
  192.             // Init
  193.             $newURL = $url;
  194.  
  195.             // Parse the headers
  196.             $lines = explode("\n", $data);
  197.             foreach ($lines as $line)
  198.             {
  199.                 if (substr($line, 0, 9) == "Location:")
  200.                 {
  201.                     $newURL = trim(substr($line, 9));
  202.                 }
  203.             }
  204.  
  205.             if ($url != $newURL)
  206.             {
  207.                 return self::getCURL($newURL, $fp);
  208.             }
  209.             else
  210.             {
  211.                 return self::getCURL($newURL, $fp, true);
  212.             }
  213.         }
  214.         else
  215.         {
  216.             @curl_setopt($ch, CURLOPT_MAXREDIRS, 20);
  217.             if (function_exists('set_time_limit'))
  218.             {
  219.                 set_time_limit(0);
  220.             }
  221.         }
  222.  
  223.         curl_setopt($ch, CURLOPT_HEADER, 0);
  224.         curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
  225.         curl_setopt($ch, CURLOPT_USERAGENT, 'Joomla/' . JVERSION);
  226.  
  227.         if (is_resource($fp))
  228.         {
  229.             curl_setopt($ch, CURLOPT_FILE, $fp);
  230.         }
  231.         else
  232.         {
  233.             curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
  234.         }
  235.  
  236.         $result = curl_exec($ch);
  237.         curl_close($ch);
  238.  
  239.         return $result;
  240.     }

  271.     /**
  272.      * Download from a URL using URL fopen() wrappers
  273.      *
  274.      * @param   string    $url  The URL to download from
  275.      * @param   resource  $fp   The file pointer to download to; leave null to return the d/l file as a string
  276.      *
  277.      * @return  bool|stringFalse on failure, true on success ($fp not null) or the URL contents (if $fp is null)
  278.      *
  279.      * @since   2.5.4
  280.      */
  281.     private static function &getFOPEN($url, $fp = null)
  282.     {
  283.         $result = false;
  284.  
  285.         // Open the URL for reading
  286.         if (function_exists('stream_context_create'))
  287.         {
  288.             $opts = stream_context_get_options(stream_context_get_default());
  289.             $opts['http']['user_agent'] = 'Joomla/' . JVERSION;
  290.             $context = stream_context_create($opts);
  291.             $ih = @fopen($url, 'r', false, $context);
  292.         }
  293.         else
  294.         {
  295.  
  296.             // PHP 4 way (actually, it's just a fallback)
  297.             if ( function_exists('ini_set') )
  298.             {
  299.                 ini_set('user_agent', 'Joomla/' . JVERSION);
  300.             }
  301.             $ih = @fopen($url, 'r');
  302.         }
  303.  
  304.         // If fopen() fails, abort
  305.         if ( !is_resource($ih) )
  306.         {
  307.             return $result;
  308.         }
  309.  
  310.         // Try to download
  311.         $bytes = 0;
  312.         $result = true;
  313.         $return = '';
  314.         while (!feof($ih) && $result)
  315.         {
  316.             $contents = fread($ih, 4096);
  317.             if ($contents === false)
  318.             {
  319.                 @fclose($ih);
  320.                 $result = false;
  321.                 return $result;
  322.             }
  323.             else
  324.             {
  325.                 $bytes += strlen($contents);
  326.                 if (is_resource($fp))
  327.                 {
  328.                     $result = @fwrite($fp, $contents);
  329.                 }
  330.                 else
  331.                 {
  332.                     $return .= $contents;
  333.                     unset($contents);
  334.                 }
  335.             }
  336.         }
  337.  
  338.         @fclose($ih);
  339.  
  340.         if (is_resource($fp))
  341.         {
  342.             return $result;
  343.         }
  344.         elseif ( $result === true )
  345.         {
  346.             return $return;
  347.         }
  348.         else
  349.         {
  350.             return $result;
  351.         }
  352.     }

  377.     /**
  378.      * Change the permissions of a file, optionally using FTP
  379.      *
  380.      * @param   string  $path  Absolute path to file
  381.      * @param   int     $mode  Permissions, e.g. 0755
  382.      *
  383.      * @return  boolean True on success
  384.      *
  385.      * @since   2.5.4
  386.      */
  387.     private static function chmod($path, $mode)
  388.     {
  389.         if (is_string($mode))
  390.         {
  391.             $mode = octdec($mode);
  392.             if ( ($mode < 0600) || ($mode > 0777) )
  393.             {
  394.                 $mode = 0755;
  395.             }
  396.         }
  397.  
  398.         $ftpOptions = JClientHelper::getCredentials('ftp');
  399.  
  400.         // Check to make sure the path valid and clean
  401.         $path = JPath::clean($path);
  402.  
  403.         if ($ftpOptions['enabled'] == 1)
  404.         {
  405.  
  406.             // Connect the FTP client
  407.             $ftp = JClientFtp::getInstance(
  408.                 $ftpOptions['host'], $ftpOptions['port'], null,
  409.                 $ftpOptions['user'], $ftpOptions['pass']
  410.             );
  411.         }
  412.  
  413.         if (@chmod($path, $mode))
  414.         {
  415.             $ret = true;
  416.         }
  417.         elseif ($ftpOptions['enabled'] == 1)
  418.         {
  419.             // Translate path and delete
  420.             $path = JPath::clean(str_replace(JPATH_ROOT, $ftpOptions['root'], $path), '/');
  421.  
  422.             // FTP connector throws an error
  423.             $ret = $ftp->chmod($path, $mode);
  424.         } else
  425.         {
  426.             return false;
  427.         }
  428.         return $ret;
  429.     }
