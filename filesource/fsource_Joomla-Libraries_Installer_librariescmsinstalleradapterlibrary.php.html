  64.         /*
  65.          * ---------------------------------------------------------------------------------------------
  66.          * Manifest Document Setup Section

  70.         // Set the extension's name

  73.         $this->set('name', $name);

  75.  
  76.         $db = $this->parent->getDbo();

  78.             ->select($db->quoteName('extension_id'))

  82.         $db->setQuery($query);
  83.         $result = $db->loadResult();
  84.  
  85.         if ($result)
  86.         {

  88.             if ($this->parent->isOverwrite() || $this->parent->isUpgrade())
  89.             {
  90.                 // We can upgrade, so uninstall the old one
  91.                 $installer = new JInstaller; // we don't want to compromise this instance!
  92.                 $installer->uninstall('library', $result);
  93.             }
  94.             else
  95.             {

  97.                 $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_LIB_INSTALL_ALREADY_INSTALLED'));

  101.         }
  102.

  104.         $description = (string) $this->manifest->description;
  105.  
  106.         if ($description)
  107.         {

  109.         }
  110.         else
  111.         {
  112.             $this->parent->set('message', '');
  113.         }
  114.  
  115.         // Set the installation path
  116.         $group = (string) $this->manifest->libraryname;
  117.  
  118.         if (!$group)
  119.         {
  120.             $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_LIB_INSTALL_NOFILE'));
  121.  
  122.             return false;
  123.         }
  124.         else
  125.         {
  126.             $this->parent->setPath('extension_root', JPATH_PLATFORM . '/' . implode(DIRECTORY_SEPARATOR, explode('/', $group)));
  127.         }
  128.  
  129.         /*
  130.          * ---------------------------------------------------------------------------------------------
  131.          * Filesystem Processing Section
  132.          * ---------------------------------------------------------------------------------------------
  133.          */
  134.  
  135.         // If the library directory does not exist, let's create it
  136.         $created = false;
  137.  
  138.         if (!file_exists($this->parent->getPath('extension_root')))
  139.         {
  140.             if (!$created = JFolder::create($this->parent->getPath('extension_root')))
  141.             {
  142.                 $this->parent->abort(
  143.                     JText::sprintf('JLIB_INSTALLER_ABORT_LIB_INSTALL_FAILED_TO_CREATE_DIRECTORY', $this->parent->getPath('extension_root'))
  144.                 );
  145.  
  146.                 return false;
  147.             }
  148.         }
  149.  
  150.         /*
  151.          * If we created the library directory and will want to remove it if we
  152.          * have to roll back the installation, let's add it to the installation
  153.          * step stack
  154.          */
  155.         if ($created)

  157.             $this->parent->pushStep(array('type' => 'folder', 'path' => $this->parent->getPath('extension_root')));

  191.         if (!$row->store())

  193.             // Install failed, roll back changes

  195.  
  196.             return false;
  197.         }
  198.  
  199.         /**
  200.          * ---------------------------------------------------------------------------------------------
  201.          * Finalization and Cleanup Section
  202.          * ---------------------------------------------------------------------------------------------
  203.          */
  204.  
  205.         // Lastly, we will copy the manifest file to its appropriate place.
  206.         $manifest = array();
  207.         $manifest['src'] = $this->parent->getPath('manifest');
  208.         $manifest['dest'] = JPATH_MANIFESTS . '/libraries/' . basename($this->parent->getPath('manifest'));
  209.  
  210.         if (!$this->parent->copyFiles(array($manifest), true))
  211.         {
  212.             // Install failed, rollback changes
  213.             $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_LIB_INSTALL_COPY_SETUP'));
  214.  
  215.             return false;
  216.         }
  217.         return $row->get('extension_id');
  218.     }

  266.     /**
  267.      * Custom uninstall method
  268.      *
  269.      * @param   string  $id  The id of the library to uninstall.
  270.      *
  271.      * @return  boolean  True on success
  272.      *
  273.      * @since   3.1
  274.      */
  275.     public function uninstall($id)
  276.     {
  277.         $retval = true;
  278.  
  279.         // First order of business will be to load the module object table from the database.
  280.         // This should give us the necessary information to proceed.
  281.         $row = JTable::getInstance('extension');
  282.  
  283.         if (!$row->load((int) $id) || !strlen($row->element))
  284.         {
  285.             JLog::add(JText::_('ERRORUNKOWNEXTENSION'), JLog::WARNING, 'jerror');
  286.  
  287.             return false;
  288.         }
  289.  
  290.         // Is the library we are trying to uninstall a core one?
  291.         // Because that is not a good idea...
  292.         if ($row->protected)
  293.         {
  294.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_LIB_UNINSTALL_WARNCORELIBRARY'), JLog::WARNING, 'jerror');
  295.  
  296.             return false;
  297.         }
  298.  
  299.         $manifestFile = JPATH_MANIFESTS . '/libraries/' . $row->element . '.xml';
  300.  
  301.         // Because libraries may not have their own folders we cannot use the standard method of finding an installation manifest
  302.         if (file_exists($manifestFile))
  303.         {
  304.             $manifest = new JInstallerManifestLibrary($manifestFile);
  305.  
  306.             // Set the library root path
  307.             $this->parent->setPath('extension_root', JPATH_PLATFORM . '/' . $manifest->libraryname);
  308.  
  309.             $xml = simplexml_load_file($manifestFile);
  310.  
  311.             // If we cannot load the XML file return null
  312.             if (!$xml)
  313.             {
  314.                 JLog::add(JText::_('JLIB_INSTALLER_ERROR_LIB_UNINSTALL_LOAD_MANIFEST'), JLog::WARNING, 'jerror');
  315.  
  316.                 return false;
  317.             }
  318.  
  319.             // Check for a valid XML root tag.
  320.             if ($xml->getName() != 'extension')
  321.             {
  322.                 JLog::add(JText::_('JLIB_INSTALLER_ERROR_LIB_UNINSTALL_INVALID_MANIFEST'), JLog::WARNING, 'jerror');
  323.  
  324.                 return false;
  325.             }
  326.  
  327.             $this->parent->removeFiles($xml->files, -1);
  328.             JFile::delete($manifestFile);
  329.  
  330.         }
  331.         else
  332.         {
  333.             // Remove this row entry since its invalid
  334.             $row->delete($row->extension_id);
  335.             unset($row);
  336.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_LIB_UNINSTALL_INVALID_NOTFOUND_MANIFEST'), JLog::WARNING, 'jerror');
  337.  
  338.             return false;
  339.         }
  340.  
  341.         // TODO: Change this so it walked up the path backwards so we clobber multiple empties
  342.         // If the folder is empty, let's delete it
  343.         if (JFolder::exists($this->parent->getPath('extension_root')))
  344.         {
  345.             if (is_dir($this->parent->getPath('extension_root')))
  346.             {
  347.                 $files = JFolder::files($this->parent->getPath('extension_root'));
  348.  
  349.                 if (!count($files))
  350.                 {
  351.                     JFolder::delete($this->parent->getPath('extension_root'));
  352.                 }
  353.             }
  354.         }
  355.  
  356.         $this->parent->removeFiles($xml->media);
  357.         $this->parent->removeFiles($xml->languages);
  358.  
  359.         $row->delete($row->extension_id);
  360.         unset($row);
  361.  
  362.         return $retval;
  363.     }

  395.     /**
  396.      * Custom discover_install method
  397.      *
  398.      * @return  boolean  True on success
  399.      *
  400.      * @since   3.1
  401.      */
  402.     public function discover_install()
  403.     {
  404.         /* Libraries are a strange beast; they are actually references to files
  405.          * There are two parts to a library which are disjunct in their locations
  406.          * 1) The manifest file (stored in /JPATH_MANIFESTS/libraries)
  407.          * 2) The actual files (stored in /JPATH_PLATFORM/libraryname)
  408.          * Thus installation of a library is the process of dumping files
  409.          * in two different places. As such it is impossible to perform
  410.          * any operation beyond mere registration of a library under the presumption
  411.          * that the files exist in the appropriate location so that come uninstall
  412.          * time they can be adequately removed.
  413.          */
  414.  
  415.         $manifestPath = JPATH_MANIFESTS . '/libraries/' . $this->parent->extension->element . '.xml';
  416.         $this->parent->manifest = $this->parent->isManifest($manifestPath);
  417.         $this->parent->setPath('manifest', $manifestPath);
  418.         $manifest_details = JInstaller::parseXMLInstallFile($this->parent->getPath('manifest'));
  419.         $this->parent->extension->manifest_cache = json_encode($manifest_details);
  420.         $this->parent->extension->state = 0;
  421.         $this->parent->extension->name = $manifest_details['name'];
  422.         $this->parent->extension->enabled = 1;
  423.         $this->parent->extension->params = $this->parent->getParams();
  424.  
  425.         if ($this->parent->extension->store())
  426.         {
  427.             return true;
  428.         }
  429.         else
  430.         {
  431.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_LIB_DISCOVER_STORE_DETAILS'), JLog::WARNING, 'jerror');
  432.  
  433.             return false;
  434.         }
  435.     }

  437.     /**
  438.      * Refreshes the extension table cache

  441.      *
  442.      * @since   3.1
  443.      */
  444.     public function refreshManifestCache()
  445.     {
  446.         // Need to find to find where the XML file is since we don't store this normally
  447.         $manifestPath = JPATH_MANIFESTS . '/libraries/' . $this->parent->extension->element . '.xml';
  448.         $this->parent->manifest = $this->parent->isManifest($manifestPath);
  449.         $this->parent->setPath('manifest', $manifestPath);
  450.  
  451.         $manifest_details = JInstaller::parseXMLInstallFile($this->parent->getPath('manifest'));
  452.         $this->parent->extension->manifest_cache = json_encode($manifest_details);
  453.         $this->parent->extension->name = $manifest_details['name'];
  454.  
  455.         try
  456.         {
  457.             return $this->parent->extension->store();
  458.         }
