  307.             // Store debug information

  311.  
  312.                 if (!array_key_exists($key, $this->used))
  313.                 {
  314.                     $this->used[$key] = array();
  315.                 }
  316.  
  317.                 $this->used[$key][] = $caller;
  318.             }
  319.         }
  320.         else
  321.         {

  323.             {
  324.                 $caller = $this->getCallerInfo();
  325.                 $caller['string'] = $string;
  326.  
  327.                 if (!array_key_exists($key, $this->orphans))
  328.                 {
  329.                     $this->orphans[$key] = array();
  330.                 }
  331.  
  332.                 $this->orphans[$key][] = $caller;
  333.  
  334.                 $string = '??' . $string . '??';
  335.             }
  336.         }
  337.  
  338.         if ($jsSafe)
  339.         {
  340.             // Javascript filter
  341.             $string = addslashes($string);
  342.         }
  343.         elseif ($interpretBackSlashes)
  344.         {
  345.             // Interpret \n and \t characters
  346.             $string = str_replace(array('\\\\', '\t', '\n'), array("\\", "\t", "\n"), $string);
  347.         }
  348.  
  349.         return $string;
  350.     }

  691.      * @param   boolean  $default    Flag that force the default language to be loaded if the current does not exist.

  697.     public function load($extension = 'joomla', $basePath = JPATH_BASE, $lang = null, $reload = false, $default = true)
  698.     {
  699.         if (!$lang)
  700.         {
  701.             $lang = $this->lang;
  702.         }
  703.  
  704.         $path = self::getLanguagePath($basePath, $lang);
  705.  
  706.         $internal = $extension == 'joomla' || $extension == '';
  707.         $filename = $internal ? $lang : $lang . '.' . $extension;
  708.         $filename = "$path/$filename.ini";
  709.  
  710.         if (isset($this->paths[$extension][$filename]) && !$reload)
  711.         {
  712.             // This file has already been tested for loading.
  713.             $result = $this->paths[$extension][$filename];
  714.         }
  715.         else
  716.         {
  717.             // Load the language file
  718.             $result = $this->loadLanguage($filename, $extension);
  719.  
  720.             // Check whether there was a problem with loading the file
  721.             if ($result === false && $default)
  722.             {
  723.                 // No strings, so either file doesn't exist or the file is invalid
  724.                 $oldFilename = $filename;
  725.  
  726.                 // Check the standard file name
  727.                 $path = self::getLanguagePath($basePath, $this->default);
  728.                 $filename = $internal ? $this->default : $this->default . '.' . $extension;
  729.                 $filename = "$path/$filename.ini";
  730.  
  731.                 // If the one we tried is different than the new name, try again
  732.                 if ($oldFilename != $filename)
  733.                 {
  734.                     $result = $this->loadLanguage($filename, $extension, false);
  735.                 }
  736.             }
  737.         }
  738.  
  739.         return $result;
  740.     }

  742.     /**
  743.      * Loads a language file.
  744.      *
  745.      * This method will not note the successful loading of a file - use load() instead.
  746.      *
  747.      * @param   string  $filename   The name of the file.
  748.      * @param   string  $extension  The name of the extension.
  749.      *
  750.      * @return  boolean  True if new strings have been added to the language
  751.      *
  752.      * @see     JLanguage::load()
  753.      * @since   11.1
  754.      */
  755.     protected function loadLanguage($filename, $extension = 'unknown')
  756.     {
  757.         $this->counter++;
  758.  
  759.         $result = false;
  760.         $strings = false;
  761.  
  762.         if (file_exists($filename))
  763.         {
  764.             $strings = $this->parse($filename);
  765.         }
  766.  
  767.         if ($strings)
  768.         {
  769.             if (is_array($strings))
  770.             {
  771.                 // Sort the underlying heap by key values to optimize merging
  772.                 ksort($strings, SORT_STRING);
  773.                 $this->strings = array_merge($this->strings, $strings);
  774.             }
  775.  
  776.             if (is_array($strings) && count($strings))
  777.             {
  778.                 // Do not bother with ksort here.  Since the originals were sorted, PHP will already have chosen the best heap.
  779.                 $this->strings = array_merge($this->strings, $this->override);
  780.                 $result = true;
  781.             }
  782.         }
  783.  
  784.         // Record the result of loading the extension's file.
  785.         if (!isset($this->paths[$extension]))
  786.         {
  787.             $this->paths[$extension] = array();
  788.         }
  789.  
  790.         $this->paths[$extension][$filename] = $result;
  791.  
  792.         return $result;
  793.     }

  795.     /**
  796.      * Parses a language file.
  797.      *
  798.      * @param   string  $filename  The name of the file.
  799.      *
  800.      * @return  array  The array of parsed strings.
  801.      *
  802.      * @since   11.1
  803.      */
  804.     protected function parse($filename)
  805.     {
  806.         if ($this->debug)
  807.         {
  808.             // Capture hidden PHP errors from the parsing.
  809.             $php_errormsg = null;
  810.             $track_errors = ini_get('track_errors');
  811.             ini_set('track_errors', true);
  812.         }
  813.  
  814.         $contents = file_get_contents($filename);
  815.         $contents = str_replace('_QQ_', '"\""', $contents);
  816.         $strings = @parse_ini_string($contents);
  817.  
  818.         if (!is_array($strings))
  819.         {
  820.             $strings = array();
  821.         }
  822.  
  823.         if ($this->debug)
  824.         {
  825.             // Restore error tracking to what it was before.
  826.             ini_set('track_errors', $track_errors);
  827.  
  828.             // Initialise variables for manually parsing the file for common errors.
  829.             $blacklist = array('YES', 'NO', 'NULL', 'FALSE', 'ON', 'OFF', 'NONE', 'TRUE');
  830.             $regex = '/^(|(\[[^\]]*\])|([A-Z][A-Z0-9_\-\.]*\s*=(\s*(("[^"]*")|(_QQ_)))+))\s*(;.*)?$/';
  831.             $this->debug = false;
  832.             $errors = array();
  833.  
  834.             // Open the file as a stream.
  835.             $file = new SplFileObject($filename);
  836.  
  837.             foreach ($file as $lineNumber => $line)
  838.             {
  839.                 // Avoid BOM error as BOM is OK when using parse_ini
  840.                 if ($lineNumber == 0)
  841.                 {
  842.                     $line = str_replace("\xEF\xBB\xBF", '', $line);
  843.                 }
  844.  
  845.                 // Check that the key is not in the blacklist and that the line format passes the regex.
  846.                 $key = strtoupper(trim(substr($line, 0, strpos($line, '='))));
  847.  
  848.                 // Workaround to reduce regex complexity when matching escaped quotes
  849.                 $line = str_replace('\"', '_QQ_', $line);
  850.  
  851.                 if (!preg_match($regex, $line) || in_array($key, $blacklist))
  852.                 {
  853.                     $errors[] = $lineNumber;
  854.                 }
  855.             }
  856.  
  857.             // Check if we encountered any errors.
  858.             if (count($errors))
  859.             {
  860.                 if (basename($filename) != $this->lang . '.ini')
  861.                 {
  862.                     $this->errorfiles[$filename] = $filename . JText::sprintf('JERROR_PARSING_LANGUAGE_FILE', implode(', ', $errors));
  863.                 }
  864.                 else
  865.                 {
  866.                     $this->errorfiles[$filename] = $filename . '&#160;: error(s) in line(s) ' . implode(', ', $errors);
  867.                 }
  868.             }
  869.             elseif ($php_errormsg)
  870.             {
  871.                 // We didn't find any errors but there's probably a parse notice.
  872.                 $this->errorfiles['PHP' . $filename] = 'PHP parser errors :' . $php_errormsg;
  873.             }
  874.  
  875.             $this->debug = true;
  876.         }
  877.  
  878.         return $strings;
  879.     }
