  359.             {

  361.                 break;

  370.  
  371.                 if ($result !== false)
  372.                 {
  373.                     return $result;
  374.                 }

  376.         }
  377.  
  378.         $layoutTemplate = $this->getLayoutTemplate();
  379.  
  380.         // Parse the path
  381.         $templateParts = $this->_parseTemplatePath($path);
  382.  
  383.         // Get the paths
  384.         $componentPaths = FOFPlatform::getInstance()->getComponentBaseDirs($templateParts['component']);
  385.         $templatePath = FOFPlatform::getInstance()->getTemplateOverridePath($templateParts['component']);
  386.

  395.         }
