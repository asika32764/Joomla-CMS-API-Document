  33.     public function applyUpdateSite()

  37.         switch ($params->get('updatesource', 'nochange'))
  38.         {
  39.             // "Long Term Support (LTS) branch - Recommended"
  40.             case 'lts':
  41.                 $updateURL = 'http://update.joomla.org/core/list.xml';
  42.                 break;
  43.  
  44.             // "Short term support (STS) branch"
  45.             case 'sts':
  46.                 $updateURL = 'http://update.joomla.org/core/sts/list_sts.xml';
  47.                 break;
  48.  
  49.             // "Testing"
  50.             case 'testing':
  51.                 $updateURL = 'http://update.joomla.org/core/test/list_test.xml';
  52.                 break;
  53.  
  54.             // "Custom" if custom URL empty no changes
  55.             case 'custom':
  56.                 if ($params->get('customurl', '') != '')
  57.                 {
  58.                     $updateURL = $params->get('customurl', '');
  59.                 }
  60.                 else
  61.                 {
  62.                     return JError::raiseWarning(403, JText::_('COM_JOOMLAUPDATE_CONFIG_UPDATESOURCE_CUSTOM_ERROR'));
  63.                 }
  64.                 break;
  65.  
  66.             // "Do not change"
  67.             case 'nochange':
  68.             default:
  69.                 return;
  70.                 break;
  71.         }
  72.  
  73.         $db = $this->getDbo();
  74.         $query = $db->getQuery(true)
  75.             ->select($db->quoteName('us') . '.*')
  76.             ->from($db->quoteName('#__update_sites_extensions') . ' AS ' . $db->quoteName('map'))
  77.             ->join(
  78.                 'INNER', $db->quoteName('#__update_sites') . ' AS ' . $db->quoteName('us')
  79.                 . ' ON (' . 'us.update_site_id = map.update_site_id)'
  80.             )
  81.             ->where('map.extension_id = ' . $db->quote(700));
  82.         $db->setQuery($query);
  83.         $update_site = $db->loadObject();
  84.  
  85.         if ($update_site->location != $updateURL)
  86.         {
  87.             // Modify the database record
  88.             $update_site->last_check_timestamp = 0;
  89.             $update_site->location = $updateURL;
  90.             $db->updateObject('#__update_sites', $update_site, 'update_site_id');
  91.  
  92.             // Remove cached updates
  93.             $query->clear()
  94.                 ->delete($db->quoteName('#__updates'))
  95.                 ->where($db->quoteName('extension_id') . ' = ' . $db->quote('700'));
  96.             $db->setQuery($query);
  97.             $db->execute();
  98.         }
  99.     }

  245.         $basename = basename($packageURL);

  250.         $target = $tempdir . '/' . $basename;
  251.  
  252.         // Do we have a cached file?
  253.         $exists = JFile::exists($target);
  254.  
  255.         if (!$exists)
  256.         {
  257.             // Not there, let's fetch it
  258.             return $this->downloadPackage($packageURL, $target);
  259.         }
  260.         else
  261.         {
  262.             // Is it a 0-byte file? If so, re-download please.
  263.             $filesize = @filesize($target);
  264.             if (empty($filesize))
  265.             {
  266.                 return $this->downloadPackage($packageURL, $target);
  267.             }
  268.  
  269.             // Yes, it's there, skip downloading
  270.             return $basename;
  271.         }
  272.     }
