  16.  * done through the 3.x series as the issue impacts migrated sites which will

  20.  * This check returns true when the eAccelerator caching method is user, meaning

  23.  * @return  integer

  35. /**

  43. function admin_postinstall_eaccelerator_action()

  65.     if (!$ftp['enabled'] && JPath::isOwner($file) && !JPath::setPermissions($file, '0644'))
  66.     {
  67.         JError::raiseNotice('SOME_ERROR_CODE', JText::_('COM_CONFIG_ERROR_CONFIGURATION_PHP_NOTWRITABLE'));
  68.     }
  69.  
  70.     // Attempt to write the configuration file as a PHP class named JConfig.
  71.     $configuration = $config->toString('PHP', array('class' => 'JConfig', 'closingtag' => false));
  72.  
  73.     if (!JFile::write($file, $configuration))
  74.     {

  80.     // Attempt to make the file unwriteable if using FTP.
  81.     if (!$ftp['enabled'] && JPath::isOwner($file) && !JPath::setPermissions($file, '0444'))
  82.     {
  83.         JError::raiseNotice('SOME_ERROR_CODE', JText::_('COM_CONFIG_ERROR_CONFIGURATION_PHP_NOTUNWRITABLE'));
  84.     }
  85. }
