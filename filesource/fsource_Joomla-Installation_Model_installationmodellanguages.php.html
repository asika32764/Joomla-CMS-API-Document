  274.     protected function getInstalledlangs($cms_client = 'administrator')
  275.     {

  280.

  286.             $file = $path . '/' . $lang . '/' . $lang . '.xml';
  287.             $info = JInstaller::parseXMLInstallFile($file);
  288.             $row = new stdClass;
  289.             $row->language = $lang;
  290.  
  291.             if (!is_array($info))
  292.             {
  293.                 continue;
  294.             }
  295.  
  296.             foreach ($info as $key => $value)
  297.             {
  298.                 $row->$key = $value;
  299.             }
  300.  
  301.             // If current then set published
  302.             $params = JComponentHelper::getParams('com_languages');
  303.  
  304.             if ($params->get($client->name, 'en-GB') == $row->language)
  305.             {
  306.                 $row->published    = 1;
  307.             }
  308.             else
  309.             {
  310.                 $row->published = 0;
  311.             }
  312.  
  313.             $row->checked_out = 0;
  314.             $data[] = $row;
  315.         }
  316.  
  317.         usort($data, array($this, 'compareLanguages'));
  318.  
  319.         return $data;
  320.     }
