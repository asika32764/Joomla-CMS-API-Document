  112.     public function install($lids)
  113.     {
  114.         /* @var InstallationApplicationWeb $app */
  115.         $app       = JFactory::getApplication();
  116.         $installer = JInstaller::getInstance();
  117.  
  118.         // Loop through every selected language
  119.         foreach ($lids as $id)
  120.         {
  121.             // Loads the update database object that represents the language
  122.             $language = JTable::getInstance('update');
  123.             $language->load($id);
  124.  
  125.             // Get the url to the XML manifest file of the selected language
  126.             $remote_manifest = $this->getLanguageManifest($id);
  127.  
  128.             if (!$remote_manifest)
  129.             {
  130.                 // Could not find the url, the information in the update server may be corrupt
  131.                 $message = JText::sprintf('INSTL_DEFAULTLANGUAGE_COULD_NOT_INSTALL_LANGUAGE', $language->name);
  132.                 $message .= ' ' . JText::_('INSTL_DEFAULTLANGUAGE_TRY_LATER');
  133.                 $app->enqueueMessage($message);
  134.                 continue;
  135.             }
  136.  
  137.             // Based on the language XML manifest get the url of the package to download
  138.             $package_url = $this->getPackageUrl($remote_manifest);
  139.  
  140.             if (!$package_url)
  141.             {
  142.                 // Could not find the url , maybe the url is wrong in the update server, or there is not internet access
  143.                 $message = JText::sprintf('INSTL_DEFAULTLANGUAGE_COULD_NOT_INSTALL_LANGUAGE', $language->name);
  144.                 $message .= ' ' . JText::_('INSTL_DEFAULTLANGUAGE_TRY_LATER');
  145.                 $app->enqueueMessage($message);
  146.                 continue;
  147.             }
  148.  
  149.             // Download the package to the tmp folder
  150.             $package = $this->downloadPackage($package_url);
  151.  
  152.             // Install the package
  153.             if (!$installer->install($package['dir']))
  154.             {
  155.                 // There was an error installing the package
  156.                 $message = JText::sprintf('INSTL_DEFAULTLANGUAGE_COULD_NOT_INSTALL_LANGUAGE', $language->name);
  157.                 $message .= ' ' . JText::_('INSTL_DEFAULTLANGUAGE_TRY_LATER');
  158.                 $app->enqueueMessage($message);
  159.                 continue;
  160.             }
  161.  
  162.             // Cleanup the install files in tmp folder
  163.             if (!is_file($package['packagefile']))
  164.             {
  165.                 $config                 = JFactory::getConfig();
  166.                 $package['packagefile'] = $config->get('tmp_path') . '/' . $package['packagefile'];
  167.             }
  168.  
  169.             JInstallerHelper::cleanupInstall($package['packagefile'], $package['extractdir']);
  170.  
  171.             // Delete the installed language from the list

  272.      * @param   string  $cms_client  Name of the cms client

  274.      * @return  array 
  275.      *
  276.      * @since   3.1
  277.      */
  278.     protected function getInstalledlangs($cms_client = 'administrator')
  279.     {
  280.         // Get information
  281.         $path     = $this->getPath();
  282.         $client   = $this->getClient($cms_client);
  283.         $langlist = $this->getLanguageList($client->id);
  284.  
  285.         // Compute all the languages
  286.         $data = array();
  287.  
  288.         foreach ($langlist as $lang)
  289.         {
  290.             $file          = $path . '/' . $lang . '/' . $lang . '.xml';
  291.             $info          = JInstaller::parseXMLInstallFile($file);
  292.             $row           = new stdClass;
  293.             $row->language = $lang;
  294.  
  295.             if (!is_array($info))
  296.             {
  297.                 continue;
  298.             }
  299.  
  300.             foreach ($info as $key => $value)
  301.             {
  302.                 $row->$key = $value;
  303.             }
  304.  
  305.             // If current then set published
  306.             $params = JComponentHelper::getParams('com_languages');
  307.  
  308.             if ($params->get($client->name, 'en-GB') == $row->language)
  309.             {
  310.                 $row->published = 1;
  311.             }
  312.             else
  313.             {
  314.                 $row->published = 0;
  315.             }
  316.  
  317.             $row->checked_out = 0;
  318.             $data[]           = $row;
  319.         }
  320.  
  321.         usort($data, array($this, 'compareLanguages'));
  322.  
  323.         return $data;
  324.     }

  655.     /**

  657.      *
  658.      * This function checks other existing language with the same code, if they exist provides a unique SEF name.
  659.      * For instance: en-GB, en-US and en-AU will share the same SEF code by default: www.mywebsite.com/en/
  660.      * To avoid this conflict, this function creates an specific SEF in case of existing conflict:
  661.      * For example: www.mywebsite.com/en-au/
  662.      *
  663.      * @param   stdClass    $itemLanguage   Language Object
  664.      * @param   stdClass[]  $siteLanguages  All Language Objects
  665.      *
  666.      * @return  string 
  667.      *
  668.      * @since   3.2
  669.      */
  670.     public function getSefString($itemLanguage, $siteLanguages)
  671.     {
  672.         $langs = explode('-', $itemLanguage->language);
  673.         $prefixToFind = $langs[0];
  674.  
  675.         $numberPrefixesFound = 0;
  676.         foreach ($siteLanguages as $siteLang)
  677.         {
  678.             $langs = explode('-', $siteLang->language);
  679.             $lang  = $langs[0];
  680.  
  681.             if ($lang == $prefixToFind)
  682.             {
  683.                 ++$numberPrefixesFound;
  684.             }
  685.         }
  686.  
  687.         if ($numberPrefixesFound == 1)
  688.         {
  689.             return $prefixToFind;
  690.         }
  691.         return strtolower($itemLanguage->language);
  692.     }
