  104.

  108.

  110.                     if ($ftp->isConnected())

  120.  
  121.             default:
  122.                 break;

  124.  
  125.         if ($return)

  131.  
  132.             // Force re-creation of the data saved within JClientHelper::getCredentials()

  206.     public static function setCredentialsFromRequest($client)
  207.     {

  225.                 }
