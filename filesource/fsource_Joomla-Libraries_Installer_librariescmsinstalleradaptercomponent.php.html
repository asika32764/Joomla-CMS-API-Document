  442.         ob_start();

  454.         }
  455.

  459.

  488.  
  489.         $eid = $row->extension_id;

  515.             // @todo remove code: return false;

  552.  
  553.         if ($msg != '')

  558.         return $row->extension_id;
  559.     }

  618.  
  619.         // Hunt for the original XML file

  648.         else

  650.             $this->oldAdminFiles = null;

  653.  
  654.         /**
  655.          * ---------------------------------------------------------------------------------------------
  656.          * Basic Checks Section
  657.          * ---------------------------------------------------------------------------------------------
  658.          */
  659.  
  660.         // Make sure that we have an admin element

  667.  
  668.         /**
  669.          * ---------------------------------------------------------------------------------------------
  670.          * Installer Trigger Loading
  671.          * ---------------------------------------------------------------------------------------------
  672.          */
  673.  
  674.         // If there is an manifest class file, lets load it; we'll copy it later (don't have dest yet)
  675.         $manifestScript = (string) $this->manifest->scriptfile;
  676.  
  677.         if ($manifestScript)
  678.         {
  679.             $manifestScriptFile = $this->parent->getPath('source') . '/' . $manifestScript;
  680.  
  681.             if (is_file($manifestScriptFile))
  682.             {
  683.                 // Load the file
  684.                 include_once $manifestScriptFile;
  685.             }
  686.  
  687.             // Set the class name
  688.             $classname = $element . 'InstallerScript';
  689.  
  690.             if (class_exists($classname))
  691.             {
  692.                 // Create a new instance
  693.                 $this->parent->manifestClass = new $classname($this);
  694.  
  695.                 // And set this so we can copy it later
  696.                 $this->set('manifest_script', $manifestScript);
  697.             }
  698.         }
  699.  
  700.         // Run preflight if possible (since we know we're not an update)
  701.         ob_start();
  702.         ob_implicit_flush(false);
  703.  
  704.         if ($this->parent->manifestClass && method_exists($this->parent->manifestClass, 'preflight'))
  705.         {
  706.             if ($this->parent->manifestClass->preflight('update', $this) === false)

  711.                 return false;
  712.             }
  713.         }
  714.  
  715.         // Create msg object; first use here
  716.         $msg = ob_get_contents();
  717.         ob_end_clean();
  718.  
  719.         /**
  720.          * ---------------------------------------------------------------------------------------------
  721.          * Filesystem Processing Section
  722.          * ---------------------------------------------------------------------------------------------
  723.          */
  724.  
  725.         // If the component directory does not exist, let's create it

  727.

  729.         {
  730.             if (!$created = JFolder::create($this->parent->getPath('extension_site')))
  731.             {
  732.                 JLog::add(
  733.                     JText::sprintf('JLIB_INSTALLER_ERROR_COMP_UPDATE_FAILED_TO_CREATE_DIRECTORY_SITE', $this->parent->getPath('extension_site')),
  734.                     JLog::WARNING, 'jerror'
  735.                 );
  736.  
  737.                 return false;
  738.             }
  739.         }
  740.  
  741.         /*
  742.          * Since we created the component directory and will want to remove it if we have to roll back
  743.          * the installation, lets add it to the installation step stack
  744.          */
  745.         if ($created)
  746.         {
  747.             $this->parent->pushStep(array('type' => 'folder', 'path' => $this->parent->getPath('extension_site')));
  748.         }
  749.  
  750.         // If the component admin directory does not exist, let's create it
  751.         $created = false;
  752.  
  753.         if (!file_exists($this->parent->getPath('extension_administrator')))
  754.         {
  755.             if (!$created = JFolder::create($this->parent->getPath('extension_administrator')))
  756.             {
  757.                 JLog::add(
  758.                     JText::sprintf('JLIB_INSTALLER_ERROR_COMP_UPDATE_FAILED_TO_CREATE_DIRECTORY_ADMIN', $this->parent->getPath('extension_administrator')),
  759.                     JLog::WARNING, 'jerror'
  760.                 );
  761.  
  762.                 // Install failed, rollback any changes
  763.                 $this->parent->abort();
  764.  
  765.                 return false;
  766.             }
  767.         }
  768.  
  769.         /*
  770.          * Since we created the component admin directory and we will want to remove it if we have to roll
  771.          * back the installation, let's add it to the installation step stack
  772.          */
  773.         if ($created)
  774.         {
  775.             $this->parent->pushStep(array('type' => 'folder', 'path' => $this->parent->getPath('extension_administrator')));
  776.         }
  777.  
  778.         // Find files to copy
  779.         if ($this->manifest->files)
  780.         {
  781.             if ($this->parent->parseFiles($this->manifest->files, 0, $this->oldFiles) === false)
  782.             {
  783.                 // Install failed, rollback any changes
  784.                 $this->parent->abort();
  785.  
  786.                 return false;
  787.             }
  788.         }
  789.  
  790.         if ($this->manifest->administration->files)
  791.         {
  792.             if ($this->parent->parseFiles($this->manifest->administration->files, 1, $this->oldAdminFiles) === false)

  794.                 // Install failed, rollback any changes
  795.                 $this->parent->abort();
  796.  
  797.                 return false;
  798.             }
  799.         }
  800.  
  801.         // Parse optional tags
  802.         $this->parent->parseMedia($this->manifest->media);
  803.         $this->parent->parseLanguages($this->manifest->languages);
  804.         $this->parent->parseLanguages($this->manifest->administration->languages, 1);
  805.  
  806.         // If there is a manifest script, let's copy it.
  807.         if ($this->get('manifest_script'))
  808.         {
  809.             $path['src'] = $this->parent->getPath('source') . '/' . $this->get('manifest_script');
  810.             $path['dest'] = $this->parent->getPath('extension_administrator') . '/' . $this->get('manifest_script');
  811.  
  812.             if (!file_exists($path['dest']) || $this->parent->isOverwrite())
  813.             {
  814.                 if (!$this->parent->copyFiles(array($path)))
  815.                 {
  816.                     // Install failed, rollback changes
  817.                     $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_COMP_UPDATE_MANIFEST'));
  818.  
  819.                     return false;
  820.                 }
  821.             }
  822.         }
  823.  
  824.         /**
  825.          * ---------------------------------------------------------------------------------------------
  826.          * Database Processing Section
  827.          * ---------------------------------------------------------------------------------------------
  828.          */
  829.  
  830.         // Let's run the update queries for the component
  831.         $row = JTable::getInstance('extension');
  832.         $eid = $row->find(array('element' => strtolower($this->get('element')), 'type' => 'component'));
  833.  
  834.         if ($this->manifest->update)
  835.         {
  836.             $result = $this->parent->parseSchemaUpdates($this->manifest->update->schemas, $eid);
  837.  
  838.             if ($result === false)
  839.             {
  840.                 // Install failed, rollback changes
  841.                 $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_COMP_UPDATE_SQL_ERROR', $db->stderr(true)));
  842.  
  843.                 return false;
  844.             }
  845.         }
  846.  
  847.         // Time to build the admin menus
  848.         if (!$this->_buildAdminMenus($eid))
  849.         {
  850.             JLog::add(JText::_('JLIB_INSTALLER_ABORT_COMP_BUILDADMINMENUS_FAILED'), JLog::WARNING, 'jerror');
  851.  
  852.             // $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_COMP_INSTALL_ROLLBACK', $db->stderr(true)));

  854.         }
  855.  
  856.         /**
  857.          * ---------------------------------------------------------------------------------------------
  858.          * Custom Installation Script Section
  859.          * ---------------------------------------------------------------------------------------------
  860.          */
  861.  
  862.         /*
  863.          * If we have an install script, let's include it, execute the custom
  864.          * update method, and append the return value from the custom update
  865.          * method to the installation message.
  866.          */
  867.         ob_start();
  868.         ob_implicit_flush(false);
  869.  
  870.         if ($this->parent->manifestClass && method_exists($this->parent->manifestClass, 'update'))
  871.         {
  872.             if ($this->parent->manifestClass->update($this) === false)
  873.             {
  874.                 // Install failed, rollback changes
  875.                 $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_COMP_INSTALL_CUSTOM_INSTALL_FAILURE'));
  876.  
  877.                 return false;
  878.             }
  879.         }
  880.  
  881.         // Append messages
  882.         $msg .= ob_get_contents();
  883.         ob_end_clean();
  884.  
  885.         /**
  886.          * ---------------------------------------------------------------------------------------------
  887.          * Finalization and Cleanup Section
  888.          * ---------------------------------------------------------------------------------------------
  889.          */
  890.  
  891.         // Clobber any possible pending updates
  892.         $update = JTable::getInstance('update');
  893.         $uid = $update->find(array('element' => $this->get('element'), 'type' => 'component', 'client_id' => 1, 'folder' => ''));
  894.  
  895.         if ($uid)
  896.         {
  897.             $update->delete($uid);
  898.         }
  899.  
  900.         // Update an entry to the extension table
  901.         if ($eid)
  902.         {
  903.             $row->load($eid);
  904.         }
  905.         else
  906.         {
  907.             // Set the defaults
  908.             // There is no folder for components
  909.             $row->folder = '';
  910.             $row->enabled = 1;
  911.             $row->protected = 0;
  912.             $row->access = 1;
  913.             $row->client_id = 1;
  914.             $row->params = $this->parent->getParams();
  915.         }
  916.  
  917.         $row->name = $this->get('name');
  918.         $row->type = 'component';
  919.         $row->element = $this->get('element');
  920.         $row->manifest_cache = $this->parent->generateManifestCache();
  921.  
  922.         if (!$row->store())
  923.         {
  924.             // Install failed, roll back changes
  925.             $this->parent->abort(JText::sprintf('JLIB_INSTALLER_ABORT_COMP_UPDATE_ROLLBACK', $db->stderr(true)));
  926.  
  927.             return false;
  928.         }
  929.  
  930.         // We will copy the manifest file to its appropriate place.
  931.         if (!$this->parent->copyManifest())
  932.         {
  933.             // Install failed, rollback changes
  934.             $this->parent->abort(JText::_('JLIB_INSTALLER_ABORT_COMP_UPDATE_COPY_SETUP'));
  935.  
  936.             return false;
  937.         }
  938.  
  939.         // And now we run the postflight
  940.         ob_start();
  941.         ob_implicit_flush(false);
  942.  
  943.         if ($this->parent->manifestClass && method_exists($this->parent->manifestClass, 'postflight'))
  944.         {
  945.             $this->parent->manifestClass->postflight('update', $this);
  946.         }
  947.  
  948.         // Append messages
  949.         $msg .= ob_get_contents();
  950.         ob_end_clean();
  951.  
  952.         if ($msg != '')
  953.         {
  954.             $this->parent->set('extension_message', $msg);
  955.         }
  956.  
  957.         return $row->extension_id;
  958.     }
  959.
