  30.     public static function getList(&$params)

  41.         $cache = JFactory::getCache('mod_menu', '');
  42.         if (!($items = $cache->get($key)))
  43.         {

  45.             $start   = (int) $params->get('startLevel');
  46.             $end     = (int) $params->get('endLevel');
  47.             $showAll = $params->get('showAllChildren');
  48.             $items   = $menu->getItems('menutype', $params->get('menutype'));
  49.  
  50.             $lastitem = 0;
  51.  
  52.             if ($items)

  54.                 foreach ($items as $i => $item)
  55.                 {
  56.                     if (($start && $start > $item->level)
  57.                         || ($end && $item->level > $end)
  58.                         || (!$showAll && $item->level > 1 && !in_array($item->parent_id, $path))
  59.                         || ($start > 1 && !in_array($item->tree[$start - 2], $path)))
  60.                     {
  61.                         unset($items[$i]);
  62.                         continue;
  63.                     }
  64.  
  65.                     $item->deeper     = false;
  66.                     $item->shallower  = false;
  67.                     $item->level_diff = 0;
  68.  
  69.                     if (isset($items[$lastitem]))
  70.                     {
  71.                         $items[$lastitem]->deeper     = ($item->level > $items[$lastitem]->level);
  72.                         $items[$lastitem]->shallower  = ($item->level < $items[$lastitem]->level);
  73.                         $items[$lastitem]->level_diff = ($items[$lastitem]->level - $item->level);
  74.                     }
  75.  
  76.                     $item->parent = (boolean) $menu->getItems('parent_id', (int) $item->id, true);
  77.  
  78.                     $lastitem     = $i;
  79.                     $item->active = false;
  80.                     $item->flink  = $item->link;
  81.  
  82.                     // Reverted back for CMS version 2.5.6
  83.                     switch ($item->type)
  84.                     {
  85.                         case 'separator':
  86.                         case 'heading':
  87.                             // No further action needed.
  88.                             continue;
  89.  
  90.                         case 'url':
  91.                             if ((strpos($item->link, 'index.php?') === 0) && (strpos($item->link, 'Itemid=') === false))
  92.                             {
  93.                                 // If this is an internal Joomla link, ensure the Itemid is set.
  94.                                 $item->flink = $item->link . '&Itemid=' . $item->id;
  95.                             }
  96.                             break;
  97.  
  98.                         case 'alias':
  99.                             // If this is an alias use the item id stored in the parameters to make the link.
  100.                             $item->flink = 'index.php?Itemid=' . $item->params->get('aliasoptions');
  101.                             break;
  102.  
  103.                         default:
  104.                             $router = $app::getRouter();
  105.                             if ($router->getMode() == JROUTER_MODE_SEF)
  106.                             {
  107.                                 $item->flink = 'index.php?Itemid=' . $item->id;
  108.                             }
  109.                             else
  110.                             {
  111.                                 $item->flink .= '&Itemid=' . $item->id;
  112.                             }
  113.                             break;
  114.                     }
  115.  
  116.                     if (strcasecmp(substr($item->flink, 0, 4), 'http') && (strpos($item->flink, 'index.php?') !== false))
  117.                     {
  118.                         $item->flink = JRoute::_($item->flink, true, $item->params->get('secure'));
  119.                     }
  120.                     else
  121.                     {
  122.                         $item->flink = JRoute::_($item->flink);
  123.                     }
  124.  
  125.                     // We prevent the double encoding because for some reason the $item is shared for menu modules and we get double encoding
  126.                     // when the cause of that is found the argument should be removed
  127.                     $item->title        = htmlspecialchars($item->title, ENT_COMPAT, 'UTF-8', false);
  128.                     $item->anchor_css   = htmlspecialchars($item->params->get('menu-anchor_css', ''), ENT_COMPAT, 'UTF-8', false);
  129.                     $item->anchor_title = htmlspecialchars($item->params->get('menu-anchor_title', ''), ENT_COMPAT, 'UTF-8', false);
  130.                     $item->menu_image   = $item->params->get('menu_image', '') ? htmlspecialchars($item->params->get('menu_image', ''), ENT_COMPAT, 'UTF-8', false) : '';
  131.                 }
  132.  
  133.                 if (isset($items[$lastitem]))
  134.                 {
  135.                     $items[$lastitem]->deeper     = (($start?$start:1) > $items[$lastitem]->level);
  136.                     $items[$lastitem]->shallower  = (($start?$start:1) < $items[$lastitem]->level);
  137.                     $items[$lastitem]->level_diff = ($items[$lastitem]->level - ($start?$start:1));
  138.                 }
  139.             }
  140.  
  141.             $cache->store($items, $key);
  142.         }
  143.         return $items;
  144.     }
