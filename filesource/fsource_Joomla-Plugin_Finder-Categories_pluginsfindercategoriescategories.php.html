  243.     protected function index(FinderIndexerResult $item, $format = 'html')

  257.             include_once $path;

  259.  
  260.         $extension = ucfirst(substr($item->extension, 4));
  261.  
  262.         // Initialize the item parameters.
  263.         $registry = new JRegistry;
  264.         $registry->loadString($item->params);
  265.         $item->params = $registry;
  266.  
  267.         $registry = new JRegistry;

  271.         /* Add the meta-data processing instructions based on the categories

  302.         // Get the menu title if it exists.

  313.

  317.         // Add the language taxonomy data.
  318.         $item->addTaxonomy('Language', $item->language);

  320.         // Get content extras.

  342.     /**
  343.      * Method to get the SQL query used to retrieve the list of content items.
  344.      *
  345.      * @param   mixed  $query  A JDatabaseQuery object or null.
  346.      *
  347.      * @return  JDatabaseQuery  A database object.
  348.      *
  349.      * @since   2.5
  350.      */
  351.     protected function getListQuery($query = null)
  352.     {
  353.         $db = JFactory::getDbo();
  354.         // Check if we can use the supplied SQL query.
  355.         $query = $query instanceof JDatabaseQuery ? $query : $db->getQuery(true)
  356.             ->select('a.id, a.title, a.alias, a.description AS summary, a.extension')
  357.             ->select('a.created_user_id AS created_by, a.modified_time AS modified, a.modified_user_id AS modified_by')
  358.             ->select('a.metakey, a.metadesc, a.metadata, a.language, a.lft, a.parent_id, a.level')
  359.             ->select('a.created_time AS start_date, a.published AS state, a.access, a.params');
  360.  
  361.         // Handle the alias CASE WHEN portion of the query
  362.         $case_when_item_alias = ' CASE WHEN ';
  363.         $case_when_item_alias .= $query->charLength('a.alias', '!=', '0');
  364.         $case_when_item_alias .= ' THEN ';
  365.         $a_id = $query->castAsChar('a.id');
  366.         $case_when_item_alias .= $query->concatenate(array($a_id, 'a.alias'), ':');
  367.         $case_when_item_alias .= ' ELSE ';
  368.         $case_when_item_alias .= $a_id.' END as slug';
  369.         $query->select($case_when_item_alias)
  370.             ->from('#__categories AS a')
  371.             ->where($db->quoteName('a.id') . ' > 1');
  372.  
  373.         return $query;
  374.     }

  376.     /**
  377.      * Method to get a SQL query to load the published and access states for
  378.      * an article and category.
  379.      *
  380.      * @return  JDatabaseQuery  A database object.
  381.      *
  382.      * @since   2.5

  384.     protected function getStateQuery()
  385.     {
  386.         $query = $this->db->getQuery(true)
  387.             ->select($this->db->quoteName('a.id'))
  388.             ->select('a.' . $this->state_field . ' AS state, c.published AS cat_state')
  389.             ->select('a.access, c.access AS cat_access')
  390.             ->from($this->db->quoteName('#__categories') . ' AS a')
  391.             ->join('LEFT', '#__categories AS c ON c.id = a.parent_id');
  392.  
  393.         return $query;
  394.     }
