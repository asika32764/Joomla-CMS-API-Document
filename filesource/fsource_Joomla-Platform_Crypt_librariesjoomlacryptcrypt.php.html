  109.     /**
  110.      * Generate random bytes.
  111.      *
  112.      * @param   integer  $length  Length of the random data to generate
  113.      *
  114.      * @return  string  Random binary data
  115.      *
  116.      * @since  12.1
  117.      */
  118.     public static function genRandomBytes($length = 16)
  119.     {
  120.         $sslStr = '';
  121.  
  122.         /*
  123.          * if a secure randomness generator exists and we don't
  124.          * have a buggy PHP version use it.
  125.          */
  126.         if (function_exists('openssl_random_pseudo_bytes')
  127.             && (version_compare(PHP_VERSION, '5.3.4') >= 0 || IS_WIN))
  128.         {
  129.             $sslStr = openssl_random_pseudo_bytes($length, $strong);
  130.  
  131.             if ($strong)
  132.             {
  133.                 return $sslStr;
  134.             }
  135.         }
  136.  
  137.         /*
  138.          * Collect any entropy available in the system along with a number
  139.          * of time measurements of operating system randomness.
  140.          */
  141.         $bitsPerRound = 2;
  142.         $maxTimeMicro = 400;
  143.         $shaHashLength = 20;
  144.         $randomStr = '';
  145.         $total = $length;
  146.  
  147.         // Check if we can use /dev/urandom.
  148.         $urandom = false;
  149.         $handle = null;
  150.  
  151.         // This is PHP 5.3.3 and up
  152.         if (function_exists('stream_set_read_buffer') && @is_readable('/dev/urandom'))
  153.         {
  154.             $handle = @fopen('/dev/urandom', 'rb');
  155.  
  156.             if ($handle)
  157.             {
  158.                 $urandom = true;
  159.             }
  160.         }
  161.  
  162.         while ($length > strlen($randomStr))
  163.         {
  164.             $bytes = ($total > $shaHashLength)? $shaHashLength : $total;
  165.             $total -= $bytes;
  166.  
  167.             /*
  168.              * Collect any entropy available from the PHP system and filesystem.
  169.              * If we have ssl data that isn't strong, we use it once.
  170.              */
  171.             $entropy = rand() . uniqid(mt_rand(), true) . $sslStr;
  172.             $entropy .= implode('', @fstat(fopen(__FILE__, 'r')));
  173.             $entropy .= memory_get_usage();
  174.             $sslStr = '';
  175.  
  176.             if ($urandom)
  177.             {
  178.                 stream_set_read_buffer($handle, 0);
  179.                 $entropy .= @fread($handle, $bytes);
  180.             }
  181.             else
  182.             {
  183.                 /*
  184.                  * There is no external source of entropy so we repeat calls
  185.                  * to mt_rand until we are assured there's real randomness in
  186.                  * the result.
  187.                  *
  188.                  * Measure the time that the operations will take on average.
  189.                  */
  190.                 $samples = 3;
  191.                 $duration = 0;
  192.  
  193.                 for ($pass = 0; $pass < $samples; ++$pass)
  194.                 {
  195.                     $microStart = microtime(true) * 1000000;
  196.                     $hash = sha1(mt_rand(), true);
  197.  
  198.                     for ($count = 0; $count < 50; ++$count)
  199.                     {
  200.                         $hash = sha1($hash, true);
  201.                     }
  202.  
  203.                     $microEnd = microtime(true) * 1000000;
  204.                     $entropy .= $microStart . $microEnd;
  205.  
  206.                     if ($microStart >= $microEnd)
  207.                     {
  208.                         $microEnd += 1000000;
  209.                     }
  210.  
  211.                     $duration += $microEnd - $microStart;
  212.                 }
  213.  
  214.                 $duration = $duration / $samples;
  215.  
  216.                 /*
  217.                  * Based on the average time, determine the total rounds so that
  218.                  * the total running time is bounded to a reasonable number.
  219.                  */
  220.                 $rounds = (int) (($maxTimeMicro / $duration) * 50);
  221.  
  222.                 /*
  223.                  * Take additional measurements. On average we can expect
  224.                  * at least $bitsPerRound bits of entropy from each measurement.
  225.                  */
  226.                 $iter = $bytes * (int) ceil(8 / $bitsPerRound);
  227.  
  228.                 for ($pass = 0; $pass < $iter; ++$pass)
  229.                 {
  230.                     $microStart = microtime(true);
  231.                     $hash = sha1(mt_rand(), true);
  232.  
  233.                     for ($count = 0; $count < $rounds; ++$count)
  234.                     {
  235.                         $hash = sha1($hash, true);
  236.                     }
  237.  
  238.                     $entropy .= $microStart . microtime(true);
  239.                 }
  240.             }
  241.  
  242.             $randomStr .= sha1($entropy, true);
  243.         }
  244.  
  245.         if ($urandom)
  246.         {
  247.             @fclose($handle);
  248.         }
  249.  
  250.         return substr($randomStr, 0, $length);
  251.     }
