  118.     /**

  125.     public function getItem($pk = null)
  126.     {
  127.         $pk = (!empty($pk)) ? $pk : (int) $this->getState('plugin.id');
  128.  
  129.         if (!isset($this->_cache[$pk]))
  130.         {
  131.             $false    = false;
  132.  
  133.             // Get a row instance.
  134.             $table = $this->getTable();
  135.  
  136.             // Attempt to load the row.
  137.             $return = $table->load($pk);
  138.  
  139.             // Check for a table object error.
  140.             if ($return === false && $table->getError())
  141.             {
  142.                 $this->setError($table->getError());
  143.                 return $false;
  144.             }
  145.  
  146.             // Convert to the JObject before adding other data.
  147.             $properties = $table->getProperties(1);
  148.             $this->_cache[$pk] = JArrayHelper::toObject($properties, 'JObject');
  149.  
  150.             // Convert the params field to an array.
  151.             $registry = new JRegistry;
  152.             $registry->loadString($table->params);
  153.             $this->_cache[$pk]->params = $registry->toArray();
  154.  
  155.             // Get the plugin XML.
  156.             $path = JPath::clean(JPATH_PLUGINS.'/'.$table->folder.'/'.$table->element.'/'.$table->element.'.xml');
  157.  
  158.             if (file_exists($path))
  159.             {
  160.                 $this->_cache[$pk]->xml = simplexml_load_file($path);
  161.             } else {
  162.                 $this->_cache[$pk]->xml = null;
  163.             }
  164.         }
  165.  
  166.         return $this->_cache[$pk];
  167.     }

  224.  
  225.         foreach ($elements as $elementa)

  230.             ||    $lang->load('plg_'.$folder.'_'.$elementa.'.sys', JPATH_PLUGINS.'/'.$folder.'/'.$elementa, $lang->getDefault(), false, false);

  250.  
  251.         if (file_exists($formFile))

  253.             // Get the plugin form.

  255.             {
  256.                 throw new Exception(JText::_('JERROR_LOADFILE_FAILED'));

  260.         // Attempt to load the xml file.
  261.         if (!$xml = simplexml_load_file($formFile))

  266.         // Get the help data from the XML file if present.

  273.             $this->helpKey = $helpKey ? $helpKey : $this->helpKey;

  276.

  278.         parent::preprocessForm($form, $data, $group);
