  120.             ->where('c.access IN (' . $groups . ')');
  121.         // Filter by category.
  122.         if ($categoryId = $this->getState('category.id'))
  123.         {
  124.             $query->where('a.catid = ' . (int) $categoryId);
  125.         }
  126.         //sqlsrv change... aliased c.published to cat_published
  127.         // Join to check for category published state in parent categories up the tree
  128.         $query->select('c.published as cat_published, CASE WHEN badcats.id is null THEN c.published ELSE 0 END AS parents_published');
  129.         $subquery = 'SELECT cat.id as id FROM #__categories AS cat JOIN #__categories AS parent ';
  130.         $subquery .= 'ON cat.lft BETWEEN parent.lft AND parent.rgt ';
  131.         $subquery .= 'WHERE parent.extension = ' . $db->quote('com_contact');
  132.         // Find any up-path categories that are not published
  133.         // If all categories are published, badcats.id will be null, and we just use the contact state
  134.         $subquery .= ' AND parent.published != 1 GROUP BY cat.id ';
  135.         // Select state to unpublished if up-path category is unpublished
  136.         $publishedWhere = 'CASE WHEN badcats.id is null THEN a.published ELSE 0 END';
  137.         $query->join('LEFT OUTER', '(' . $subquery . ') AS badcats ON badcats.id = c.id');
  138.  
  139.         // Filter by state
  140.         $state = $this->getState('filter.published');
  141.         if (is_numeric($state))
