  171.      *
  172.      * @return  array 
  173.      *
  174.      * @since   12.1
  175.      */
  176.     protected function getAlterTableSQL(SimpleXMLElement $structure)
  177.     {
  178.         $table = $this->getRealTableName($structure['name']);
  179.         $oldFields = $this->db->getTableColumns($table);
  180.         $oldKeys = $this->db->getTableKeys($table);
  181.         $oldSequence = $this->db->getTableSequences($table);
  182.         $alters = array();
  183.  
  184.         // Get the fields and keys from the XML that we are aiming for.
  185.         $newFields = $structure->xpath('field');
  186.         $newKeys = $structure->xpath('key');
  187.         $newSequence = $structure->xpath('sequence');
  188.  
  189.         /* Sequence section */
  190.         $oldSeq = $this->getSeqLookup($oldSequence);
  191.         $newSequenceLook = $this->getSeqLookup($newSequence);
  192.  
  193.         foreach ($newSequenceLook as $kSeqName => $vSeq)
  194.         {
  195.             if (isset($oldSeq[$kSeqName]))
  196.             {
  197.                 // The field exists, check it's the same.
  198.                 $column = $oldSeq[$kSeqName][0];
  199.  
  200.                 /* For older database version that doesn't support these fields use default values */
  201.                 if (version_compare($this->db->getVersion(), '9.1.0') < 0)
  202.                 {
  203.                     $column->Min_Value = '1';
  204.                     $column->Max_Value = '9223372036854775807';
  205.                     $column->Increment = '1';
  206.                     $column->Cycle_option = 'NO';
  207.                     $column->Start_Value = '1';
  208.                 }
  209.  
  210.                 // Test whether there is a change.
  211.                 $change = ((string) $vSeq[0]['Type'] != $column->Type) || ((string) $vSeq[0]['Start_Value'] != $column->Start_Value)
  212.                     || ((string) $vSeq[0]['Min_Value'] != $column->Min_Value) || ((string) $vSeq[0]['Max_Value'] != $column->Max_Value)
  213.                     || ((string) $vSeq[0]['Increment'] != $column->Increment) || ((string) $vSeq[0]['Cycle_option'] != $column->Cycle_option)
  214.                     || ((string) $vSeq[0]['Table'] != $column->Table) || ((string) $vSeq[0]['Column'] != $column->Column)
  215.                     || ((string) $vSeq[0]['Schema'] != $column->Schema) || ((string) $vSeq[0]['Name'] != $column->Name);
  216.  
  217.                 if ($change)
  218.                 {
  219.                     $alters[] = $this->getChangeSequenceSQL($kSeqName, $vSeq);
  220.                 }
  221.  
  222.                 // Unset this field so that what we have left are fields that need to be removed.
  223.                 unset($oldSeq[$kSeqName]);
  224.             }
  225.             else
  226.             {
  227.                 // The sequence is new
  228.                 $alters[] = $this->getAddSequenceSQL($newSequenceLook[$kSeqName][0]);
  229.             }
  230.         }
  231.  
  232.         // Any sequences left are orphans
  233.         foreach ($oldSeq as $name => $column)
  234.         {
  235.             // Delete the sequence.
  236.             $alters[] = $this->getDropSequenceSQL($name);
  237.         }
  238.  
  239.         /* Field section */
  240.         // Loop through each field in the new structure.
  241.         foreach ($newFields as $field)
  242.         {
  243.             $fName = (string) $field['Field'];
  244.  
  245.             if (isset($oldFields[$fName]))
  246.             {
  247.                 // The field exists, check it's the same.
  248.                 $column = $oldFields[$fName];
  249.  
  250.                 // Test whether there is a change.
  251.                 $change = ((string) $field['Type'] != $column->Type) || ((string) $field['Null'] != $column->Null)
  252.                     || ((string) $field['Default'] != $column->Default);
  253.  
  254.                 if ($change)
  255.                 {
  256.                     $alters[] = $this->getChangeColumnSQL($table, $field);
  257.                 }
  258.  
  259.                 // Unset this field so that what we have left are fields that need to be removed.
  260.                 unset($oldFields[$fName]);
  261.             }
  262.             else
  263.             {
  264.                 // The field is new.
  265.                 $alters[] = $this->getAddColumnSQL($table, $field);
  266.             }
  267.         }
  268.  
  269.         // Any columns left are orphans
  270.         foreach ($oldFields as $name => $column)
  271.         {
  272.             // Delete the column.
  273.             $alters[] = $this->getDropColumnSQL($table, $name);
  274.         }
  275.  
  276.         /* Index section */
  277.         // Get the lookups for the old and new keys
  278.         $oldLookup = $this->getIdxLookup($oldKeys);
  279.         $newLookup = $this->getIdxLookup($newKeys);
  280.  
  281.         // Loop through each key in the new structure.
  282.         foreach ($newLookup as $name => $keys)
  283.         {
  284.             // Check if there are keys on this field in the existing table.
  285.             if (isset($oldLookup[$name]))
  286.             {
  287.                 $same = true;
  288.                 $newCount = count($newLookup[$name]);
  289.                 $oldCount = count($oldLookup[$name]);
  290.  
  291.                 // There is a key on this field in the old and new tables. Are they the same?
  292.                 if ($newCount == $oldCount)
  293.                 {
  294.                     for ($i = 0; $i < $newCount; $i++)
  295.                     {
  296.                         // Check only query field -> different query means different index
  297.                         $same = ((string) $newLookup[$name][$i]['Query'] == $oldLookup[$name][$i]->Query);
  298.  
  299.                         if (!$same)
  300.                         {
  301.                             // Break out of the loop. No need to check further.
  302.                             break;
  303.                         }
  304.                     }
  305.                 }
  306.                 else
  307.                 {
  308.                     // Count is different, just drop and add.
  309.                     $same = false;
  310.                 }
  311.  
  312.                 if (!$same)
  313.                 {
  314.                     $alters[] = $this->getDropIndexSQL($name);
  315.                     $alters[]  = (string) $newLookup[$name][0]['Query'];
  316.                 }
  317.  
  318.                 // Unset this field so that what we have left are fields that need to be removed.
  319.                 unset($oldLookup[$name]);
  320.             }
  321.             else
  322.             {
  323.                 // This is a new key.
  324.                 $alters[] = (string) $newLookup[$name][0]['Query'];
  325.             }
  326.         }
  327.  
  328.         // Any keys left are orphans.
  329.         foreach ($oldLookup as $name => $keys)
  330.         {
  331.             if ($oldLookup[$name][0]->is_primary == 'TRUE')
  332.             {
  333.                 $alters[] = $this->getDropPrimaryKeySQL($table, $oldLookup[$name][0]->Index);
  334.             }
  335.             else
  336.             {
  337.                 $alters[] = $this->getDropIndexSQL($name);
  338.             }
  339.         }
  340.  
  341.         return $alters;
  342.     }

  432.      * @param   string            $table  The name of the database table to alter.
  433.      * @param   SimpleXMLElement  $field  The XML field definition.

  436.      *
  437.      * @since   12.1

  439.     protected function getAlterColumnSQL($table, $field)
  440.     {
  441.         // TODO Incorporate into parent class and use $this.
  442.         $blobs = array('text', 'smalltext', 'mediumtext', 'largetext');
  443.  
  444.         $fName = (string) $field['Field'];
  445.         $fType = (string) $field['Type'];
  446.         $fNull = (string) $field['Null'];

  448.                         preg_match('/^[0-9]$/', $field['Default']) ? $field['Default'] : $this->db->quote((string) $field['Default'])

  450.
