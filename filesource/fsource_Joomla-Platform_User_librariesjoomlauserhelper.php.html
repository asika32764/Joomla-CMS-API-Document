  298.     /**
  299.      * Formats a password using the current encryption.
  300.      *
  301.      * @param   string   $plaintext     The plaintext password to encrypt.
  302.      * @param   string   $salt          The salt to use to encrypt the password. []
  303.      *                                   If not present, a new salt will be
  304.      *                                   generated.
  305.      * @param   string   $encryption    The kind of password encryption to use.
  306.      *                                   Defaults to md5-hex.
  307.      * @param   boolean  $show_encrypt  Some password systems prepend the kind of
  308.      *                                   encryption to the crypted password ({SHA},
  309.      *                                   etc). Defaults to false.
  310.      *
  311.      * @return  string  The encrypted password.
  312.      *
  313.      * @since   11.1
  314.      */
  315.     public static function getCryptedPassword($plaintext, $salt = '', $encryption = 'md5-hex', $show_encrypt = false)
  316.     {
  317.         // Get the salt to use.
  318.         $salt = self::getSalt($encryption, $salt, $plaintext);
  319.  
  320.         // Encrypt the password.
  321.         switch ($encryption)
  322.         {
  323.             case 'plain':
  324.                 return $plaintext;
  325.  
  326.             case 'sha':
  327.                 $encrypted = base64_encode(mhash(MHASH_SHA1, $plaintext));
  328.  
  329.                 return ($show_encrypt) ? '{SHA}' . $encrypted : $encrypted;
  330.  
  331.             case 'crypt':
  332.             case 'crypt-des':
  333.             case 'crypt-md5':
  334.             case 'crypt-blowfish':
  335.                 return ($show_encrypt ? '{crypt}' : '') . crypt($plaintext, $salt);
  336.  
  337.             case 'md5-base64':
  338.                 $encrypted = base64_encode(mhash(MHASH_MD5, $plaintext));
  339.  
  340.                 return ($show_encrypt) ? '{MD5}' . $encrypted : $encrypted;
  341.  
  342.             case 'ssha':
  343.                 $encrypted = base64_encode(mhash(MHASH_SHA1, $plaintext . $salt) . $salt);
  344.  
  345.                 return ($show_encrypt) ? '{SSHA}' . $encrypted : $encrypted;
  346.  
  347.             case 'smd5':
  348.                 $encrypted = base64_encode(mhash(MHASH_MD5, $plaintext . $salt) . $salt);
  349.  
  350.                 return ($show_encrypt) ? '{SMD5}' . $encrypted : $encrypted;
  351.  
  352.             case 'aprmd5':
  353.                 $length = strlen($plaintext);
  354.                 $context = $plaintext . '$apr1$' . $salt;
  355.                 $binary = self::_bin(md5($plaintext . $salt . $plaintext));
  356.  
  357.                 for ($i = $length; $i > 0; $i -= 16)
  358.                 {
  359.                     $context .= substr($binary, 0, ($i > 16 ? 16 : $i));
  360.                 }
  361.                 for ($i = $length; $i > 0; $i >>= 1)
  362.                 {
  363.                     $context .= ($i & 1) ? chr(0) : $plaintext[0];
  364.                 }
  365.  
  366.                 $binary = self::_bin(md5($context));
  367.  
  368.                 for ($i = 0; $i < 1000; $i++)
  369.                 {
  370.                     $new = ($i & 1) ? $plaintext : substr($binary, 0, 16);
  371.  
  372.                     if ($i % 3)
  373.                     {
  374.                         $new .= $salt;
  375.                     }
  376.                     if ($i % 7)
  377.                     {
  378.                         $new .= $plaintext;
  379.                     }
  380.                     $new .= ($i & 1) ? substr($binary, 0, 16) : $plaintext;
  381.                     $binary = self::_bin(md5($new));
  382.                 }
  383.  
  384.                 $p = array();
  385.  
  386.                 for ($i = 0; $i < 5; $i++)
  387.                 {
  388.                     $k = $i + 6;
  389.                     $j = $i + 12;
  390.  
  391.                     if ($j == 16)
  392.                     {
  393.                         $j = 5;
  394.                     }
  395.                     $p[] = self::_toAPRMD5((ord($binary[$i]) << 16) | (ord($binary[$k]) << 8) | (ord($binary[$j])), 5);
  396.                 }
  397.  
  398.                 return '$apr1$' . $salt . '$' . implode('', $p) . self::_toAPRMD5(ord($binary[11]), 3);
  399.  
  400.             case 'md5-hex':
  401.             default:
  402.                 $encrypted = ($salt) ? md5($plaintext . $salt) : md5($plaintext);
  403.  
  404.                 return ($show_encrypt) ? '{MD5}' . $encrypted : $encrypted;
  405.         }
  406.     }

  408.     /**
  409.      * Returns a salt for the appropriate kind of password encryption.
  410.      * Optionally takes a seed and a plaintext password, to extract the seed
  411.      * of an existing password, or for encryption types that use the plaintext
  412.      * in the generation of the salt.
  413.      *
  414.      * @param   string  $encryption  The kind of password encryption to use.
  415.      *                                Defaults to md5-hex.
  416.      * @param   string  $seed        The seed to get the salt from (probably a
  417.      *                                previously generated password). Defaults to
  418.      *                                generating a new seed.
  419.      * @param   string  $plaintext   The plaintext password that we're generating
  420.      *                                a salt for. Defaults to none.
  421.      *
  422.      * @return  string  The generated or extracted salt.
  423.      *
  424.      * @since   11.1
  425.      */
  426.     public static function getSalt($encryption = 'md5-hex', $seed = '', $plaintext = '')
  427.     {
  428.         // Encrypt the password.
  429.         switch ($encryption)
  430.         {
  431.             case 'crypt':
  432.             case 'crypt-des':
  433.                 if ($seed)
  434.                 {
  435.                     return substr(preg_replace('|^{crypt}|i', '', $seed), 0, 2);
  436.                 }
  437.                 else
  438.                 {
  439.                     return substr(md5(mt_rand()), 0, 2);
  440.                 }
  441.                 break;
  442.  
  443.             case 'crypt-md5':
  444.                 if ($seed)
  445.                 {
  446.                     return substr(preg_replace('|^{crypt}|i', '', $seed), 0, 12);
  447.                 }
  448.                 else
  449.                 {
  450.                     return '$1$' . substr(md5(mt_rand()), 0, 8) . '$';
  451.                 }
  452.                 break;
  453.  
  454.             case 'crypt-blowfish':
  455.                 if ($seed)
  456.                 {
  457.                     return substr(preg_replace('|^{crypt}|i', '', $seed), 0, 16);
  458.                 }
  459.                 else
  460.                 {

  528.     {
  529.         $salt = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
  530.         $base = strlen($salt);
  531.         $makepass = '';
  532.  
  533.         /*
  534.          * Start with a cryptographic strength random string, then convert it to
  535.          * a string with the numeric base of the salt.
  536.          * Shift the base conversion on each character so the character
  537.          * distribution is even, and randomize the start shift so it's not
  538.          * predictable.
  539.          */
  540.         $random = JCrypt::genRandomBytes($length + 1);
  541.         $shift = ord($random[0]);
  542.  
  543.         for ($i = 1; $i <= $length; ++$i)
  544.         {
  545.             $makepass .= $salt[($shift + ord($random[$i])) % $base];
  546.             $shift += ord($random[$i]);
  547.         }
  548.  
  549.         return $makepass;
  550.     }
