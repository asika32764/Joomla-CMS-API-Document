  142.     /**

  202.             return (preg_match('/[' . $vars['vowels'] . ']/', utf8_encode($reversed_stem)));

  207.      * Paice/Husk stemmer which returns a stem for the given $input
  208.      *
  209.      * @param   string  $input  The word for which we want the stem in UTF-8
  210.      *
  211.      * @return  string  The stem

  215.     private static function _getStem($input)
  216.     {
  217.         $vars = static::getStemRules();

  222.  
  223.         // This loop goes through the rules' array until it finds an ending one (ending by '.') or the last one ('end0.')
  224.         while (true)
  225.         {

  227.  
  228.             if ($rule_number == -1)

  230.                 // No other rule can be applied => the stem has been found
  231.                 break;

  233.             $rule = $vars['rules'][$rule_number];
  234.             preg_match($vars['rule_pattern'], $rule, $matches);
  235.  
  236.             if (($matches[2] != '*') || ($intact))
  237.             {

  239.  
  240.                 if (self::_check($reversed_stem))
  241.                 {
  242.                     $reversed_input = $reversed_stem;
  243.  
  244.                     if ($matches[5] == '.')
  245.                     {
  246.                         break;
  247.                     }
  248.                 }
  249.                 else
  250.                 {
  251.                     // Go to another rule
  252.                     $rule_number++;
  253.                 }
  254.             }
  255.             else
  256.             {
  257.                 // Go to another rule
  258.                 $rule_number++;
  259.             }
  260.         }
  261.  
  262.         return utf8_encode(strrev($reversed_input));
  263.     }
