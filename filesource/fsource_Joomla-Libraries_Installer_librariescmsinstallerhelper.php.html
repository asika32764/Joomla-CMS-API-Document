  25.     /**
  26.      * Downloads a package
  27.      *
  28.      * @param   string  $url     URL of file to download
  29.      * @param   string  $target  Download target filename [optional]
  30.      *
  31.      * @return  mixed  Path to downloaded package or boolean false on failure
  32.      *
  33.      * @since   3.1
  34.      */
  35.     public static function downloadPackage($url, $target = false)
  36.     {
  37.         $config = JFactory::getConfig();
  38.  
  39.         // Capture PHP errors
  40.         $track_errors = ini_get('track_errors');
  41.         ini_set('track_errors', true);
  42.  
  43.         // Set user agent
  44.         $version = new JVersion;
  45.         ini_set('user_agent', $version->getUserAgent('Installer'));
  46.  
  47.         $http = JHttpFactory::getHttp();
  48.         $response = $http->get($url);
  49.  
  50.         if (302 == $response->code && isset($response->headers['Location']))
  51.         {
  52.             return self::downloadPackage($response->headers['Location']);
  53.         }
  54.         elseif (200 != $response->code)
  55.         {
  56.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_DOWNLOAD_SERVER_CONNECT'), JLog::WARNING, 'jerror');
  57.  
  58.             return false;
  59.         }
  60.  
  61.         if (isset($response->headers['Content-Disposition']))
  62.         {
  63.             $contentfilename = explode("\"", $response->headers['Content-Disposition']);
  64.             $target = $contentfilename[1];
  65.         }
  66.  
  67.         // Set the target path if not given
  68.         if (!$target)
  69.         {
  70.             $target = $config->get('tmp_path') . '/' . self::getFilenameFromURL($url);
  71.         }
  72.         else
  73.         {
  74.             $target = $config->get('tmp_path') . '/' . basename($target);
  75.         }
  76.  
  77.         // Write buffer to file
  78.         JFile::write($target, $response->body);
  79.  
  80.         // Restore error tracking to what it was before
  81.         ini_set('track_errors', $track_errors);
  82.  
  83.         // Bump the max execution time because not using built in php zip libs are slow
  84.         @set_time_limit(ini_get('max_execution_time'));
  85.  
  86.         // Return the name of the downloaded package
  87.         return basename($target);
  88.     }

  90.     /**
  91.      * Unpacks a file and verifies it as a Joomla element package
  92.      * Supports .gz .tar .tar.gz and .zip
  93.      *
  94.      * @param   string  $p_filename  The uploaded package filename or install directory
  95.      *
  96.      * @return  mixed  Array on success or boolean false on failure
  97.      *
  98.      * @since   3.1
  99.      */
  100.     public static function unpack($p_filename)
  101.     {
  102.         // Path to the archive
  103.         $archivename = $p_filename;
  104.  
  105.         // Temporary folder to extract the archive into
  106.         $tmpdir = uniqid('install_');
  107.  
  108.         // Clean the paths to use for archive extraction
  109.         $extractdir = JPath::clean(dirname($p_filename) . '/' . $tmpdir);
  110.         $archivename = JPath::clean($archivename);
  111.  
  112.         // Do the unpacking of the archive
  113.         try
  114.         {
  115.             JArchive::extract($archivename, $extractdir);
  116.         }
  117.         catch (Exception $e)
  118.         {
  119.             return false;
  120.         }
  121.  
  122.         /*
  123.          * Let's set the extraction directory and package file in the result array so we can
  124.          * cleanup everything properly later on.
  125.          */
  126.         $retval['extractdir'] = $extractdir;
  127.         $retval['packagefile'] = $archivename;
  128.  
  129.         /*
  130.          * Try to find the correct install directory.  In case the package is inside a
  131.          * subdirectory detect this and set the install directory to the correct path.
  132.          *
  133.          * List all the items in the installation directory.  If there is only one, and
  134.          * it is a folder, then we will set that folder to be the installation folder.
  135.          */
  136.         $dirList = array_merge(JFolder::files($extractdir, ''), JFolder::folders($extractdir, ''));
  137.  
  138.         if (count($dirList) == 1)
  139.         {
  140.             if (JFolder::exists($extractdir . '/' . $dirList[0]))
  141.             {
  142.                 $extractdir = JPath::clean($extractdir . '/' . $dirList[0]);
  143.             }
  144.         }
  145.  
  146.         /*
  147.          * We have found the install directory so lets set it and then move on
  148.          * to detecting the extension type.
  149.          */
  150.         $retval['dir'] = $extractdir;
  151.  
  152.         /*
  153.          * Get the extension type and return the directory/type array on success or
  154.          * false on fail.
  155.          */
  156.         $retval['type'] = self::detectType($extractdir);
  157.  
  158.         if ($retval['type'])
  159.         {
  160.             return $retval;
  161.         }
  162.         else
  163.         {
  164.             return false;
  165.         }
  166.     }

  168.     /**
  169.      * Method to detect the extension type from a package directory
  170.      *
  171.      * @param   string  $p_dir  Path to package directory
  172.      *
  173.      * @return  mixed  Extension type string or boolean false on fail
  174.      *
  175.      * @since   3.1
  176.      */
  177.     public static function detectType($p_dir)
  178.     {
  179.         // Search the install dir for an XML file
  180.         $files = JFolder::files($p_dir, '\.xml$', 1, true);
  181.  
  182.         if (!count($files))
  183.         {
  184.             JLog::add(JText::_('JLIB_INSTALLER_ERROR_NOTFINDXMLSETUPFILE'), JLog::WARNING, 'jerror');
  185.  
  186.             return false;
  187.         }
  188.  
  189.         foreach ($files as $file)
  190.         {
  191.             $xml = simplexml_load_file($file);
  192.  
  193.             if (!$xml)
  194.             {
  195.                 continue;
  196.             }
  197.  
  198.             if ($xml->getName() != 'extension')
  199.             {
  200.                 unset($xml);
  201.                 continue;
  202.             }
  203.  
  204.             $type = (string) $xml->attributes()->type;
  205.  
  206.             // Free up memory
  207.             unset($xml);
  208.  
  209.             return $type;
  210.         }
  211.  
  212.         JLog::add(JText::_('JLIB_INSTALLER_ERROR_NOTFINDJOOMLAXMLSETUPFILE'), JLog::WARNING, 'jerror');
  213.  
  214.         // Free up memory.
  215.         unset($xml);
  216.  
  217.         return false;
  218.     }

  220.     /**

  229.     public static function getFilenameFromURL($url)

  251.     {
  252.         $config = JFactory::getConfig();
  253.  
  254.         // Does the unpacked extension directory exist?
  255.         if (is_dir($resultdir))
  256.         {
  257.             JFolder::delete($resultdir);
  258.         }
  259.

  261.         if (is_file($package))

  263.             JFile::delete($package);
  264.         }
  265.         elseif (is_file(JPath::clean($config->get('tmp_path') . '/' . $package)))
  266.         {
  267.             // It might also be just a base filename

  269.         }
  270.     }
