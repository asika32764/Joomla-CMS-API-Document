  316.     /**
  317.      * Redirect to another URL.
  318.      *
  319.      * Optionally enqueues a message in the system message queue (which will be displayed
  320.      * the next time a page is loaded) using the enqueueMessage method. If the headers have
  321.      * not been sent the redirect will be accomplished using a "301 Moved Permanently"
  322.      * code in the header pointing to the new location. If the headers have already been
  323.      * sent this will be accomplished using a JavaScript statement.
  324.      *
  325.      * @param   string   $url      The URL to redirect to. Can only be http/https URL
  326.      * @param   string   $msg      An optional message to display on redirect.
  327.      * @param   string   $msgType  An optional message type. Defaults to message.
  328.      * @param   boolean  $moved    True if the page is 301 Permanently Moved, otherwise 303 See Other is assumed.
  329.      *
  330.      * @return  void  Calls exit().
  331.      *
  332.      * @since   11.1
  333.      *
  334.      * @see     JApplication::enqueueMessage()
  335.      */
  336.     public function redirect($url, $msg = '', $msgType = 'message', $moved = false)
  337.     {
  338.         // Check for relative internal links.
  339.         if (preg_match('#^index2?\.php#', $url))
  340.         {
  341.             $url = JUri::base() . $url;
  342.         }
  343.  
  344.         // Strip out any line breaks.
  345.         $url = preg_split("/[\r\n]/", $url);
  346.         $url = $url[0];
  347.  
  348.         /*
  349.          * If we don't start with a http we need to fix this before we proceed.
  350.          * We could validly start with something else (e.g. ftp), though this would
  351.          * be unlikely and isn't supported by this API.
  352.          */
  353.         if (!preg_match('#^http#i', $url))
  354.         {
  355.             $uri = JUri::getInstance();
  356.             $prefix = $uri->toString(array('scheme', 'user', 'pass', 'host', 'port'));
  357.  
  358.             if ($url[0] == '/')
  359.             {
  360.                 // We just need the prefix since we have a path relative to the root.
  361.                 $url = $prefix . $url;
  362.             }
  363.             else
  364.             {
  365.                 // It's relative to where we are now, so lets add that.
  366.                 $parts = explode('/', $uri->toString(array('path')));
  367.                 array_pop($parts);
  368.                 $path = implode('/', $parts) . '/';
  369.                 $url = $prefix . $path . $url;
  370.             }
  371.         }
  372.  
  373.         // If the message exists, enqueue it.
  374.         if (trim($msg))
  375.         {
  376.             $this->enqueueMessage($msg, $msgType);
  377.         }
  378.  
  379.         // Persist messages if they exist.
  380.         if (count($this->_messageQueue))
  381.         {
  382.             $session = JFactory::getSession();
  383.             $session->set('application.queue', $this->_messageQueue);
  384.         }
  385.  
  386.         // If the headers have been sent, then we cannot send an additional location header
  387.         // so we will output a javascript redirect statement.
  388.         if (headers_sent())
  389.         {
  390.             echo "<script>document.location.href='" . str_replace("'", "&apos;", $url) . "';</script>\n";
  391.         }
  392.         else
  393.         {
  394.             $document = JFactory::getDocument();
  395.  
  396.             jimport('phputf8.utils.ascii');
  397.             if (($this->client->engine == JApplicationWebClient::TRIDENT) && !utf8_is_ascii($url))
  398.             {
  399.                 // MSIE type browser and/or server cause issues when url contains utf8 character,so use a javascript redirect method
  400.                 echo '<html><head><meta http-equiv="content-type" content="text/html; charset=' . $document->getCharset() . '" />'
  401.                     . '<script>document.location.href=\'' . str_replace("'", "&apos;", $url) . '\';</script></head></html>';
  402.             }
  403.             else
  404.             {
  405.                 // All other browsers, use the more efficient HTTP header method
  406.                 header($moved ? 'HTTP/1.1 301 Moved Permanently' : 'HTTP/1.1 303 See other');
  407.                 header('Location: ' . $url);
  408.                 header('Content-Type: text/html; charset=' . $document->getCharset());
  409.             }
  410.         }
  411.         $this->close();
  412.     }

  414.     /**
  415.      * Enqueue a system message.
  416.      *
  417.      * @param   string  $msg   The message to enqueue.
  418.      * @param   string  $type  The message type. Default is message.
  419.      *
  420.      * @return  void 
  421.      *
  422.      * @since   11.1
  423.      */
  424.     public function enqueueMessage($msg, $type = 'message')
  425.     {
  426.         // For empty queue, if messages exists in the session, enqueue them first.
  427.         if (!count($this->_messageQueue))
  428.         {
  429.             $session = JFactory::getSession();
  430.             $sessionQueue = $session->get('application.queue');
  431.  
  432.             if (count($sessionQueue))
  433.             {
  434.                 $this->_messageQueue = $sessionQueue;
  435.                 $session->set('application.queue', null);
  436.             }
  437.         }
  438.  
  439.         // Enqueue the message.
  440.         $this->_messageQueue[] = array('message' => $msg, 'type' => strtolower($type));
  441.     }
  442.  
  443.     /**
  444.      * Get the system message queue.
  445.      *

  447.      *
  448.      * @since   11.1
  449.      */
  450.     public function getMessageQueue()
  451.     {
  452.         // For empty queue, if messages exists in the session, enqueue them.
  453.         if (!count($this->_messageQueue))
  454.         {
  455.             $session = JFactory::getSession();
  456.             $sessionQueue = $session->get('application.queue');
  457.  
  458.             if (count($sessionQueue))
  459.             {
  460.                 $this->_messageQueue = $sessionQueue;
  461.                 $session->set('application.queue', null);
  462.             }
  463.         }
  464.  
  465.         return $this->_messageQueue;
  466.     }

  472.      *
  473.      * @param   string  $varname  The name of the value to get.

  589.     /**
  590.      * Login authentication function.
  591.      *
  592.      * Username and encoded password are passed the onUserLogin event which
  593.      * is responsible for the user validation. A successful validation updates
  594.      * the current session record with the user's details.
  595.      *
  596.      * Username and encoded password are sent as credentials (along with other
  597.      * possibilities) to each observer (authentication plugin) for user
  598.      * validation.  Successful validation will update the current session with
  599.      * the user details.
  600.      *
  601.      * @param   array  $credentials  Array('username' => string, 'password' => string)
  602.      * @param   array  $options      Array('remember' => boolean)
  603.      *
  604.      * @return  boolean  True on success.
  605.      *
  606.      * @since   11.1
  607.      */
  608.     public function login($credentials, $options = array())
  609.     {
  610.         // Get the global JAuthentication object.
  611.         jimport('joomla.user.authentication');
  612.  
  613.         $authenticate = JAuthentication::getInstance();
  614.         $response = $authenticate->authenticate($credentials, $options);
  615.  
  616.         if ($response->status === JAuthentication::STATUS_SUCCESS)
  617.         {
  618.             // Validate that the user should be able to login (different to being authenticated).
  619.             // This permits authentication plugins blocking the user
  620.             $authorisations = $authenticate->authorise($response, $options);
  621.             foreach ($authorisations as $authorisation)
  622.             {
  623.                 $denied_states = array(JAuthentication::STATUS_EXPIRED, JAuthentication::STATUS_DENIED);
  624.                 if (in_array($authorisation->status, $denied_states))
  625.                 {
  626.                     // Trigger onUserAuthorisationFailure Event.
  627.                     $this->triggerEvent('onUserAuthorisationFailure', array((array) $authorisation));
  628.  
  629.                     // If silent is set, just return false.
  630.                     if (isset($options['silent']) && $options['silent'])
  631.                     {
  632.                         return false;
  633.                     }
  634.  
  635.                     // Return the error.
  636.                     switch ($authorisation->status)
  637.                     {
  638.                         case JAuthentication::STATUS_EXPIRED:
  639.                             return JError::raiseWarning('102002', JText::_('JLIB_LOGIN_EXPIRED'));
  640.                             break;
  641.                         case JAuthentication::STATUS_DENIED:
  642.                             return JError::raiseWarning('102003', JText::_('JLIB_LOGIN_DENIED'));
  643.                             break;
  644.                         default:
  645.                             return JError::raiseWarning('102004', JText::_('JLIB_LOGIN_AUTHORISATION'));
  646.                             break;
  647.                     }
  648.                 }
  649.             }
  650.  
  651.             // Import the user plugin group.
  652.             JPluginHelper::importPlugin('user');
  653.  
  654.             // OK, the credentials are authenticated and user is authorised.  Lets fire the onLogin event.
  655.             $results = $this->triggerEvent('onUserLogin', array((array) $response, $options));
  656.  
  657.             /*
  658.              * If any of the user plugins did not successfully complete the login routine
  659.              * then the whole method fails.
  660.              *
  661.              * Any errors raised should be done in the plugin as this provides the ability
  662.              * to provide much more information about why the routine may have failed.
  663.              */
  664.  
  665.             if (!in_array(false, $results, true))
  666.             {
  667.                 // Set the remember me cookie if enabled.
  668.                 if (isset($options['remember']) && $options['remember'])
  669.                 {
  670.                     // Create the encryption key, apply extra hardening using the user agent string.
  671.                     $privateKey = self::getHash(@$_SERVER['HTTP_USER_AGENT']);
  672.  
  673.                     $key = new JCryptKey('simple', $privateKey, $privateKey);
  674.                     $crypt = new JCrypt(new JCryptCipherSimple, $key);
  675.                     $rcookie = $crypt->encrypt(json_encode($credentials));
  676.                     $lifetime = time() + 365 * 24 * 60 * 60;
  677.  
  678.                     // Use domain and path set in config for cookie if it exists.
  679.                     $cookie_domain = $this->getCfg('cookie_domain', '');
  680.                     $cookie_path = $this->getCfg('cookie_path', '/');
  681.  
  682.                     $secure = $this->isSSLConnection();
  683.                     setcookie(self::getHash('JLOGIN_REMEMBER'), $rcookie, $lifetime, $cookie_path, $cookie_domain, $secure, true);
  684.                 }
  685.  
  686.                 return true;
  687.             }
  688.         }
  689.  
  690.         // Trigger onUserLoginFailure Event.
  691.         $this->triggerEvent('onUserLoginFailure', array((array) $response));
  692.  
  693.         // If silent is set, just return false.
  694.         if (isset($options['silent']) && $options['silent'])
  695.         {
  696.             return false;
  697.         }
  698.  
  699.         // If status is success, any error will have been raised by the user plugin
  700.         if ($response->status !== JAuthentication::STATUS_SUCCESS)
  701.         {
  702.             JLog::add($response->error_message, JLog::WARNING, 'jerror');
  703.         }
  704.  
  705.         return false;
  706.     }
  707.  
  708.     /**
  709.      * Logout authentication function.
  710.      *
  711.      * Passed the current user information to the onUserLogout event and reverts the current
  712.      * session record back to 'anonymous' parameters.
  713.      * If any of the authentication plugins did not successfully complete
  714.      * the logout routine then the whole method fails. Any errors raised
  715.      * should be done in the plugin as this provides the ability to give
  716.      * much more information about why the routine may have failed.
  717.      *
  718.      * @param   integer  $userid   The user to load - Can be an integer or string - If string, it is converted to ID automatically
  719.      * @param   array    $options  Array('clientid' => array of client id's)
  720.      *
  721.      * @return  boolean  True on success
  722.      *
  723.      * @since   11.1
  724.      */
  725.     public function logout($userid = null, $options = array())
  726.     {
  727.         // Get a user object from the JApplication.
  728.         $user = JFactory::getUser($userid);
  729.  
  730.         // Build the credentials array.
  731.         $parameters['username'] = $user->get('username');
  732.         $parameters['id'] = $user->get('id');
  733.  
  734.         // Set clientid in the options array if it hasn't been set already.
  735.         if (!isset($options['clientid']))
  736.         {
  737.             $options['clientid'] = $this->getClientId();
  738.         }
  739.  
  740.         // Import the user plugin group.
  741.         JPluginHelper::importPlugin('user');
  742.  
  743.         // OK, the credentials are built. Lets fire the onLogout event.
  744.         $results = $this->triggerEvent('onUserLogout', array($parameters, $options));
  745.  
  746.         // Check if any of the plugins failed. If none did, success.
  747.  
  748.         if (!in_array(false, $results, true))
  749.         {
  750.             // Use domain and path set in config for cookie if it exists.
  751.             $cookie_domain = $this->getCfg('cookie_domain', '');
  752.             $cookie_path = $this->getCfg('cookie_path', '/');
  753.             setcookie(self::getHash('JLOGIN_REMEMBER'), false, time() - 86400, $cookie_path, $cookie_domain);
  754.  
  755.             return true;
  756.         }
  757.  
  758.         // Trigger onUserLoginFailure Event.
  759.         $this->triggerEvent('onUserLogoutFailure', array($parameters));
  760.  
  761.         return false;
  762.     }
