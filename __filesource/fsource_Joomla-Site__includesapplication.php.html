  233.         $document = JFactory::getDocument();

  253.                 }
  254.  
  255.                 if ($this->getCfg('offline') && !$user->authorise('core.login.offline'))
  256.                 {
  257.                     $uri = JUri::getInstance();

  262.                 }

  265.                     $file = 'component';
  266.                 }
  267.                 $params = array(
  268.                     'template' => $template->template,
  269.                     'file' => $file . '.php',
  270.                     'directory' => JPATH_THEMES,
  271.                     'params' => $template->params
  272.                 );
  273.                 break;

  275.  
  276.         // Parse the document.
  277.         $document = JFactory::getDocument();

  279.  
  280.         // Trigger the onBeforeRender event.

  284.         $caching = false;

  286.         {
  287.             $caching = true;
  288.         }
  289.  
  290.         // Render the document.
  291.         JResponse::setBody($document->render($caching, $params));
  292.  
  293.         // Trigger the onAfterRender event.

  295.     }
