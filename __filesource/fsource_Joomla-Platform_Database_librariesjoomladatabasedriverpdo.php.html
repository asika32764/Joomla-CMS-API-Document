  351.     /**

  360.     public function execute()
  361.     {

  363.  
  364.         if (!is_object($this->connection))
  365.         {

  377.  
  378.         // Increment the query counter.

  381.         // Reset the error values.
  382.         $this->errorNum = 0;
  383.         $this->errorMsg = '';
  384.  
  385.         // If debugging is enabled then let's log the query.
  386.         if ($this->debug)
  387.         {
  388.             // Add the query to the object queue.
  389.             $this->log[] = $query;
  390.  
  391.             JLog::add($query, JLog::DEBUG, 'databasequery');
  392.  
  393.             $this->timings[] = microtime(true);
  394.         }
  395.  
  396.         // Execute the query.
  397.         $this->executed = false;
  398.         if ($this->prepared instanceof PDOStatement)
  399.         {
  400.             // Bind the variables:
  401.             if ($this->sql instanceof JDatabaseQueryPreparable)
  402.             {
  403.                 $bounded =& $this->sql->getBounded();
  404.                 foreach ($bounded as $key => $obj)
  405.                 {
  406.                     $this->prepared->bindParam($key, $obj->value, $obj->dataType, $obj->length, $obj->driverOptions);
  407.                 }
  408.             }
  409.  
  410.             $this->executed = $this->prepared->execute();
  411.         }
  412.  
  413.         if ($this->debug)
  414.         {
  415.             $this->timings[] = microtime(true);
  416.             if (defined('DEBUG_BACKTRACE_IGNORE_ARGS'))
  417.             {
  418.                 $this->callStacks[] = debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS);
  419.             }
  420.             else
  421.             {
  422.                 $this->callStacks[] = debug_backtrace();
  423.             }
  424.         }
  425.  
  426.         // If an error occurred handle it.
  427.         if (!$this->executed)
  428.         {
  429.             // Get the error number and message before we execute any more queries.
  430.             $errorNum = (int) $this->connection->errorCode();
  431.             $errorMsg = (string) 'SQL: ' . implode(", ", $this->connection->errorInfo());
  432.  
  433.             // Check if the server was disconnected.
  434.             if (!$this->connected())
  435.             {
  436.                 try
  437.                 {
  438.                     // Attempt to reconnect.
  439.                     $this->connection = null;
  440.                     $this->connect();
  441.                 }
  442.                 // If connect fails, ignore that exception and throw the normal exception.
  443.                 catch (RuntimeException $e)
  444.                 {
  445.                     // Get the error number and message.
  446.                     $this->errorNum = (int) $this->connection->errorCode();
  447.                     $this->errorMsg = (string) 'SQL: ' . implode(", ", $this->connection->errorInfo());
  448.  
  449.                     // Throw the normal query exception.
  450.                     JLog::add(JText::sprintf('JLIB_DATABASE_QUERY_FAILED', $this->errorNum, $this->errorMsg), JLog::ERROR, 'databasequery');
  451.                     throw new RuntimeException($this->errorMsg, $this->errorNum);
  452.                 }
  453.  
  454.                 // Since we were able to reconnect, run the query again.
  455.                 return $this->execute();
  456.             }
  457.             // The server was not disconnected.
  458.             else
  459.             {
  460.                 // Get the error number and message from before we tried to reconnect.
  461.                 $this->errorNum = $errorNum;

  546.     public function connected()
  547.     {
  548.         // Flag to prevent recursion into this function.

  550.

  556.         }
  557.  
  558.         // Backup the query state.
  559.         $query = $this->sql;
  560.         $limit = $this->limit;
  561.         $offset = $this->offset;
  562.         $prepared = $this->prepared;
  563.  
  564.         try
  565.         {
  566.             // Set the checking connection flag.
  567.             $checkingConnected = true;

  570.             $this->setQuery($this->getConnectedQuery());
  571.             $status = (bool) $this->loadResult();

  580.         $this->sql = $query;
