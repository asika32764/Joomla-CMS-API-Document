  186. /**

  206.     if (!is_numeric($userId))
  207.     {
  208.         $vars['view'] = 'profile';

  210.     }
  211.  
  212.     if (is_numeric($userId))

  215.         $db = JFactory::getDbo();

  219.                 ' WHERE ' . $db->quoteName('id') . ' = ' . (int) $userId
