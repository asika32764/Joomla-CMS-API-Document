  372.     /**
  373.      * Get the list of files/data from a ZIP archive buffer.
  374.      *
  375.      * <pre>
  376.      * KEY: Position in zipfile
  377.      * VALUES: 'attr'  --  File attributes
  378.      * 'crc'   --  CRC checksum
  379.      * 'csize' --  Compressed file size
  380.      * 'date'  --  File modification time
  381.      * 'name'  --  Filename
  382.      * 'method'--  Compression method
  383.      * 'size'  --  Original file size
  384.      * 'type'  --  File type
  385.      * </pre>
  386.      *
  387.      * @param   string  &$data  The ZIP archive buffer.
  388.      *
  389.      * @return  boolean True on success
  390.      *
  391.      * @since   11.1
  392.      * @throws  RuntimeException
  393.      */
  394.     private function _readZipInfo(&$data)
  395.     {
  396.         $entries = array();
  397.  
  398.         // Find the last central directory header entry
  399.         $fhLast = strpos($data, $this->_ctrlDirEnd);
  400.  
  401.         do
  402.         {
  403.             $last = $fhLast;
  404.         }
  405.         while (($fhLast = strpos($data, $this->_ctrlDirEnd, $fhLast + 1)) !== false);
  406.  
  407.         // Find the central directory offset
  408.         $offset = 0;
  409.  
  410.         if ($last)
  411.         {
  412.             $endOfCentralDirectory = unpack(
  413.                 'vNumberOfDisk/vNoOfDiskWithStartOfCentralDirectory/vNoOfCentralDirectoryEntriesOnDisk/' .
  414.                 'vTotalCentralDirectoryEntries/VSizeOfCentralDirectory/VCentralDirectoryOffset/vCommentLength',
  415.                 substr($data, $last + 4)
  416.             );
  417.             $offset = $endOfCentralDirectory['CentralDirectoryOffset'];
  418.         }
  419.  
  420.         // Get details from central directory structure.
  421.         $fhStart = strpos($data, $this->_ctrlDirHeader, $offset);
  422.         $dataLength = strlen($data);
  423.  
  424.         do
  425.         {
  426.             if ($dataLength < $fhStart + 31)
  427.             {
  428.                 if (class_exists('JError'))
  429.                 {
  430.                     return JError::raiseWarning(100, 'Invalid Zip Data');
  431.                 }
  432.                 else
  433.                 {
  434.                     throw new RuntimeException('Invalid Zip Data');
  435.                 }
  436.             }
  437.  
  438.             $info = unpack('vMethod/VTime/VCRC32/VCompressed/VUncompressed/vLength', substr($data, $fhStart + 10, 20));
  439.             $name = substr($data, $fhStart + 46, $info['Length']);
  440.  
  441.             $entries[$name] = array(
  442.                 'attr' => null,
  443.                 'crc' => sprintf("%08s", dechex($info['CRC32'])),
  444.                 'csize' => $info['Compressed'],
  445.                 'date' => null,
  446.                 '_dataStart' => null,
  447.                 'name' => $name,
  448.                 'method' => $this->_methods[$info['Method']],
  449.                 '_method' => $info['Method'],
  450.                 'size' => $info['Uncompressed'],
  451.                 'type' => null
  452.             );
  453.  
  454.             $entries[$name]['date'] = mktime(
  455.                 (($info['Time'] >> 11) & 0x1f),
  456.                 (($info['Time'] >> 5) & 0x3f),
  457.                 (($info['Time'] << 1) & 0x3e),
  458.                 (($info['Time'] >> 21) & 0x07),
  459.                 (($info['Time'] >> 16) & 0x1f),
  460.                 ((($info['Time'] >> 25) & 0x7f) + 1980)
  461.             );
  462.  
  463.             if ($dataLength < $fhStart + 43)
  464.             {
  465.                 if (class_exists('JError'))
  466.                 {
  467.                     return JError::raiseWarning(100, 'Invalid ZIP data');
  468.                 }
  469.                 else
  470.                 {
  471.                     throw new RuntimeException('Invalid ZIP data');
  472.                 }
  473.             }
  474.  
  475.             $info = unpack('vInternal/VExternal/VOffset', substr($data, $fhStart + 36, 10));
  476.  
  477.             $entries[$name]['type'] = ($info['Internal'] & 0x01) ? 'text' : 'binary';
  478.             $entries[$name]['attr'] = (($info['External'] & 0x10) ? 'D' : '-') . (($info['External'] & 0x20) ? 'A' : '-')
  479.                 . (($info['External'] & 0x03) ? 'S' : '-') . (($info['External'] & 0x02) ? 'H' : '-') . (($info['External'] & 0x01) ? 'R' : '-');
  480.             $entries[$name]['offset'] = $info['Offset'];
  481.  
  482.             // Get details from local file header since we have the offset
  483.             $lfhStart = strpos($data, $this->_fileHeader, $entries[$name]['offset']);
  484.  
  485.             if ($dataLength < $lfhStart + 34)
  486.             {
  487.                 if (class_exists('JError'))
  488.                 {
  489.                     return JError::raiseWarning(100, 'Invalid Zip Data');
  490.                 }
  491.                 else
  492.                 {
  493.                     throw new RuntimeException('Invalid Zip Data');
  494.                 }
  495.             }
  496.  
  497.             $info = unpack('vMethod/VTime/VCRC32/VCompressed/VUncompressed/vLength/vExtraLength', substr($data, $lfhStart + 8, 25));
  498.             $name = substr($data, $lfhStart + 30, $info['Length']);
  499.             $entries[$name]['_dataStart'] = $lfhStart + 30 + $info['Length'] + $info['ExtraLength'];
  500.  
  501.             // Bump the max execution time because not using the built in php zip libs makes this process slow.
  502.             @set_time_limit(ini_get('max_execution_time'));
  503.         }
  504.         while ((($fhStart = strpos($data, $this->_ctrlDirHeader, $fhStart + 46)) !== false));
  505.

  507.  
  508.         return true;
  509.     }

  511.     /**
  512.      * Returns the file data for a file by offsest in the ZIP archive
  513.      *
  514.      * @param   integer  $key  The position of the file in the archive.
  515.      *
  516.      * @return  string  Uncompressed file data buffer.
  517.      *
  518.      * @since   11.1
  519.      */
  520.     private function _getFileData($key)
  521.     {
  522.         if ($this->_metadata[$key]['_method'] == 0x8)
  523.         {
  524.             return gzinflate(substr($this->_data, $this->_metadata[$key]['_dataStart'], $this->_metadata[$key]['csize']));
  525.         }
  526.         elseif ($this->_metadata[$key]['_method'] == 0x0)
  527.         {
  528.             /* Files that aren't compressed. */
  529.             return substr($this->_data, $this->_metadata[$key]['_dataStart'], $this->_metadata[$key]['csize']);
  530.         }
  531.         elseif ($this->_metadata[$key]['_method'] == 0x12)
  532.         {
  533.             // If bz2 extension is loaded use it
  534.             if (extension_loaded('bz2'))
  535.             {
  536.                 return bzdecompress(substr($this->_data, $this->_metadata[$key]['_dataStart'], $this->_metadata[$key]['csize']));
  537.             }
  538.         }
  539.  
  540.         return '';
  541.     }

  543.     /**
  544.      * Converts a UNIX timestamp to a 4-byte DOS date and time format
  545.      * (date in high 2-bytes, time in low 2-bytes allowing magnitude
  546.      * comparison).
  547.      *
  548.      * @param   int  $unixtime  The current UNIX timestamp.
  549.      *
  550.      * @return  int  The current date in a 4-byte DOS format.
  551.      *
  552.      * @since   11.1
  553.      */
  554.     protected function _unix2DOSTime($unixtime = null)
  555.     {
  556.         $timearray = (is_null($unixtime)) ? getdate() : getdate($unixtime);
  557.  
  558.         if ($timearray['year'] < 1980)
  559.         {
  560.             $timearray['year'] = 1980;
  561.             $timearray['mon'] = 1;
  562.             $timearray['mday'] = 1;
  563.             $timearray['hours'] = 0;
  564.             $timearray['minutes'] = 0;
  565.             $timearray['seconds'] = 0;
  566.         }
  567.  
  568.         return (($timearray['year'] - 1980) << 25) | ($timearray['mon'] << 21) | ($timearray['mday'] << 16) | ($timearray['hours'] << 11) |
  569.             ($timearray['minutes'] << 5) | ($timearray['seconds'] >> 1);
  570.     }

  572.     /**
  573.      * Adds a "file" to the ZIP archive.
  574.      *
  575.      * @param   array  &$file      File data array to add
  576.      * @param   array  &$contents  An array of existing zipped files.
  577.      * @param   array  &$ctrldir   An array of central directory information.
  578.      *
  579.      * @return  void 
  580.      *
  581.      * @since   11.1
  582.      *
  583.      * @todo    Review and finish implementation
  584.      */
  585.     private function _addToZIPFile(array &$file, array &$contents, array &$ctrldir)
  586.     {
  587.         $data = &$file['data'];
  588.         $name = str_replace('\\', '/', $file['name']);
  589.  
  590.         /* See if time/date information has been provided. */
  591.         $ftime = null;
  592.  
  593.         if (isset($file['time']))
  594.         {
  595.             $ftime = $file['time'];
  596.         }
  597.  
  598.         // Get the hex time.
  599.         $dtime = dechex($this->_unix2DosTime($ftime));
  600.         $hexdtime = chr(hexdec($dtime[6] . $dtime[7])) . chr(hexdec($dtime[4] . $dtime[5])) . chr(hexdec($dtime[2] . $dtime[3]))
  601.             . chr(hexdec($dtime[0] . $dtime[1]));
  602.  
  603.         /* Begin creating the ZIP data. */
  604.         $fr = $this->_fileHeader;
  605.         /* Version needed to extract. */
  606.         $fr .= "\x14\x00";
  607.         /* General purpose bit flag. */
  608.         $fr .= "\x00\x00";
  609.         /* Compression method. */
  610.         $fr .= "\x08\x00";
  611.         /* Last modification time/date. */
  612.         $fr .= $hexdtime;
  613.  
  614.         /* "Local file header" segment. */
  615.         $unc_len = strlen($data);
  616.         $crc = crc32($data);
  617.         $zdata = gzcompress($data);
  618.         $zdata = substr(substr($zdata, 0, strlen($zdata) - 4), 2);
  619.         $c_len = strlen($zdata);
  620.  
  621.         /* CRC 32 information. */
  622.         $fr .= pack('V', $crc);
  623.         /* Compressed filesize. */
  624.         $fr .= pack('V', $c_len);
  625.         /* Uncompressed filesize. */
  626.         $fr .= pack('V', $unc_len);
  627.         /* Length of filename. */
  628.         $fr .= pack('v', strlen($name));
  629.         /* Extra field length. */
  630.         $fr .= pack('v', 0);
  631.         /* File name. */
  632.         $fr .= $name;
  633.  
  634.         /* "File data" segment. */
  635.         $fr .= $zdata;
  636.  
  637.         /* Add this entry to array. */
  638.         $old_offset = strlen(implode('', $contents));
  639.         $contents[] = &$fr;
  640.  
  641.         /* Add to central directory record. */
  642.         $cdrec = $this->_ctrlDirHeader;
  643.         /* Version made by. */
  644.         $cdrec .= "\x00\x00";
  645.         /* Version needed to extract */
  646.         $cdrec .= "\x14\x00";
  647.         /* General purpose bit flag */
  648.         $cdrec .= "\x00\x00";
  649.         /* Compression method */
  650.         $cdrec .= "\x08\x00";
  651.         /* Last mod time/date. */
  652.         $cdrec .= $hexdtime;
  653.         /* CRC 32 information. */
  654.         $cdrec .= pack('V', $crc);
  655.         /* Compressed filesize. */
  656.         $cdrec .= pack('V', $c_len);
  657.         /* Uncompressed filesize. */
  658.         $cdrec .= pack('V', $unc_len);
  659.         /* Length of filename. */
  660.         $cdrec .= pack('v', strlen($name));
  661.         /* Extra field length. */
  662.         $cdrec .= pack('v', 0);
  663.         /* File comment length. */
  664.         $cdrec .= pack('v', 0);
  665.         /* Disk number start. */
  666.         $cdrec .= pack('v', 0);
  667.         /* Internal file attributes. */
  668.         $cdrec .= pack('v', 0);
  669.         /* External file attributes -'archive' bit set. */
  670.         $cdrec .= pack('V', 32);
  671.         /* Relative offset of local header. */
  672.         $cdrec .= pack('V', $old_offset);
  673.         /* File name. */
  674.         $cdrec .= $name;
  675.         /* Optional extra field, file comment goes here. */
  676.  
  677.         /* Save to central directory array. */
  678.         $ctrldir[] = &$cdrec;
  679.     }

  696.     private function _createZIPFile(array &$contents, array &$ctrlDir, $path)
  697.     {
  698.         $data = implode('', $contents);
  699.         $dir = implode('', $ctrlDir);
  700.  
  701.         $buffer = $data . $dir . $this->_ctrlDirEnd . /* Total # of entries "on this disk". */

  708.         if (JFile::write($path, $buffer) === false)
  709.         {

  713.         {
