  471.      * @param   boolean  $includeChildren  True to include the results from child tags
  472.      * @param   string   $orderByOption    Column to order the results by
  473.      * @param   string   $orderDir         Direction to sort the results in
  474.      * @param   boolean  $anyOrAll         True to include items matching at least one tag, false to include
  475.      *                                      items all tags in the array.
  476.      * @param   string   $languageFilter   Optional filter on language. Options are 'all', 'current' or any string.
  477.      * @param   string   $stateFilter      Optional filtering on publication state, defaults to published or unpublished.
  478.      *
  479.      * @return  JDatabaseQuery  Query to retrieve a list of tags
  480.      *
  481.      * @since   3.1
  482.      */
  483.     public function getTagItemsQuery($tagId, $typesr = null, $includeChildren = false, $orderByOption = 'c.core_title', $orderDir = 'ASC',
  484.         $anyOrAll = true, $languageFilter = 'all', $stateFilter = '0,1')
  485.     {
  486.         // Create a new query object.
  487.         $db = JFactory::getDbo();
  488.         $query = $db->getQuery(true);
  489.         $user = JFactory::getUser();
  490.         $nullDate = $db->quote($db->getNullDate());
  491.  
  492.         $ntagsr = substr_count($tagId, ',') + 1;
  493.  
  494.         // If we want to include children we have to adjust the list of tags.
  495.         // We do not search child tags when the match all option is selected.
  496.         if ($includeChildren)
  497.         {
  498.             if (!is_array($tagId))
  499.             {
  500.                 $tagIdArray = explode(',', $tagId);
  501.             }
  502.             else
  503.             {
  504.                 $tagIdArray = $tagId;
  505.             }
  506.  
  507.             $tagTreeList = '';
  508.  
  509.             foreach ($tagIdArray as $tag)
  510.             {
  511.                 if ($this->getTagTreeArray($tag, $tagTreeArray))
  512.                 {
  513.                     $tagTreeList .= implode(',', $this->getTagTreeArray($tag, $tagTreeArray)) . ',';
  514.                 }
  515.             }
  516.  
  517.             if ($tagTreeList)
  518.             {
  519.                 $tagId = trim($tagTreeList, ',');
  520.             }
  521.         }
  522.  
  523.         if (is_array($tagId))
  524.         {
  525.             $tagId = implode(',', $tagId);
  526.         }
  527.  
  528.         // M is the mapping table. C is the core_content table. Ct is the content_types table.
  529.         $query->select('m.type_alias, m.content_item_id, m.core_content_id, count(m.tag_id) AS match_count,  MAX(m.tag_date) as tag_date, MAX(c.core_title) AS core_title')
  530.             ->select('MAX(c.core_alias) AS core_alias, MAX(c.core_body) AS core_body, MAX(c.core_state) AS core_state, MAX(c.core_access) AS core_access')
  531.             ->select('MAX(c.core_metadata) AS core_metadata, MAX(c.core_created_user_id) AS core_created_user_id, MAX(c.core_created_by_alias) AS core_created_by_alias')
  532.             ->select('MAX(c.core_created_time) as core_created_time, MAX(c.core_images) as core_images')
  533.             ->select('CASE WHEN c.core_modified_time = ' . $nullDate . ' THEN c.core_created_time ELSE c.core_modified_time END as core_modified_time')
  534.             ->select('MAX(c.core_language) AS core_language, MAX(c.core_catid) AS core_catid')
  535.             ->select('MAX(c.core_publish_up) AS core_publish_up, MAX(c.core_publish_down) as core_publish_down')
  536.             ->select('MAX(ct.type_title) AS content_type_title, MAX(ct.router) AS router')
  537.  
  538.             ->from('#__contentitem_tag_map AS m')
  539.             ->join('INNER', '#__ucm_content AS c ON m.type_alias = c.core_type_alias AND m.core_content_id = c.core_content_id')
  540.             ->join('INNER', '#__content_types AS ct ON ct.type_alias = m.type_alias')
  541.  
  542.             // Join over the users for the author and email
  543.             ->select("CASE WHEN c.core_created_by_alias > ' ' THEN c.core_created_by_alias ELSE ua.name END AS author")
  544.             ->select("ua.email AS author_email")
  545.  
  546.             ->join('LEFT', '#__users AS ua ON ua.id = c.core_created_user_id')
  547.  
  548.             ->where('m.tag_id IN (' . $tagId . ')')
  549.             ->where('c.core_state IN (' . $stateFilter . ')');
  550.  
  551.         // Optionally filter on language
  552.         if (empty($language))
  553.         {
  554.             $language = $languageFilter;
  555.         }
  556.  
  557.         if ($language != 'all')
  558.         {
  559.             if ($language == 'current_language')
  560.             {
  561.                 $language = JHelperContent::getCurrentLanguage();
  562.             }
  563.  
  564.             $query->where($db->quoteName('c.core_language') . ' IN (' . $db->quote($language) . ', ' . $db->quote('*') . ')');
  565.         }
  566.  
  567.         // Get the type data, limited to types in the request if there are any specified.
  568.         $typesarray = self::getTypes('assocList', $typesr, false);
  569.  
  570.         $typeAliases = '';
  571.  
  572.         foreach ($typesarray as $type)
  573.         {
  574.             $typeAliases .= "'" . $type['type_alias'] . "'" . ',';
  575.         }
  576.  
  577.         $typeAliases = rtrim($typeAliases, ',');
  578.         $query->where('m.type_alias IN (' . $typeAliases . ')');
  579.  
  580.         $groups = '0,' . implode(',', array_unique($user->getAuthorisedViewLevels()));
  581.         $query->where('c.core_access IN (' . $groups . ')')
  582.             ->group('m.type_alias, m.content_item_id, m.core_content_id');
  583.  
  584.         // Use HAVING if matching all tags and we are matching more than one tag.
  585.         if ($ntagsr > 1 && $anyOrAll != 1 && $includeChildren != 1)
  586.         {
  587.             // The number of results should equal the number of tags requested.
  588.             $query->having("COUNT('m.tag_id') = " . $ntagsr);
  589.         }
  590.  
  591.         // Set up the order by using the option chosen
  592.         if ($orderByOption == 'match_count')
  593.         {
  594.             $orderBy = 'COUNT(m.tag_id)';
  595.         }
  596.         else
  597.         {
  598.             $orderBy = 'MAX(' . $orderByOption . ')';
  599.         }
  600.  
  601.         $query->order($orderBy . ' ' . $orderDir);
  602.  
  603.         return $query;
  604.     }

  638.     /**
  639.      * Method to get an array of tag ids for the current tag and its children

  644.      * @return  mixed 
  645.      *

  648.     public function getTagTreeArray($id, &$tagTreeArray = array())
  649.     {

  756.     /**
  757.      * Function that handles saving tags used in a table class after a store()
  758.      *
  759.      * @param   JTable   $table    JTable being processed
  760.      * @param   array    $newTags  Array of new tags
  761.      * @param   boolean  $replace  Flag indicating if all exising tags should be replaced
  762.      *
  763.      * @return  boolean 
  764.      *
  765.      * @since   3.1
  766.      */
  767.     public function postStoreProcess($table, $newTags = array(), $replace = true)
  768.     {
  769.         if (!empty($table->newTags) && empty($newTags))
  770.         {
  771.                 $newTags = $table->newTags;
  772.         }
  773.  
  774.         // If existing row, check to see if tags have changed.
  775.         $newTable = clone $table;
  776.         $newTable->reset();
  777.         $key = $newTable->getKeyName();
  778.         $typeAlias = $this->typeAlias;
  779.  
  780.         $result = true;
  781.  
  782.         // Process ucm_content and ucm_base if either tags have changed or we have some tags.
  783.         if ($this->tagsChanged || $newTags)
  784.         {
  785.             if (!$newTags)
  786.             {
  787.                 // Delete all tags data
  788.                 $key = $table->getKeyName();
  789.                 $result = $this->deleteTagData($table, $table->$key);
  790.             }
  791.             else
  792.             {
  793.                 // Process the tags
  794.                 $rowdata = new JHelperContent;
  795.  
  796.                 $data = $rowdata->getRowData($table);
  797.                 $ucmContentTable = JTable::getInstance('Corecontent');
  798.  
  799.                 $ucm = new JUcmContent($table, $this->typeAlias);
  800.                 $ucmData = $data ? $ucm->mapData($data) : $ucm->ucmData;
  801.  
  802.                 $primaryId = $ucm->getPrimaryKey($ucmData['common']['core_type_id'], $ucmData['common']['core_content_item_id']);
  803.                 $result = $ucmContentTable->load($primaryId);
  804.                 $result = $result && $ucmContentTable->bind($ucmData['common']);
  805.                 $result = $result && $ucmContentTable->check();
  806.                 $result = $result && $ucmContentTable->store();
  807.                 $ucmId = $ucmContentTable->core_content_id;
  808.  
  809.                 // Store the tag data if the article data was saved and run related methods.
  810.                 $result = $result && $this->tagItem($ucmId, $table, $newTags, $replace);
  811.             }
  812.         }
  813.         return $result;
  814.     }
  815.  
  816.     /**
  817.      * Function that preProcesses data from a table prior to a store() to ensure proper tag handling
  818.      *
  819.      * @param   JTable  $table    JTable being processed
  820.      * @param   array   $newTags  Array of new tags
  821.      *
  822.      * @return  null 
  823.      *
  824.      * @since   3.1
  825.      */
  826.     public function preStoreProcess($table, $newTags = array())
  827.     {
  828.         if ($newTags != array())
  829.         {
  830.             $this->newTags = $newTags;
  831.         }
  832.  
  833.         // If existing row, check to see if tags have changed.
  834.         $oldTable = clone $table;
  835.         $oldTable->reset();
  836.         $key = $oldTable->getKeyName();
  837.         $typeAlias = $this->typeAlias;
  838.  
  839.         if ($oldTable->$key && $oldTable->load())
  840.         {
  841.             $this->oldTags = $this->getTagIds($oldTable->$key, $typeAlias);
  842.         }
  843.  
  844.         // New items with no tags bypass this step.
  845.         if ((!empty($newTags) || (isset($newTags[0]) && $newTags[0] != '')) || isset($this->oldTags))
  846.         {
  847.             // We need to process tags if the tags have changed or if we have a new row
  848.             $this->tagsChanged = ($this->oldTags != $newTags) || !$table->$key;
  849.         }
  850.     }
  851.  
  852.     /**
  853.      * Function to search tags
  854.      *
  855.      * @param   array  $filters  Filter to apply to the search
  856.      *
  857.      * @return  array 
  858.      *
  859.      * @since   3.1
  860.      */
  861.     public static function searchTags($filters = array())
  862.     {
  863.         $db = JFactory::getDbo();
  864.         $query = $db->getQuery(true)
  865.             ->select('a.id AS value')
  866.             ->select('a.path AS text')
  867.             ->select('a.path')
  868.             ->from('#__tags AS a')
  869.             ->join('LEFT', $db->quoteName('#__tags', 'b') . ' ON a.lft > b.lft AND a.rgt < b.rgt');
  870.  
  871.         // Filter language
  872.         if (!empty($filters['flanguage']))
  873.         {
  874.             $query->where('a.language IN (' . $db->quote($filters['flanguage']) . ',' . $db->quote('*') . ') ');
  875.         }
  876.  
  877.         // Do not return root
  878.         $query->where($db->quoteName('a.alias') . ' <> ' . $db->quote('root'));
  879.  
  880.         // Search in title or path
  881.         if (!empty($filters['like']))
