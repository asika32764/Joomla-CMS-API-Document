  168.         if (!empty($commands['category_id']))

  178.                 }
  179.                 else
  180.                 {
  181.                     return false;
  182.                 }
  183.             }
  184.             elseif ($cmd == 'm' && !$this->batchMove($commands['category_id'], $pks, $contexts))
  185.             {

  190.  
  191.         if (!empty($commands['assetgroup_id']))
  192.         {
  193.             if (!$this->batchAccess($commands['assetgroup_id'], $pks, $contexts))
  194.             {
  195.                 return false;

  322.  
  323.         // Check that the user has create permission for the component

  329.             return false;
  330.         }
  331.  
  332.         // Parent exists so we let's proceed
  333.         while (!empty($pks))
  334.         {
  335.             // Pop the first ID off the stack
  336.             $pk = array_shift($pks);
  337.  
  338.             $table->reset();
  339.  
  340.             // Check that the row actually exists
  341.             if (!$table->load($pk))
  342.             {
  343.                 if ($error = $table->getError())
  344.                 {
  345.                     // Fatal error

  347.                     return false;
  348.                 }
  349.                 else
  350.                 {
  351.                     // Not fatal error
  352.                     $this->setError(JText::sprintf('JLIB_APPLICATION_ERROR_BATCH_MOVE_ROW_NOT_FOUND', $pk));
  353.                     continue;
  354.                 }
  355.             }
  356.  
  357.             // Alter the title & alias
  358.             $data = $this->generateNewTitle($categoryId, $table->alias, $table->title);
  359.             $table->title = $data['0'];
  360.             $table->alias = $data['1'];
  361.  
  362.             // Reset the ID because we are making a copy
  363.             $table->id = 0;
  364.  
  365.             // New category ID
  366.             $table->catid = $categoryId;
  367.  
  368.             // TODO: Deal with ordering?
  369.             // $table->ordering    = 1;
  370.  
  371.             // Check the row.
  372.             if (!$table->check())
  373.             {
  374.                 $this->setError($table->getError());
  375.                 return false;
  376.             }
  377.  
  378.             // Store the row.
  379.             if (!$table->store())
  380.             {
  381.                 $this->setError($table->getError());
  382.                 return false;
  383.             }
  384.  
  385.             // Get the new item ID
  386.             $newId = $table->get('id');
  387.  
  388.             // Add the new ID to the array
  389.             $newIds[$i]    = $newId;
  390.             $i++;
  391.         }
  392.  
  393.         // Clean the cache
  394.         $this->cleanCache();
  395.  
  396.         return $newIds;
  397.     }
