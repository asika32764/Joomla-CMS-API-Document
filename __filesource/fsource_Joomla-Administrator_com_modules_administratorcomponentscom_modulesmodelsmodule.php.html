  242.             }

  246.                 return false;
  247.             }
  248.         }
  249.  
  250.         // Clean the cache

  855.     /**

  904.         $this->prepareTable($table);
  905.

  925.             return false;
  926.         }
  927.  
  928.         // Process the menu link mappings.
  929.         $assignment = isset($data['assignment']) ? $data['assignment'] : 0;
  930.  
  931.         // Delete old module to menu item associations
  932.         // $db->setQuery(
  933.         //    'DELETE FROM #__modules_menu'.
  934.         //    ' WHERE moduleid = '.(int) $table->id
  935.         // );
  936.  
  937.         $db    = $this->getDbo();

  939.             ->delete('#__modules_menu')
  940.             ->where('moduleid = ' . (int) $table->id);
  941.         $db->setQuery($query);
  942.  
  943.         try
  944.         {
  945.             $db->execute();
  946.         }
  947.         catch (RuntimeException $e)
  948.         {
  949.             $this->setError($e->getMessage());
  950.             return false;
  951.         }
  952.  
  953.         // If the assignment is numeric, then something is selected (otherwise it's none).
  954.         if (is_numeric($assignment))
  955.         {
  956.             // Variable is numeric, but could be a string.
  957.             $assignment = (int) $assignment;
  958.  
  959.             // Logic check: if no module excluded then convert to display on all.

  961.             {
