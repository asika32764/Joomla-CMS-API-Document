  170.     public function generateThumbs($thumbSizes, $creationMethod = self::SCALE_INSIDE)
  171.     {
  172.         // Make sure the resource handle is valid.
  173.         if (!$this->isLoaded())
  174.         {
  175.             throw new LogicException('No valid image was loaded.');
  176.         }
  177.  
  178.         // Accept a single thumbsize string as parameter
  179.         if (!is_array($thumbSizes))
  180.         {
  181.             $thumbSizes = array($thumbSizes);
  182.         }
  183.  
  184.         // Process thumbs
  185.         $generated = array();
  186.  
  187.         if (!empty($thumbSizes))
  188.         {
  189.             foreach ($thumbSizes as $thumbSize)
  190.             {
  191.                 // Desired thumbnail size
  192.                 $size = explode('x', strtolower($thumbSize));
  193.  
  194.                 if (count($size) != 2)
  195.                 {
  196.                     throw new InvalidArgumentException('Invalid thumb size received: ' . $thumbSize);
  197.                 }
  198.                 $thumbWidth     = $size[0];
  199.                 $thumbHeight    = $size[1];
  200.  
  201.                 switch ($creationMethod)
  202.                 {
  203.                     // Case for self::CROP
  204.                     case 4:
  205.                         $thumb = $this->crop($thumbWidth, $thumbHeight, null, null, true);
  206.                         break;
  207.  
  208.                     // Case for self::CROP_RESIZE
  209.                     case 5:
  210.                         $thumb = $this->cropResize($thumbWidth, $thumbHeight, true);
  211.                         break;
  212.  
  213.                     default:
  214.                         $thumb = $this->resize($thumbWidth, $thumbHeight, true, $creationMethod);
  215.                         break;
  216.                 }
  217.  
  218.                 // Store the thumb in the results array
  219.                 $generated[] = $thumb;
  220.             }
  221.         }
  222.  
  223.         return $generated;
  224.     }

  242.     {
  243.         // Make sure the resource handle is valid.
  244.         if (!$this->isLoaded())

  289.             }

  292.     }

  294.     /**
  295.      * Method to crop the current image.
  296.      *
  297.      * @param   mixed    $width      The width of the image section to crop in pixels or a percentage.
  298.      * @param   mixed    $height     The height of the image section to crop in pixels or a percentage.
  299.      * @param   integer  $left       The number of pixels from the left to start cropping.
  300.      * @param   integer  $top        The number of pixels from the top to start cropping.
  301.      * @param   bool     $createNew  If true the current image will be cloned, cropped and returned; else
  302.      *                                the current image will be cropped and returned.
  303.      *
  304.      * @return  JImage 
  305.      *
  306.      * @since   11.3
  307.      * @throws  LogicException
  308.      */
  309.     public function crop($width, $height, $left = null, $top = null, $createNew = true)
  310.     {
  311.         // Make sure the resource handle is valid.
  312.         if (!$this->isLoaded())
  313.         {
  314.             throw new LogicException('No valid image was loaded.');
  315.         }
  316.  
  317.         // Sanitize width.
  318.         $width = $this->sanitizeWidth($width, $height);
  319.  
  320.         // Sanitize height.
  321.         $height = $this->sanitizeHeight($height, $width);
  322.  
  323.         // Autocrop offsets
  324.         if (is_null($left))
  325.         {
  326.             $left = round(($this->getWidth() - $width) / 2);
  327.         }
  328.         if (is_null($top))
  329.         {
  330.             $top = round(($this->getHeight() - $height) / 2);
  331.         }
  332.  
  333.         // Sanitize left.
  334.         $left = $this->sanitizeOffset($left);
  335.  
  336.         // Sanitize top.
  337.         $top = $this->sanitizeOffset($top);
  338.  
  339.         // Create the new truecolor image handle.
  340.         $handle = imagecreatetruecolor($width, $height);
  341.  
  342.         // Allow transparency for the new image handle.
  343.         imagealphablending($handle, false);
  344.         imagesavealpha($handle, true);
  345.  
  346.         if ($this->isTransparent())
  347.         {
  348.             // Get the transparent color values for the current image.
  349.             $rgba = imageColorsForIndex($this->handle, imagecolortransparent($this->handle));
  350.             $color = imageColorAllocate($this->handle, $rgba['red'], $rgba['green'], $rgba['blue']);
  351.  
  352.             // Set the transparent color values for the new image.
  353.             imagecolortransparent($handle, $color);
  354.             imagefill($handle, 0, 0, $color);
  355.  
  356.             imagecopyresized($handle, $this->handle, 0, 0, $left, $top, $width, $height, $width, $height);
  357.         }
  358.         else
  359.         {
  360.             imagecopyresampled($handle, $this->handle, 0, 0, $left, $top, $width, $height, $width, $height);
  361.         }
  362.  
  363.         // If we are cropping to a new image, create a new JImage object.
  364.         if ($createNew)
  365.         {
  366.             // @codeCoverageIgnoreStart
  367.             $new = new JImage($handle);
  368.  
  369.             return $new;
  370.  
  371.             // @codeCoverageIgnoreEnd
  372.         }
  373.         // Swap out the current handle for the new image handle.
  374.         else
  375.         {
  376.             // Free the memory from the current handle
  377.             $this->destroy();
  378.  
  379.             $this->handle = $handle;
  380.  
  381.             return $this;
  382.         }
  383.     }

  550.                 }

  552.                 break;
  553.

  612.     /**
  613.      * Method to resize the current image.
  614.      *
  615.      * @param   mixed    $width        The width of the resized image in pixels or a percentage.
  616.      * @param   mixed    $height       The height of the resized image in pixels or a percentage.
  617.      * @param   bool     $createNew    If true the current image will be cloned, resized and returned; else
  618.      *                                  the current image will be resized and returned.
  619.      * @param   integer  $scaleMethod  Which method to use for scaling
  620.      *
  621.      * @return  JImage 
  622.      *
  623.      * @since   11.3
  624.      * @throws  LogicException
  625.      */
  626.     public function resize($width, $height, $createNew = true, $scaleMethod = self::SCALE_INSIDE)
  627.     {
  628.         // Make sure the resource handle is valid.
  629.         if (!$this->isLoaded())
  630.         {
  631.             throw new LogicException('No valid image was loaded.');
  632.         }
  633.  
  634.         // Sanitize width.
  635.         $width = $this->sanitizeWidth($width, $height);
  636.  
  637.         // Sanitize height.
  638.         $height = $this->sanitizeHeight($height, $width);
  639.  
  640.         // Prepare the dimensions for the resize operation.
  641.         $dimensions = $this->prepareDimensions($width, $height, $scaleMethod);
  642.  
  643.         // Create the new truecolor image handle.
  644.         $handle = imagecreatetruecolor($dimensions->width, $dimensions->height);
  645.  
  646.         // Allow transparency for the new image handle.
  647.         imagealphablending($handle, false);
  648.         imagesavealpha($handle, true);
  649.  
  650.         if ($this->isTransparent())
  651.         {
  652.             // Get the transparent color values for the current image.
  653.             $rgba = imageColorsForIndex($this->handle, imagecolortransparent($this->handle));
  654.             $color = imageColorAllocate($this->handle, $rgba['red'], $rgba['green'], $rgba['blue']);
  655.  
  656.             // Set the transparent color values for the new image.
  657.             imagecolortransparent($handle, $color);
  658.             imagefill($handle, 0, 0, $color);
  659.  
  660.             imagecopyresized($handle, $this->handle, 0, 0, 0, 0, $dimensions->width, $dimensions->height, $this->getWidth(), $this->getHeight());
  661.         }
  662.         else
  663.         {
  664.             imagecopyresampled($handle, $this->handle, 0, 0, 0, 0, $dimensions->width, $dimensions->height, $this->getWidth(), $this->getHeight());
  665.         }
  666.  
  667.         // If we are resizing to a new image, create a new JImage object.
  668.         if ($createNew)
  669.         {
  670.             // @codeCoverageIgnoreStart
  671.             $new = new JImage($handle);
  672.  
  673.             return $new;
  674.  
  675.             // @codeCoverageIgnoreEnd
  676.         }
  677.         // Swap out the current handle for the new image handle.
  678.         else
  679.         {
  680.             // Free the memory from the current handle
  681.             $this->destroy();
  682.  
  683.             $this->handle = $handle;
  684.  
  685.             return $this;
  686.         }
  687.     }

  689.     /**
  690.      * Method to crop an image after resizing it to maintain
  691.      * proportions without having to do all the set up work.
  692.      *
  693.      * @param   integer  $width      The desired width of the image in pixels or a percentage.
  694.      * @param   integer  $height     The desired height of the image in pixels or a percentage.
  695.      * @param   integer  $createNew  If true the current image will be cloned, resized, cropped and returned.
  696.      *
  697.      * @return  object  JImage Object for chaining.
  698.      *
  699.      * @since   12.3
  700.      */
  701.     public function cropResize($width, $height, $createNew = true)
  702.     {
  703.         $width   = $this->sanitizeWidth($width, $height);
  704.         $height  = $this->sanitizeHeight($height, $width);
  705.  
  706.         if (($this->getWidth() / $width) < ($this->getHeight() / $height))
  707.         {
  708.             $this->resize($width, 0, false);
  709.         }
  710.         else

  712.             $this->resize(0, $height, false);

  716.     }

  766.         else
  767.         {
  768.             // Free the memory from the current handle
  769.             $this->destroy();
  770.  
  771.             $this->handle = $handle;
  772.  
  773.             return $this;
  774.         }
  775.     }

  814.     /**

  854.     /**
  855.      * Method to get the new dimensions for a resized image.
  856.      *
  857.      * @param   integer  $width        The width of the resized image in pixels.
  858.      * @param   integer  $height       The height of the resized image in pixels.
  859.      * @param   integer  $scaleMethod  The method to use for scaling
  860.      *
  861.      * @return  stdClass 
  862.      *
  863.      * @since   11.3
  864.      * @throws  InvalidArgumentException  If width, height or both given as zero
  865.      */
  866.     protected function prepareDimensions($width, $height, $scaleMethod)
  867.     {
  868.         // Instantiate variables.
  869.         $dimensions = new stdClass;
  870.  
  871.         switch ($scaleMethod)
  872.         {
  873.             case self::SCALE_FILL:
  874.                 $dimensions->width = (int) round($width);
  875.                 $dimensions->height = (int) round($height);
  876.                 break;
  877.  
  878.             case self::SCALE_INSIDE:
  879.             case self::SCALE_OUTSIDE:
  880.                 $rx = ($width > 0) ? ($this->getWidth() / $width) : 0;
  881.                 $ry = ($height > 0) ? ($this->getHeight() / $height) : 0;
  882.  
  883.                 if ($scaleMethod == self::SCALE_INSIDE)
  884.                 {
  885.                     $ratio = ($rx > $ry) ? $rx : $ry;
  886.                 }
  887.                 else
  888.                 {
  889.                     $ratio = ($rx < $ry) ? $rx : $ry;
  890.                 }
  891.  
  892.                 $dimensions->width = (int) round($this->getWidth() / $ratio);
  893.                 $dimensions->height = (int) round($this->getHeight() / $ratio);
  894.                 break;
  895.  
  896.             default:
  897.                 throw new InvalidArgumentException('Invalid scale method.');
  898.                 break;
  899.         }
  900.  
  901.         return $dimensions;
  902.     }
  903.  
  904.     /**
  905.      * Method to sanitize a height value.
  906.      *
  907.      * @param   mixed  $height  The input height value to sanitize.
  908.      * @param   mixed  $width   The input width value for reference.
  909.      *
  910.      * @return  integer 
  911.      *
  912.      * @since   11.3
  913.      */
  914.     protected function sanitizeHeight($height, $width)
  915.     {
  916.         // If no height was given we will assume it is a square and use the width.
  917.         $height = ($height === null) ? $width : $height;
  918.  
  919.         // If we were given a percentage, calculate the integer value.
  920.         if (preg_match('/^[0-9]+(\.[0-9]+)?\%$/', $height))
  921.         {
