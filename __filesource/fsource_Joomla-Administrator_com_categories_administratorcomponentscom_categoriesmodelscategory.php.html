  454.             list($title, $alias) = $this->generateNewTitle($data['parent_id'], $data['alias'], $data['title']);

  459.  
  460.         // Bind the data.
  461.         if (!$table->bind($data))

  465.         }

  472.         }
  473.  
  474.         // Check the data.
  475.         if (!$table->check())
  476.         {

  478.             return false;

  496.         $assoc = $this->getAssoc();

  498.         {
  499.  
  500.             // Adding self to the association

  502.  
  503.             foreach ($associations as $tag => $id)
  504.             {

  506.                 {
  507.                     unset($associations[$tag]);
  508.                 }
  509.             }
  510.  
  511.             // Detecting all item menus
  512.             $all_language = $table->language == '*';
  513.  
  514.             if ($all_language && !empty($associations))
  515.             {
  516.                 JError::raiseNotice(403, JText::_('COM_CATEGORIES_ERROR_ALL_LANGUAGE_ASSOCIATED'));
  517.             }
  518.  
  519.             $associations[$table->language] = $table->id;
  520.  
  521.             // Deleting old association for these items
  522.             $db = JFactory::getDbo();
  523.             $query = $db->getQuery(true)

  534.             }
  535.  
  536.             if (!$all_language && count($associations))
  537.             {
  538.                 // Adding new association for these items
  539.                 $key = md5(json_encode($associations));
  540.                 $query->clear()
  541.                     ->insert('#__associations');
  542.  
  543.                 foreach ($associations as $id)

  552.                 {
  553.                     $this->setError($error);
  554.                     return false;
  555.                 }
  556.             }
  557.         }
  558.  
  559.         // Trigger the onContentAfterSave event.
  560.         $dispatcher->trigger($this->event_after_save, array($this->option . '.' . $this->name, &$table, $isNew));
  561.  
  562.         // Rebuild the path for the category:
  563.         if (!$table->rebuildPath($table->id))
  564.         {
  565.             $this->setError($table->getError());
  566.             return false;

  569.         // Rebuild the paths of the category's children:

  573.             return false;

  575.

  577.  
  578.         // Clear the cache
  579.         $this->cleanCache();

  581.         return true;
  582.     }

  705.     /**
  706.      * Batch copy categories to a new category.
  707.      *
  708.      * @param   integer  $value     The new category.
  709.      * @param   array    $pks       An array of row IDs.
  710.      * @param   array    $contexts  An array of item contexts.
  711.      *
  712.      * @return  mixed    An array of new IDs on success, boolean false on failure.
  713.      *
  714.      * @since   1.6
  715.      */
  716.     protected function batchCopy($value, $pks, $contexts)
  717.     {
  718.         // $value comes as {parent_id}.{extension}
  719.         $parts = explode('.', $value);
  720.         $parentId = (int) JArrayHelper::getValue($parts, 0, 1);
  721.  
  722.         $table = $this->getTable();
  723.         $db = $this->getDbo();
  724.         $user = JFactory::getUser();
  725.         $extension = JFactory::getApplication()->input->get('extension', '', 'word');
  726.         $i = 0;
  727.  
  728.         // Check that the parent exists
  729.         if ($parentId)
  730.         {
  731.             if (!$table->load($parentId))
  732.             {
  733.                 if ($error = $table->getError())
  734.                 {
  735.                     // Fatal error
  736.                     $this->setError($error);
  737.                     return false;
  738.                 }
  739.                 else
  740.                 {
  741.                     // Non-fatal error
  742.                     $this->setError(JText::_('JGLOBAL_BATCH_MOVE_PARENT_NOT_FOUND'));
  743.                     $parentId = 0;
  744.                 }
  745.             }
  746.             // Check that user has create permission for parent category
  747.             $canCreate = ($parentId == $table->getRootId()) ? $user->authorise('core.create', $extension) : $user->authorise('core.create', $extension . '.category.' . $parentId);
  748.             if (!$canCreate)
  749.             {
  750.                 // Error since user cannot create in parent category
  751.                 $this->setError(JText::_('COM_CATEGORIES_BATCH_CANNOT_CREATE'));
  752.                 return false;
  753.             }
  754.         }
  755.  
  756.         // If the parent is 0, set it to the ID of the root item in the tree
  757.         if (empty($parentId))
  758.         {
  759.             if (!$parentId = $table->getRootId())
  760.             {
  761.                 $this->setError($db->getErrorMsg());
  762.                 return false;
  763.             }
  764.             // Make sure we can create in root
  765.             elseif (!$user->authorise('core.create', $extension))
  766.             {
  767.                 $this->setError(JText::_('COM_CATEGORIES_BATCH_CANNOT_CREATE'));
  768.                 return false;
  769.             }
  770.         }
  771.  
  772.         // We need to log the parent ID
  773.         $parents = array();
  774.  
  775.         // Calculate the emergency stop count as a precaution against a runaway loop bug
  776.         $query = $db->getQuery(true)
  777.             ->select('COUNT(id)')
  778.             ->from($db->quoteName('#__categories'));
  779.         $db->setQuery($query);
  780.  
  781.         try
  782.         {
  783.             $count = $db->loadResult();
  784.         }
  785.         catch (RuntimeException $e)
  786.         {
  787.             $this->setError($e->getMessage());
  788.             return false;
  789.         }
  790.  
  791.         // Parent exists so we let's proceed
  792.         while (!empty($pks) && $count > 0)
  793.         {
  794.             // Pop the first id off the stack
  795.             $pk = array_shift($pks);
  796.  
  797.             $table->reset();
  798.  
  799.             // Check that the row actually exists
  800.             if (!$table->load($pk))
  801.             {
  802.                 if ($error = $table->getError())
  803.                 {
  804.                     // Fatal error
  805.                     $this->setError($error);
  806.                     return false;
  807.                 }
  808.                 else
  809.                 {
  810.                     // Not fatal error
  811.                     $this->setError(JText::sprintf('JGLOBAL_BATCH_MOVE_ROW_NOT_FOUND', $pk));
  812.                     continue;
  813.                 }
  814.             }
  815.  
  816.             // Copy is a bit tricky, because we also need to copy the children
  817.             $query->clear()
  818.                 ->select('id')
  819.                 ->from($db->quoteName('#__categories'))
  820.                 ->where('lft > ' . (int) $table->lft)
  821.                 ->where('rgt < ' . (int) $table->rgt);
  822.             $db->setQuery($query);
  823.             $childIds = $db->loadColumn();
  824.  
  825.             // Add child ID's to the array only if they aren't already there.
  826.             foreach ($childIds as $childId)
  827.             {
  828.                 if (!in_array($childId, $pks))
  829.                 {
  830.                     array_push($pks, $childId);
  831.                 }
  832.             }
  833.  
  834.             // Make a copy of the old ID and Parent ID
  835.             $oldId = $table->id;
  836.             $oldParentId = $table->parent_id;
  837.  
  838.             // Reset the id because we are making a copy.
  839.             $table->id = 0;
  840.  
  841.             // If we a copying children, the Old ID will turn up in the parents list
  842.             // otherwise it's a new top level item
  843.             $table->parent_id = isset($parents[$oldParentId]) ? $parents[$oldParentId] : $parentId;
  844.  
  845.             // Set the new location in the tree for the node.
  846.             $table->setLocation($table->parent_id, 'last-child');
  847.  
  848.             // TODO: Deal with ordering?
  849.             // $table->ordering    = 1;
  850.             $table->level = null;
  851.             $table->asset_id = null;
  852.             $table->lft = null;
  853.             $table->rgt = null;
  854.  
  855.             // Alter the title & alias
  856.             list($title, $alias) = $this->generateNewTitle($table->parent_id, $table->alias, $table->title);
  857.             $table->title = $title;
  858.             $table->alias = $alias;
  859.  
  860.             // Store the row.
  861.             if (!$table->store())
  862.             {
  863.                 $this->setError($table->getError());
  864.                 return false;
  865.             }
  866.  
  867.             // Get the new item ID
  868.             $newId = $table->get('id');
  869.  
  870.             // Add the new ID to the array
  871.             $newIds[$i] = $newId;
  872.             $i++;
  873.  
  874.             // Now we log the old 'parent' to the new 'parent'
  875.             $parents[$oldId] = $table->id;
  876.             $count--;
  877.         }
  878.  
  879.         // Rebuild the hierarchy.
  880.         if (!$table->rebuild())
  881.         {
  882.             $this->setError($table->getError());
  883.             return false;
  884.         }
  885.  
  886.         // Rebuild the tree path.
  887.         if (!$table->rebuildPath($table->id))
  888.         {
  889.             $this->setError($table->getError());
  890.             return false;
  891.         }
  892.  
  893.         return $newIds;
  894.     }
  895.  
  896.     /**
  897.      * Batch move categories to a new category.
  898.      *
  899.      * @param   integer  $value     The new category ID.
  900.      * @param   array    $pks       An array of row IDs.
  901.      * @param   array    $contexts  An array of item contexts.
  902.      *
  903.      * @return  boolean  True on success.
  904.      *
  905.      * @since   1.6
  906.      */
  907.     protected function batchMove($value, $pks, $contexts)
  908.     {
  909.         $parentId = (int) $value;
  910.  
  911.         $table = $this->getTable();
  912.         $db = $this->getDbo();
  913.         $query = $db->getQuery(true);
  914.         $user = JFactory::getUser();
  915.         $extension = JFactory::getApplication()->input->get('extension', '', 'word');
  916.  
  917.         // Check that the parent exists.
  918.         if ($parentId)
  919.         {
  920.             if (!$table->load($parentId))
  921.             {
  922.                 if ($error = $table->getError())
  923.                 {
  924.                     // Fatal error
  925.                     $this->setError($error);
  926.  
  927.                     return false;
  928.                 }
  929.                 else
  930.                 {
  931.                     // Non-fatal error
  932.                     $this->setError(JText::_('JGLOBAL_BATCH_MOVE_PARENT_NOT_FOUND'));
  933.                     $parentId = 0;
  934.                 }
