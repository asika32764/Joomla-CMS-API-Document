  241.     protected function index(FinderIndexerResult $item, $format = 'html')

  262.         $item->route = WeblinksHelperRoute::getWeblinkRoute($item->slug, $item->catslug);

  273.         $item->addInstruction(FinderIndexer::META_CONTEXT, 'link');
  274.         $item->addInstruction(FinderIndexer::META_CONTEXT, 'metakey');
  275.         $item->addInstruction(FinderIndexer::META_CONTEXT, 'metadesc');

  311.     /**
  312.      * Method to get the SQL query used to retrieve the list of content items.
  313.      *
  314.      * @param   mixed  $query  A JDatabaseQuery object or null.
  315.      *
  316.      * @return  JDatabaseQuery  A database object.
  317.      *
  318.      * @since   2.5
  319.      */
  320.     protected function getListQuery($query = null)
  321.     {
  322.         $db = JFactory::getDbo();
  323.         // Check if we can use the supplied SQL query.
  324.         $query = $query instanceof JDatabaseQuery ? $query : $db->getQuery(true)
  325.             ->select('a.id, a.catid, a.title, a.alias, a.url AS link, a.description AS summary')
  326.             ->select('a.metakey, a.metadesc, a.metadata, a.language, a.access, a.ordering')
  327.             ->select('a.created_by_alias, a.modified, a.modified_by')
  328.             ->select('a.publish_up AS publish_start_date, a.publish_down AS publish_end_date')
  329.             ->select('a.state AS state, a.created AS start_date, a.params')
  330.             ->select('c.title AS category, c.published AS cat_state, c.access AS cat_access');
  331.  
  332.         // Handle the alias CASE WHEN portion of the query
  333.         $case_when_item_alias = ' CASE WHEN ';
  334.         $case_when_item_alias .= $query->charLength('a.alias', '!=', '0');
  335.         $case_when_item_alias .= ' THEN ';
  336.         $a_id = $query->castAsChar('a.id');
  337.         $case_when_item_alias .= $query->concatenate(array($a_id, 'a.alias'), ':');
  338.         $case_when_item_alias .= ' ELSE ';
  339.         $case_when_item_alias .= $a_id.' END as slug';
  340.         $query->select($case_when_item_alias);
  341.  
  342.         $case_when_category_alias = ' CASE WHEN ';
  343.         $case_when_category_alias .= $query->charLength('c.alias', '!=', '0');
  344.         $case_when_category_alias .= ' THEN ';
  345.         $c_id = $query->castAsChar('c.id');
  346.         $case_when_category_alias .= $query->concatenate(array($c_id, 'c.alias'), ':');
  347.         $case_when_category_alias .= ' ELSE ';
  348.         $case_when_category_alias .= $c_id.' END as catslug';
  349.         $query->select($case_when_category_alias)
  350.  
  351.             ->from('#__weblinks AS a')
  352.             ->join('LEFT', '#__categories AS c ON c.id = a.catid');
  353.  
  354.         return $query;
  355.     }
