  306.     public function setup(SimpleXMLElement $element, $value, $group = null)
  307.     {
  308.         // Make sure there is a valid JFormField XML element.
  309.         if ((string) $element->getName() != 'field')
  310.         {
  311.             return false;
  312.         }
  313.  
  314.         // Reset the input and label values.
  315.         $this->input = null;
  316.         $this->label = null;
  317.  
  318.         // Set the XML element object.
  319.         $this->element = $element;
  320.  
  321.         // Get some important attributes from the form field element.
  322.         $class = (string) $element['class'];
  323.         $id = (string) $element['id'];
  324.         $multiple = (string) $element['multiple'];
  325.         $name = (string) $element['name'];
  326.         $required = (string) $element['required'];
  327.  
  328.         // Set the required and validation options.
  329.         $this->required = ($required == 'true' || $required == 'required' || $required == '1');
  330.         $this->validate = (string) $element['validate'];
  331.  
  332.         // Add the required class if the field is required.
  333.         if ($this->required)
  334.         {
  335.             if ($class)
  336.             {
  337.                 if (strpos($class, 'required') === false)
  338.                 {
  339.                     $this->element['class'] = $class . ' required';
  340.                 }
  341.             }
  342.             else
  343.             {
  344.                 $this->element['class'] = 'required';
  345.             }
  346.         }
  347.  
  348.         // Set the multiple values option.
  349.         $this->multiple = ($multiple == 'true' || $multiple == 'multiple');
  350.  
  351.         // Allow for field classes to force the multiple values option.
  352.         if (isset($this->forceMultiple))
  353.         {
  354.             $this->multiple = (bool) $this->forceMultiple;
  355.         }
  356.  
  357.         // Set the field description text.
  358.         $this->description = (string) $element['description'];
  359.  
  360.         // Set the visibility.
  361.         $this->hidden = ((string) $element['type'] == 'hidden' || (string) $element['hidden'] == 'true');
  362.  
  363.         // Determine whether to translate the field label and/or description.
  364.         $this->translateLabel = !((string) $this->element['translate_label'] == 'false' || (string) $this->element['translate_label'] == '0');
  365.         $this->translateDescription = !((string) $this->element['translate_description'] == 'false'
  366.             || (string) $this->element['translate_description'] == '0');
  367.  
  368.         // Set the group of the field.
  369.         $this->group = $group;
  370.  
  371.         // Set the field name and id.
  372.         $this->fieldname = $this->getFieldName($name);
  373.         $this->name = $this->getName($this->fieldname);
  374.         $this->id = $this->getId($id, $this->fieldname);
  375.  
  376.         // Set the field default value.
  377.         $this->value = $value;
  378.  
  379.         // Set the CSS class of field label
  380.         $this->labelClass = (string) $element['labelclass'];
  381.  
  382.         return true;
  383.     }

  395.     protected function getId($fieldId, $fieldName)
  396.     {
  397.         $id = '';
  398.  
  399.         // If there is a form control set for the attached form add it first.
  400.         if ($this->formControl)
  401.         {
  402.             $id .= $this->formControl;
  403.         }
  404.  
  405.         // If the field is in a group add the group control to the field id.
  406.         if ($this->group)
  407.         {
  408.             // If we already have an id segment add the group control as another level.
  409.             if ($id)
  410.             {
  411.                 $id .= '_' . str_replace('.', '_', $this->group);
  412.             }
  413.             else
  414.             {
  415.                 $id .= str_replace('.', '_', $this->group);
  416.             }
  417.         }
  418.  
  419.         // If we already have an id segment add the field id/name as another level.
  420.         if ($id)
  421.         {
  422.             $id .= '_' . ($fieldId ? $fieldId : $fieldName);
  423.         }
  424.         else
  425.         {
  426.             $id .= ($fieldId ? $fieldId : $fieldName);
  427.         }
  428.  
  429.         // Clean up any invalid characters.
  430.         $id = preg_replace('#\W#', '_', $id);
  431.  
  432.         return $id;
  433.     }

  468.     /**
  469.      * Method to get the field label markup.
  470.      *
  471.      * @return  string  The field label markup.
  472.      *
  473.      * @since   11.1
  474.      */
  475.     protected function getLabel()
  476.     {
  477.         $label = '';
  478.  
  479.         if ($this->hidden)
  480.         {
  481.             return $label;
  482.         }
  483.  
  484.         // Get the label text from the XML element, defaulting to the element name.
  485.         $text = $this->element['label'] ? (string) $this->element['label'] : (string) $this->element['name'];
  486.         $text = $this->translateLabel ? JText::_($text) : $text;
  487.  
  488.         // Build the class for the label.
  489.         $class = !empty($this->description) ? 'hasTooltip' : '';
  490.         $class = $this->required == true ? $class . ' required' : $class;
  491.         $class = !empty($this->labelClass) ? $class . ' ' . $this->labelClass : $class;
  492.  
  493.         // Add the opening label tag and main attributes attributes.
  494.         $label .= '<label id="' . $this->id . '-lbl" for="' . $this->id . '" class="' . $class . '"';
  495.  
  496.         // If a description is specified, use it to build a tooltip.
  497.         if (!empty($this->description))
  498.         {
  499.             JHtml::_('bootstrap.tooltip');
  500.             $label .= ' title="' . JHtml::tooltipText(trim($text, ':'), JText::_($this->description), 0) . '"';
