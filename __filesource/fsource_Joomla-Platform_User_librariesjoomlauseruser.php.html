  317.     /**
  318.      * Method to check JUser object authorisation against an access control
  319.      * object and optionally an access extension object
  320.      *
  321.      * @param   string  $action     The name of the action to check for permission.
  322.      * @param   string  $assetname  The name of the asset on which to perform the action.
  323.      *
  324.      * @return  boolean  True if authorised
  325.      *
  326.      * @since   11.1

  328.     public function authorise($action, $assetname = null)
  329.     {
  330.         // Make sure we only check for core.admin once during the run.
  331.         if ($this->isRoot === null)
  332.         {
  333.             $this->isRoot = false;
  334.  
  335.             // Check for the configuration file failsafe.
  336.             $config = JFactory::getConfig();
  337.             $rootUser = $config->get('root_user');
  338.  
  339.             // The root_user variable can be a numeric user ID or a username.
  340.             if (is_numeric($rootUser) && $this->id > 0 && $this->id == $rootUser)
  341.             {
  342.                 $this->isRoot = true;
  343.             }
  344.             elseif ($this->username && $this->username == $rootUser)
  345.             {
  346.                 $this->isRoot = true;
  347.             }
  348.             else
  349.             {
  350.                 // Get all groups against which the user is mapped.
  351.                 $identities = $this->getAuthorisedGroups();
  352.                 array_unshift($identities, $this->id * -1);
  353.  
  354.                 if (JAccess::getAssetRules(1)->allow('core.admin', $identities))
  355.                 {
  356.                     $this->isRoot = true;
  357.                     return true;
  358.                 }
  359.             }
  360.         }
  361.  
  362.         return $this->isRoot ? true : JAccess::check($this->id, $action, $assetname);
  363.     }

  365.     /**
  366.      * Method to return a list of all categories that a user has permission for a given action
  367.      *
  368.      * @param   string  $component  The component from which to retrieve the categories
  369.      * @param   string  $action     The name of the section within the component from which to retrieve the actions.
  370.      *
  371.      * @return  array  List of categories that this group can do this action to (empty array if none). Categories must be published.
  372.      *
  373.      * @since   11.1
  374.      */
  375.     public function getAuthorisedCategories($component, $action)
  376.     {
  377.         // Brute force method: get all published category rows for the component and check each one
  378.         // TODO: Modify the way permissions are stored in the db to allow for faster implementation and better scaling
  379.         $db = JFactory::getDbo();
  380.         $query = $db->getQuery(true)
  381.             ->select('c.id AS id, a.name AS asset_name')
  382.             ->from('#__categories AS c')
  383.             ->join('INNER', '#__assets AS a ON c.asset_id = a.id')
  384.             ->where('c.extension = ' . $db->quote($component))
  385.             ->where('c.published = 1');
  386.         $db->setQuery($query);
  387.         $allCategories = $db->loadObjectList('id');
  388.         $allowedCategories = array();
  389.         foreach ($allCategories as $category)
  390.         {
  391.             if ($this->authorise($action, $category->asset_name))
  392.             {
  393.                 $allowedCategories[] = (int) $category->id;
  394.             }
  395.         }
  396.         return $allowedCategories;
  397.     }
